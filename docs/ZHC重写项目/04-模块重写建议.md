// zhc_types.h
// C++ йҮҚеҶҷзүҲзұ»еһӢзі»з»ҹ

namespace zhc {

class Type {
public:
    enum TypeKind {
        Void, Bool, Char, Short, Int, Long, LongLong,
        Float, Double, LongDouble,
        Pointer, Array, Function, Struct, Union, Enum,
        Vector, Atomic, Complex
    };
    TypeKind getKind() const { return Kind; }
    bool isScalar() const;
    bool isInteger() const;
    bool isFloatingPoint() const;
    bool isPointer() const;
    bool isArray() const;
    bool isFunction() const;
    bool isRecord() const;
    bool isVoid() const;
    bool isBool() const;
    bool isConst() const { return Qualifiers & QUAL_CONST; }
    bool isVolatile() const { return Qualifiers & QUAL_VOLATILE; }
    bool isRestrict() const { return Qualifiers & QUAL_RESTRICT; }
    unsigned getAlign() const;
    unsigned getSize() const;
    std::string getAsString() const;
protected:
    TypeKind Kind;
    unsigned Qualifiers = 0;
    enum { QUAL_CONST = 1, QUAL_VOLATILE = 2, QUAL_RESTRICT = 4, QUAL_ATOMIC = 8 };
};

// QualType: йҷҗе®ҡзҡ„зұ»еһӢпјҲеёҰ const/volatile/restrictпјү
class QualType {
public:
    QualType(Type *T = nullptr, unsigned Q = 0) : T(T), Q(Q) {}
    Type *getType() const { return T; }
    unsigned getQualifiers() const { return Q; }
    QualType withConst() const { return QualType(T, Q | 1); }
    QualType withVolatile() const { return QualType(T, Q | 2); }
    bool isConstQualified() const { return Q & 1; }
    static QualType usualArithmeticConversions(QualType, QualType);
private:

    unsigned Q;
};
