# 第八章：新增功能——可视化执行追踪层

**版本**: v2.0 | **日期**: 2026-04-12

> **来源**: 基于 AI 接入报告.xml 的"三层价值架构"中第二层（追踪层）展开设计

---

## 8.1 功能定位

**目标**：在 AI 编程时代，通过编译时注入追踪探针 + 运行时行为可视化，让人类（学生、老师、开发者）能够完整理解任意代码的实际执行过程。

核心解决矛盾：
- **矛盾一**：AI 生成代码的能力 >> 人类理解代码的能力
- **矛盾二**：代码能运行，但没人知道它在运行时实际做了什么
- **矛盾三**：学生用 AI 写代码，老师无法评估学生是否真正理解

---

## 8.2 三层价值架构

| 层次 | 核心问题 | 实现方式 | 成熟度 |
|:---|:---|:---|:---:|
| **第一层：编译层** | 代码能否运行，有无明显错误 | Lexer/Parser/语义分析/IR优化 | ✅ 已有 |
| **第二层：追踪层** | 代码运行时实际发生了什么 | 编译时注入追踪探针，输出 JSON 执行轨迹 | 🆕 新增 |
| **第三层：解释层** | 为什么这样做，有没有问题 | AI 分析追踪轨迹，生成中文行为报告 | 🆕 新增 |

## 8.4 核心模块设计

### 8.4.1 追踪探针注入器

```cpp
// zhc_trace_probe_injector.h
#pragma once

#include "llvm/IR/PassManager.h"
#include "llvm/IR/Module.h"

namespace zhc {
namespace trace {

// 追踪探针注入 Pass
class ProbeInjectorPass : public llvm::PassInfoMixin<ProbeInjectorPass> {
public:
    explicit ProbeInjectorPass() = default;
    
    llvm::PreservedAnalyses run(llvm::Module &M, llvm::ModuleAnalysisManager &AM);
    
    // 注入的追踪函数声明（由运行时库提供）
    static void declareTraceFunctions(llvm::Module &M);
    
    // 为函数注入探针
    void instrumentFunction(llvm::Function &F);
    
    // 为基本块注入探针
    void instrumentBasicBlock(llvm::BasicBlock &BB);
    
    // 为指令注入探针
    void instrumentInstruction(llvm::Instruction &I);
    
private:
    llvm::LLVMContext *Context = nullptr;
    llvm::IRBuilder<> *Builder = nullptr;
    
    // 创建追踪调用
    llvm::CallInst *createTraceCall(const std::string &EventType,
                                   llvm::ArrayRef<llvm::Value *> Args);
    
    // 判断是否需要追踪某条指令
    bool shouldInstrument(const llvm::Instruction &I) const;
    
    // 追踪级别
    enum class TraceLevel {
        Minimal,    // 仅函数调用/返回
        Standard,   // 标准：函数+变量+分支
        Full,       // 完整：包括内存操作
    };
    TraceLevel Level = TraceLevel::Standard;
};

} // namespace trace
} // namespace zhc
```

### 8.4.2 追踪运行时库

```c
/* zhc_trace_runtime.h
 * 
 * 追踪运行时库 - 轻量级追踪探针实现
 * 
 * 设计原则：
 * 1. 零分配：使用固定大小环形缓冲区
 * 2. 无锁：单线程执行，使用原子操作
 * 3. 可关闭：非追踪模式下零开销
 */

#ifndef ZHC_TRACE_RUNTIME_H
#define ZHC_TRACE_RUNTIME_H

#include <stdint.h>
#include <stdbool.h>

// 追踪事件类型
typedef enum {
    ZHC_TRACE_EVENT_CALL,      // 函数调用
    ZHC_TRACE_EVENT_RETURN,     // 函数返回
    ZHC_TRACE_EVENT_ASSIGN,     // 变量赋值
    ZHC_TRACE_EVENT_BRANCH,    // 条件分支
    ZHC_TRACE_EVENT_LOOP,      // 循环执行
    ZHC_TRACE_EVENT_MEM_READ,  // 内存读取
    ZHC_TRACE_EVENT_MEM_WRITE, // 内存写入
    ZHC_TRACE_EVENT_ERROR,     // 错误发生
} TraceEventType;

// 追踪事件结构（紧凑设计，16 字节固定大小）
typedef struct __attribute__((packed)) {
    uint8_t  type;         // 事件类型
    uint8_t  flags;        // 标志位
    uint16_t name_len;     // 名称长度
    uint32_t line;         // 源代码行号
    uint64_t value;        // 值（可用于存储整数/指针）
    // char name[];         // 变长名称（紧跟在结构后）
} TraceEvent;

// 全局追踪状态
typedef struct {
    TraceEvent *buffer;     // 环形缓冲区
    uint32_t capacity;      // 缓冲区容量
    uint32_t head;          // 写指针
    uint32_t count;         // 事件计数
    bool enabled;           // 是否启用
    uint32_t nesting_depth; // 嵌套深度（调用栈）
} TraceState;

// 全局追踪状态（单例）
extern TraceState __zhc_trace_state;

// 初始化追踪系统
void __zhc_trace_init(uint32_t capacity);

// 关闭追踪并序列化
void __zhc_trace_flush(const char *output_path);

// 追踪函数调用
static inline void __zhc_trace_call(const char *func_name, uint64_t arg1, uint64_t arg2) {
    if (!__zhc_trace_state.enabled) return;
    
    TraceEvent *evt = &__zhc_trace_state.buffer[__zhc_trace_state.head];
    evt->type = ZHC_TRACE_EVENT_CALL;
    evt->flags = 0;
    evt->line = __builtin_LINE();
    evt->value = arg1;  // 第一个参数值
    
    __zhc_trace_state.head = (__zhc_trace_state.head + 1) % __zhc_trace_state.capacity;
    __zhc_trace_state.count++;
    __zhc_trace_state.nesting_depth++;
}

// 追踪变量赋值
static inline void __zhc_trace_assign(const char *var_name, uint64_t value) {
    if (!__zhc_trace_state.enabled) return;
    
    TraceEvent *evt = &__zhc_trace_state.buffer[__zhc_trace_state.head];
    evt->type = ZHC_TRACE_EVENT_ASSIGN;
    evt->flags = 0;
    evt->line = __builtin_LINE();
    evt->value = value;
    
    __zhc_trace_state.head = (__zhc_trace_state.head + 1) % __zhc_trace_state.capacity;
    __zhc_trace_state.count++;
}

// 追踪函数返回
static inline void __zhc_trace_return(uint64_t return_value) {
    if (!__zhc_trace_state.enabled) return;
    
    TraceEvent *evt = &__zhc_trace_state.buffer[__zhc_trace_state.head];
    evt->type = ZHC_TRACE_EVENT_RETURN;
    evt->flags = 0;
    evt->line = __builtin_LINE();
    evt->value = return_value;
    
    __zhc_trace_state.head = (__zhc_trace_state.head + 1) % __zhc_trace_state.capacity;
    __zhc_trace_state.count++;
    if (__zhc_trace_state.nesting_depth > 0) {
        __zhc_trace_state.nesting_depth--;
    }
}

// 追踪条件分支
static inline void __zhc_trace_branch(bool condition, uint64_t line) {
    if (!__zhc_trace_state.enabled) return;
    
    TraceEvent *evt = &__zhc_trace_state.buffer[__zhc_trace_state.head];
    evt->type = ZHC_TRACE_EVENT_BRANCH;
    evt->flags = condition ? 1 : 0;  // 1=true, 0=false
    evt->line = (uint32_t)line;
    evt->value = 0;
    
    __zhc_trace_state.head = (__zhc_trace_state.head + 1) % __zhc_trace_state.capacity;
    __zhc_trace_state.count++;
}

#endif // ZHC_TRACE_RUNTIME_H
```

### 8.4.3 追踪序列化器

```cpp
// zhc_trace_serializer.h
#pragma once

#include "llvm/Support/JSON.h"
#include <string>

namespace zhc {
namespace trace {

// JSON 格式的追踪事件
struct TraceEventJSON {
    std::string Event;      // 事件类型（中文）
    std::string Function;   // 函数名
    std::string Variable;   // 变量名
    uint64_t Value;         // 值
    uint32_t Line;          // 行号
    std::string Location;   // 位置描述
    std::string Condition;  // 条件（分支事件）
    uint32_t NestingDepth;  // 嵌套深度
    std::string Timestamp;  // 时间戳
};

// 完整追踪报告
struct TraceReport {
    std::string SourceFile;
    std::string CompileTime;
    std::string ExecutionTimeMs;
    uint32_t TotalEvents;
    uint32_t FunctionCalls;
    uint32_t VariableAssignments;
    uint32_t Branches;
    uint32_t MaxNestingDepth;
    std::vector<TraceEventJSON> Events;
};

// 序列化器
class TraceSerializer {
public:
    TraceSerializer();
    ~TraceSerializer();
    
    // 从原始事件生成 JSON 报告
    bool serialize(const std::string &raw_trace_path,
                   const std::string &source_file,
                   const std::string &output_json_path);
    
    // 生成 HTML 可视化页面
    bool generateHTML(const TraceReport &report,
                     const std::string &output_html_path);
    
    // 生成理解度评估报告（调用 AI 层）
    std::string generateComprehensionReport(const TraceReport &report,
                                           const std::string &model = "auto");
    
private:
    // 生成 trace.html 模板数据
    std::string generateHTMLTemplate(const TraceReport &report);
    
    // 生成 trace.json
    std::string generateJSON(const TraceReport &report);
};

} // namespace trace
} // namespace zhc
```

## 8.5 命令行接口

```bash
# 基本追踪
zhc run --trace 输入文件.zhc
# 输出: trace.json + trace.html

# 追踪级别
zhc run --trace=minimal 输入文件.zhc     # 仅函数调用
zhc run --trace=standard 输入文件.zhc    # 标准（默认）
zhc run --trace=full 输入文件.zhc       # 完整（含内存）

# 指定输出
zhc run --trace --trace-output=/path/to/output.json 输入文件.zhc
zhc run --trace --trace-html=/path/to/trace.html 输入文件.zhc

# 仅生成 HTML（不保留 JSON）
zhc run --trace --html-only 输入文件.zhc

# 浏览器打开追踪结果
zhc run --trace --open 输入文件.zhc

# 追踪 + AI 理解度评估
zhc run --trace --comprehension 输入文件.zhc
# 输出: trace.html + comprehension_report.txt

# 与 AI 解释集成
zhc run --trace --ai-explain 输入文件.zhc
# AI 对每一步执行进行中文解释
```
