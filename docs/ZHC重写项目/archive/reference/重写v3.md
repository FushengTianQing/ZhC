## 一、战略定位与愿景

### 1.1 当前定位 vs 目标定位

| 维度 | 当前 ZhC (Python) | 目标 ZhC (C++ + LLVM + AI + 多语言) |
|:---|:---|:---|
| **角色** | 独立中文C编译器 | LLVM 前端 + AI-Native 多语言编译工具链 |
| **源语言** | 仅 ZHC（中文C语法） | ZHC + 标准 C + Python |
| **编译路径** | ZHC → C 代码 → gcc | 多语言 → LLVM IR → LLVM 后端 |
| **AI 能力** | ❌ 无 | ✅ 完整 AI 编程接口 + 可信监控 |
| **优化能力** | 依赖外部编译器 | 直接获得 LLVM 200+ Pass |
| **目标平台** | 受限于 gcc/clang | 60+ LLVM Target 全覆盖 |
| **生态位** | 小众中文编译器 | LLVM 官方前端候选 + AI 编译先行者 + 多语言编译器 |
| **对标项目** | — | Clang (C/C++) + PyOxide (Python) + GitHub Copilot + Cursor AI |

### 1.2 愿景声明

> **将 ZhC 打造为 LLVM 生态中的"中文 C 前端 + 多语言编译器"，让中文程序员能用母语编写高性能系统级代码，同时支持标准 C 和 Python 代码编译。在编译器层面深度集成 AI 大模型能力，实现"AI 辅助编程 + AI 可信执行监控"的完整闭环，成为全球首个 AI-Native 编译工具链。**

#### 3.2.2 预处理重写架构

```cpp
// zhc_preprocessor.h
// 基于 LLVM 的预处理架构，与 Clang 预处理能力完全对齐

namespace zhc {
namespace preprocessor {

// 预处理符号状态
struct MacroInfo {
    llvm::StringRef Name;
    std::vector<llvm::Token> Body;        // 宏体 Token 序列
    std::vector<llvm::StringRef> Args;     // 参数列表（函数宏）
    bool IsFunctionLike;
    bool IsBuiltin;                         // 内置宏 (__FILE__, __LINE__ 等)
    bool IsObjectLike;                     // 对象宏
    SourceLocation DefinitionLoc;           // 定义位置
    bool IsEnabled = true;
};

// 预处理上下文
struct PreprocessorContext {
    llvm::StringRef SourceFile;
    std::vector<llvm::StringRef> IncludeSearchPath;  // -I 路径
    std::vector<llvm::StringRef> FrameworkSearchPath; // -F 路径
    std::unordered_map<llvm::StringRef, MacroInfo> Macros;
    std::unordered_set<llvm::StringRef> GuardMap;     // #ifndef 头文件保护
    std::vector<SourceLocation> IncludeStack;         // #include 嵌套栈
    bool InMacroArgument = false;                      // 解析宏实参
};

// 预处理引擎核心
class Preprocessor {
public:
    // 初始化
    void initialize(const PreprocessorOptions &Opts,
                   DiagnosticsEngine &Diags,
                   SourceManager &SM);
    
    // ===== 宏处理 =====
    
    // 定义宏
    void defineMacro(llvm::StringRef Name, llvm::StringRef Body);
    
    // 取消定义宏
    void undefineMacro(llvm::StringRef Name);
    
    // 获取宏信息
    const MacroInfo *getMacroInfo(llvm::StringRef Name) const;
    
    // 展开宏（支持递归）
    // 关键：实现递归展开，防止无限递归
    bool expandMacro(const llvm::Token &Tok,
                    llvm::SmallVectorImpl<llvm::Token> &Result,
                    std::unordered_set<llvm::StringRef> &ActiveMacros);
    
    // ===== 字符串化与拼接 =====
    
    // # 操作符：将实参转换为字符串
    llvm::StringRef stringifyArgument(const llvm::Token &Tok,
                                     const MacroInfo::arg_iterator &ArgBegin,
                                     const MacroInfo::arg_iterator &ArgEnd);
    
    // ## 操作符：拼接两个 Token
    llvm::Token pasteTokens(const llvm::Token &LHS, const llvm::Token &RHS);
    
    // ===== 条件编译 =====
    
    // 计算 #if 表达式
    // 支持：整数常量、defined 操作符、逻辑运算、位运算、比较运算
    llvm::APSInt evaluateConditionalExpression(llvm::StringRef Expr);
    
    // 内置宏
    // __FILE__ → 当前文件名
    // __LINE__ → 当前行号
    // __DATE__ → 编译日期 "MMM DD YYYY"
    // __TIME__ → 编译时间 "HH:MM:SS"
    // __STDC__ → 1
    // __STDC_VERSION__ → C版本
    // __COUNTER__ → 从0开始的计数器
    llvm::Optional<llvm::StringRef> getBuiltinMacroValue(llvm::StringRef Name);
    
    // ===== #include 处理 =====
    
    // 搜索并加载头文件
    // 搜索顺序：先当前目录 → -I 路径 → 系统目录
    // 支持 "path" 和 <path> 两种形式
    llvm::Optional<llvm::MemoryBufferRef> getIncludedFile(
        llvm::StringRef Filename,
        bool IsAngled,               // < > 形式
        const SourceLocation &IncludeLoc);
    
    // ===== 预编译头 =====
    
    // 生成 PCH
    void generatePCH(llvm::StringRef OutputPath);
    
    // 加载 PCH
    void loadPCH(llvm::StringRef PCHPath);
    
private:
    // 内部状态
    PreprocessorContext Context;
    
    // 宏展开引擎
    class MacroExpander {
    public:
        bool expand(const llvm::Token &Tok,
                   llvm::SmallVectorImpl<llvm::Token> &Result,
                   std::unordered_set<llvm::StringRef> &Active,
                   unsigned MaxDepth = 256);  // 防止无限递归
        
        // 检测循环展开
        bool wouldExpandToRecursiveMacro(const MacroInfo &MI,
                                        llvm::StringRef Arg);
    private:
        unsigned CurrentDepth = 0;
    };
    
    // ## 拼接处理器
    class TokenPaster {
    public:
        llvm::Token paste(const llvm::Token &LHS, const llvm::Token &RHS);
        
    private:
        // 处理标识符拼接：ABC + DEF → ABCDEF
        // 处理数字拼接：123 + 456 → 123456
        // 处理字符串拼接："a" + "b" → "ab"
        // 处理无效拼接：int + 123 → 错误
    };
    
    // 条件表达式求值器
    class ConditionalEvaluator {
    public:
        llvm::APSInt evaluate(llvm::StringRef Expr,
                             const PreprocessorContext &Ctx);
        
    private:
        llvm::APSInt parseExpression(const TokenIterator &Begin,
                                    const TokenIterator &End);
    };
};

} // namespace preprocessor
} // namespace zhc
```

#### 3.3.2 语义分析器重写架构

```cpp
// zhc_sema.h
// C++ 重写版语义分析器，完全对齐 Clang 语义分析能力

namespace zhc {
namespace sema {

// 符号表条目
struct Symbol {
    enum Kind { Function, Variable, Type, Enum, Struct, Union, Parameter, Label };
    Kind kind;
    llvm::StringRef Name;
    QualType Type;                           // 符号类型
    Declaration *Decl = nullptr;              // 声明节点
    Scope *DefinedIn = nullptr;               // 所属作用域
    bool IsDefined = false;                  // 是否已定义
    bool IsReferenced = false;               // 是否被引用

## 七、新增功能：AI 大模型可信执行监控层

### 7.1 功能定位

**目标**：在 AI 生成代码执行前、中、后三个阶段，通过静态分析和运行时监控，确保 AI 生成的代码不会执行危险操作，防止 AI 幻觉导致的安全事故。

### 7.2 为什么需要可信执行监控

| 风险类型 | 说明 | 示例 | 危害级别 |
|:---|:---|:---|:---:|
| **内存安全** | AI 可能生成不安全的内存操作 | `ptr[out_of_bounds]` | 🔴 高 |
| **系统调用滥用** | AI 可能生成危险的系统调用 | `system("rm -rf /")` | 🔴 高 |
| **数据泄露** | AI 可能意外暴露敏感数据 | 打印密码变量 | 🟠 中 |
| **无限循环** | AI 可能生成死循环 | `while(1)` 无退出条件 | 🟠 中 |
| **资源耗尽** | AI 可能生成内存泄漏 | 无界递归 | 🟠 中 |
| **权限提升** | AI 可能尝试绕过安全检查 | 修改安全标志 | 🔴 高 |
| **外部数据注入** | AI 可能引入恶意输入 | 未验证的用户输入直接使用 | 🔴 高 |
| **AI 幻觉代码** | AI 幻觉出不存在的行为 | 声称执行成功实际未执行 | 🟠 中 |

```cpp
// zhc_ai_monitor.h
#pragma once

#include "llvm/ADT/SmallVector.h"
#include "llvm/ADT/StringRef.h"
#include "llvm/Support/Error.h"
#include "llvm/Support/JSON.h"

namespace zhc {
namespace monitor {

// 告警级别
enum class AlertLevel {
    Info,       // 信息性提示
    Warning,    // 警告（可能有问题）
    Error,      // 错误（阻止执行）
    Critical    // 严重（立即终止）
};

// 告警类别
enum class AlertCategory {
    // 安全类
    SystemCallAbuse,       // 系统调用滥用
    PrivilegeEscalation,    // 权限提升尝试
    FileSystemManipulation, // 文件系统操作
    NetworkAccess,          // 网络访问
    ProcessManipulation,    // 进程操作
    ShellInjection,         // Shell 注入
    
    // 内存安全类
    BufferOverflow,         // 缓冲区溢出
    UseAfterFree,          // 使用已释放内存
    DoubleFree,            // 重复释放
    MemoryLeak,            // 内存泄漏
    UninitializedAccess,    // 访问未初始化内存
    
    // 资源类
    InfiniteLoop,           // 无限循环
    UnboundedAllocation,    // 无界内存分配
    UnboundedRecursion,    // 无界递归
    ResourceExhaustion,     // 资源耗尽
    
    // AI 专项类
    AIHallucination,       // AI 幻觉代码
    DangerousPattern,       // 危险模式
    UnverifiedAssumption,  // 未验证的假设
    OverconfidentCode,     // 过度自信的代码
    
    // 数据类
    SensitiveDataAccess,   // 敏感数据访问
    DataLeak,              // 数据泄露
    CredentialsExposure,    // 凭据暴露
    
    // 代码质量类
    UndefinedBehavior,     // 未定义行为
    ImplicitConversion,    // 隐式类型转换
    SignedOverflow,       // 有符号整数溢出
};

// 告警记录
struct Alert {
    AlertLevel Level;
    AlertCategory Category;
    std::string Message;
    std::string CodeSnippet;
    std::string FileName;
    unsigned Line;
    unsigned Column;
    
    // AI 相关
    bool IsAIGenerated;
    std::string AIModel;
    double Confidence;  // AI 生成代码的可信度 (0.0-1.0)
    
    // 上下文
    std::string CallStack;
    llvm::SmallVector<std::pair<StringRef, StringRef>, 8> LocalVariables;
    
    // 建议
    llvm::SmallVector<StringRef, 4> SuggestedFixes;
    StringRef SeverityExplanation;
    
    // 时间戳
    uint64_t Timestamp;
};

// 监控配置

#### 8.2.3 C AST → ZHC AST 桥接器

**核心设计**：C Parser 将 C 代码解析为中间 AST，然后通过桥接器转换为 ZHC 编译器的通用 AST 节点，后续流水线（语义分析 → IR → LLVM）完全复用。

```python
# src/zhc_c/ast_bridge.py
# C AST → ZHC 通用 AST 桥接器

# C 类型名 → ZHC 类型名的映射
C_TYPE_TO_ZHC = {
    "int": "整数型", "char": "字符型",
    "float": "浮点型", "double": "双精度浮点型",
    "void": "空型", "long": "长整数型",
    "short": "短整数型", "signed": "有符号整数型",
    "unsigned": "无符号整数型",
    "_Bool": "逻辑型",
    "FILE": "文件型", "FILE*": "文件型*",
    # stdint.h 类型映射
    "int8_t": "整数型", "int16_t": "整数型",
    "int32_t": "长整数型", "int64_t": "长长整数型",
    "uint8_t": "无符号整数型", "size_t": "无符号长整数型",
}

class CAstBridge:
    """C AST → ZHC 通用 AST 桥接器
    
    设计原则：
    - 输入：C Parser 生成的中间 AST
    - 输出：与 ZHC Parser 输出一致的 AST 节点
    - 后续流水线（语义分析、IR生成、后端）完全无感知
    """

    def transform(self, c_ast) -> ProgramNode:
        """执行 C AST → ZHC AST 转换"""
        declarations = []
        
        for decl in c_ast.declarations:
            if isinstance(decl, FunctionDecl):
                declarations.append(self._transform_function(decl))
            elif isinstance(decl, GlobalVar):
                declarations.append(self._transform_global_variable(decl))
            elif isinstance(decl, (StructDecl, UnionDecl, EnumDecl)):
                declarations.append(self._transform_type_decl(decl))
        
        return ProgramNode(declarations=declarations)

    def _transform_function(self, c_func) -> FunctionDeclNode:
        """函数声明转换

        C:  int add(int a, int b) { return a + b; }
        ZHC: 整数型 add(整数型 a, 整数型 b) {
                  返回 a + b;
              }
        """
        return_type = self._transform_type(c_func.return_type)
        params = [self._transform_param(p) for p in c_func.params]
        body = self._transform_statement(c_func.body)

        return FunctionDeclNode(
            name=c_func.name,
            return_type=return_type,
            params=params,
            body=body,
        )

    def _transform_type(self, c_type) -> ASTNode:
        """C 类型 → ZHC 类型"""
        if isinstance(c_type, PointerType):
            base = self._transform_type(c_type.base_type)
            return PointerTypeNode(base)
        elif isinstance(c_type, ArrayType):
            base = self._transform_type(c_type.base_type)
            return ArrayTypeNode(base, c_type.size)
        elif isinstance(c_type, StructType):
            return StructTypeNode(c_type.name)
        else:
            zhc_name = C_TYPE_TO_ZHC.get(c_type.name, c_type.name)
            return PrimitiveTypeNode(zhc_name)

    def _transform_statement(self, c_stmt) -> ASTNode:
        """C 语句 → ZHC 语句"""
        if isinstance(c_stmt, CompoundStmt):
            stmts = [self._transform_statement(s) for s in c_stmt
