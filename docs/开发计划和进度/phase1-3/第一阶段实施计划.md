> 版本：v1.0 | 创建日期：2026-04-02

---

## ⚠️ 重要说明

**技术方案已升级**：从「修改 Clang 核心」改为「预处理器方案」

| 原方案 | 问题 | 新方案 | 优势 |
|:---|:---|:---|:---|
| 修改 Clang 源码 | 风险高、编译慢、破坏性 | 预处理器转换 | 安全、快速、可回退 |

**预处理器方案原理**：
```
中文 C (.zhc)  →  预处理器  →  标准 C (.c)  →  系统 Clang 编译  →  可执行文件
```

## 一、技术方案

### 1.1 架构设计

```
┌─────────────────────────────────────────────────────────────┐
│                   用户源码 (.zhc)                          │
│                                                              │
│  中文整数型 主函数() {                                      │
│      中文如果 (x > 0) {                                    │
│          中文返回 x;                                        │
│      }                                                     │
│      中文返回 0;                                           │
│  }                                                         │
└──────────────────────┬────────────────────────────────────┘
                       │
                       ▼
┌─────────────────────────────────────────────────────────────┐
│                   预处理器 (zhcpp)                         │
│                                                              │
│  ① 关键字替换                                              │
│     中文如果 → if                                          │
│     中文整数型 → int                                        │
│                                                              │
│  ② 标识符保留                                              │
│     主函数、x → 不变                                        │
│                                                              │
│  ③ 字符串/注释 → 原样保留                                  │
│     "中文如果" → "中文如果"（不变）                         │
└──────────────────────┬────────────────────────────────────┘
                       │
                       ▼
┌─────────────────────────────────────────────────────────────┐
│                   标准 C 代码 (.c)                          │
│                                                              │
│  int main() {                                              │
│      if (x > 0) {                                          │
│          return x;                                          │
│      }                                                     │
│      return 0;                                             │
│  }                                                         │
└──────────────────────┬────────────────────────────────────┘
                       │
                       ▼
┌─────────────────────────────────────────────────────────────┐
│                   系统 Clang 编译器                         │
└──────────────────────┬────────────────────────────────────┘
                       │
                       ▼
                   可执行文件
```

### 1.2 为什么这个方案更稳

| 检查项 | 原方案（改 Clang） | 新方案（预处理器） |
|:---|:---|:---|
| 破坏源码 | ❌ 可能 | ✅ 不破坏 |
| 编译时间 | 2-4小时 | 0（用现成 Clang） |
| 回退难度 | 高（需恢复源码） | 极低（删预处理器即可） |
| 调试复杂度 | 极高（Clang 源码） | 低（独立工具） |
| 测试难度 | 高 | 低 |

### 阶段 B：开发预处理器（Day 2-7）

#### B.1 预处理器核心实现

```python
#!/usr/bin/env python3
"""
中文 C 预处理器 (zhcpp)

功能：
1. 关键字替换（中文关键字 → 标准 C 关键字）
2. 标识符保留（变量名/函数名不变）
3. 字符串/注释保护（内容不变）
4. 预处理器指令支持（#define → #中文定义）
"""

import re
import sys
from pathlib import Path

class 中文C预处理器:
    """中文 C 预处理器核心类"""

    # 关键字映射表
    关键字映射 = {
        # 类型关键字
        '中文无类型': 'void',
        '中文整数型': 'int',
        '中文长整数型': 'long',
        '中文短整数型': 'short',
        '中文字符型': 'char',
        '中文浮点型': 'float',
        '中文双精度浮点型': 'double',
        '中文逻辑型': '_Bool',

        # 类型修饰符
        '中文有符号': 'signed',
        '中文无符号': 'unsigned',
        '中文常量': 'const',
        '中文静态': 'static',
        '中文寄存器': 'register',
        '中文外部': 'extern',
        '中文自动': 'auto',
        '中文内联': 'inline',

        # 流程控制
        '中文如果': 'if',
        '中文否则': 'else',
        '中文判断': 'while',
        '中文循环': 'for',
        '中文执行': 'do',
        '中文选择': 'switch',
        '中文分支': 'case',
        '中文默认': 'default',
        '中文跳出': 'break',
        '中文继续': 'continue',
        '中文返回': 'return',
        '中文去向': 'goto',

        # 结构体
        '中文结构体': 'struct',
        '中文共用体': 'union',
        '中文别名': 'typedef',
        '中文枚举': 'enum',
        '中文获取大小': 'sizeof',
    }

    def __init__(self):
        self.字符串模式 = re.compile(r'"(?:[^"\\]|\\.)*"')
        self.字符模式 = re.compile(r"'(?:[^'\\]|\\.)*'")
        self.注释单行模式 = re.compile(r'//.*?$', re.MULTILINE)
        self.注释多行模式 = re.compile(r'/\*.*?\*/', re.DOTALL)
        self.标识符模式 = re.compile(r'\b([\u4e00-\u9fff]+)\b')

    def 处理(self, 源码: str) -> str:
        """处理中文 C 源码"""
        # 步骤 1：保护字符串和字符常量
        字符串保护 = {}
        源码 = self._保护字符串(源码, 字符串保护)

        # 步骤 2：保护注释
        源码 = self._保护注释(源码)

        # 步骤 3：关键字替换
        源码 = self._替换关键字(源码)

        # 步骤 4：恢复字符串（不做替换）
        源码 = self._恢复字符串(源码)

        return 源码

    def _保护字符串(self, 源码, 保护字典):
        """保护字符串常量"""
        def 保存(match):
            键 = f'__字符串保护_{len(保护字典)}__'
            保护字典[键] = match.group(0)
            return 键
        return self.字符串模式.sub(保存, 源码)

    def _保护注释(self, 源码):
        """保护注释（不做替换）"""
        # 单行注释
        源码 = self.注释单行模式.sub(lambda m: f'__注释_{hash(m.group())}__', 源码)
        # 多行注释
        源码 = self.注释多行模式.sub(lambda m: f'__注释_{hash(m.group())}__', 源码)
        return 源码

    def _替换关键字(self, 源码):
        """替换中文关键字为标准 C 关键字"""
        for 中文关键字, 标准关键字 in sorted(self.关键字映射.items(), key=lambda x: len(x[0]), reverse=True):
            # 使用单词边界确保完整匹配
            模式 = rf'\b{re.escape(中文关键字)}\b'
            源码 = re.sub(模式, 标准关键字, 源码)
        return 源码

    def _恢复字符串(self, 源码):
        """恢复注释为原文"""
        # 简单处理：注释替换为原文（这里需要实现完整的保护-恢复机制）
        return 源码

def main():
    if len(sys.argv) < 2:
        print("用法: zhcpp <输入文件.zhc>")
        sys.exit(1)

    输入文件 = Path(sys.argv[1])
    输出文件 = 输入文件.with_suffix('.c')

    预处理器 = 中文C预处理器()
    源码 = 输入文件.read_text(encoding='utf-8')
    结果 = 预处理器.处理(源码)

    输出文件.write_text(结果, encoding='utf-8')
    print(f"✅ 转换成功: {输入文件} → {输出文件}")

if __name__ == '__main__':
    main()
```
