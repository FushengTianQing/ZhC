### 代码生成示例

```c
/* 运算符重载: 向量 */
向量_operator_add(向量_t *obj1, 向量 *obj2);
向量_operator_sub(向量_t *obj1, 向量 *obj2);
向量_operator_mul(向量_t *obj1, double *obj2);

/* 实现 */
向量_t* 向量_operator_add(向量_t *v1, 向量_t *v2) {
    向量_t *result = (向量_t*)malloc(sizeof(向量_t));
    result->x = v1->x + v2->x;
    result->y = v1->y + v2->y;
    return result;
}
```
