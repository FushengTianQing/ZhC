def _infer_expr_type(self, node: ASTNode):
    if node.node_type == ASTNodeType.CALL_EXPR:
        func_name = node.name
        
        # 检查是否是泛型函数
        if self.generic_manager.is_generic_function(func_name):
            # 从实参推导类型参数
            arg_types = [self._infer_expr_type(arg) for arg in node.args]
            type_args = self._infer_generic_args(func_name, arg_types)
            
            # 实例化
            instance = self.instantiator.instantiate_function(
                self.generic_manager.get_generic_functions(func_name)[0],
                type_args
            )
            
            # 注册实例化函数
            self._register_function_instance(instance)
            
            return instance.specialized_return_type
    
    # ... 原有逻辑

def _instantiate_body(self, body: ASTNode, type_mapping: Dict[str, str], context) -> ASTNode:
