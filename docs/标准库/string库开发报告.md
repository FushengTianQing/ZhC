## 📝 使用示例

### 基本使用
```c
#include "zhc_string.h"
#include "zhc_stdio.h"

int main() {
    // 字符串长度
    int len = zhc_strlen("Hello, World!");  // 13
    
    // 字符串连接
    char buf[100] = "Hello, ";
    zhc_strcat(buf, "World!");  // "Hello, World!"
    
    // 提取子串
    char *sub = zhc_substr("Hello, World!", 0, 5);  // "Hello"
    free(sub);
    
    // 查找子串
    int pos = zhc_find("Hello, World!", "World");  // 7
    
    // 替换子串
    char *replaced = zhc_replace("Hello, World!", "World", "中文C");
    // "Hello, 中文C!"
    free(replaced);
    
    // 大小写转换
    char str[] = "Hello";
    zhc_tolower(str);  // "hello"
    zhc_toupper(str);  // "HELLO"
    
    // 去除空白
    char str2[] = "  Hello  ";
    zhc_trim(str2);  // "Hello"
    
    return 0;
}
```
