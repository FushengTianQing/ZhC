## 1. 需求分析

### 1.1 当前状态

ZhC 编译器已有基础数组支持：
- 数组声明和初始化
- 数组下标访问
- 基础数组类型推导

**现有问题**：
1. 缺少数组边界检查
2. 数组类型推导不够精确
3. 多维数组支持有限
4. 变长数组（VLA）支持不完整
5. 数组作为函数参数处理不完善

### 1.2 目标状态

实现完整的数组类型增强：

```zhc
整数型 主函数() {
    // 静态数组
    整数型 arr1[10];
    整数型 arr2[] = [1, 2, 3, 4, 5];
    
    // 类型推导
    自动 arr3 = [1, 2, 3];        // 推导为 整数型[3]
    
    // 边界检查（可选，运行时）
    arr1[10] = 42;              // 运行时错误: 数组越界
    arr1[-1] = 42;              // 运行时错误: 负数索引
    
    // 多维数组
    整数型 matrix[3][4];
    matrix[1][2] = 10;
    
    // 动态数组（变长数组）
    整数型 n = 10;
    整数型 dyn_arr[n];           // C99 VLA 支持
    
    返回 0;
}

// 数组作为参数
整数型 求和(整数型 arr[], 整数型 长度) {
    整数型 总和 = 0;
    对于 (整数型 i = 0; i < 长度; i = i + 1) {
        总和 = 总和 + arr[i];
    }
    返回 总和;
}
```

#### 2.2.1 数组类型定义

```python
@dataclass
class ArrayTypeInfo:
    """数组类型信息"""
    
    element_type: str            # 元素类型
    dimensions: List[int]        # 各维度大小（None 表示未指定）
    is_dynamic: bool = False     # 是否为变长数组
    is_pointer: bool = False     # 是否退化为指针
    total_size: Optional[int] = None  # 总大小
    
    @property
    def rank(self) -> int:
        """数组维度"""
        return len(self.dimensions)
    
    @property
    def is_complete(self) -> bool:
        """是否完整类型（所有维度已知）"""
        return all(d is not None for d in self.dimensions)
    
    def get_element_type(self) -> str:
        """获取元素类型（递归）"""
        if len(self.dimensions) == 1:
            return self.element_type
        
        inner = ArrayTypeInfo(
            element_type=self.element_type,
            dimensions=self.dimensions[1:]
        )
        return str(inner)
    
    def to_string(self) -> str:
        """转换为字符串表示"""
        dims = "".join(
            f"[{d if d else ''}]" 
            for d in self.dimensions
        )
        return f"{self.element_type}{dims}"


class ArrayTypeFactory:
    """数组类型工厂"""
    
    @staticmethod
    def create(
        element_type: str,
        *dimensions: int
    ) -> ArrayTypeInfo:
        """创建静态数组类型"""
        return ArrayTypeInfo(
            element_type=element_type,
            dimensions=list(dimensions),
            is_dynamic=False
        )
    
    @staticmethod
    def create_dynamic(
        element_type: str,
        dimensions: List[Optional[int]]
    ) -> ArrayTypeInfo:
        """创建动态数组类型"""
        return ArrayTypeInfo(
            element_type=element_type,
            dimensions=dimensions,
            is_dynamic=True
        )
    
    @staticmethod
    def create_incomplete(element_type: str) -> ArrayTypeInfo:
        """创建不完整数组类型（用于函数参数）"""
        return ArrayTypeInfo(
            element_type=element_type,
            dimensions=[None],  # 大小未知
            is_pointer=True
        )
```

#### 2.2.2 数组边界检查器

```python
class ArrayBoundsChecker:
    """数组边界检查"""
    
    def __init__(self):
        self.array_sizes: Dict[str, ArrayTypeInfo] = {}
    
    def register_array(
        self,
        name: str,
        array_type: ArrayTypeInfo
    ) -> None:
        """注册数组大小信息"""
        self.array_sizes[name] = array_type
    
    def check_access(
        self,
        array_name: str,
        index_expr: ASTNode,
        index_type: str,
        context: AnalysisContext
    ) -> Optional[BoundsError]:
        """
        检查数组访问
        
        Returns:
            如果越界，返回 BoundsError；否则返回 None
        """
        
        if array_name not in self.array_sizes:
            return None  # 未知数组，无法检查
        
        array_info = self.array_sizes[array_name]
        
        # 检查索引类型
        if index_type not in ("整数型", "字符型"):
            return BoundsError(
                message="数组索引必须为整数类型",
                severity="warning"
            )
        
        # 静态分析：检查常量索引
        if isinstance(index_expr, IntLiteralNode):
            index_value = index_expr.value
            
            # 检查负数索引
            if index_value < 0:
                return BoundsError(
                    message=f"数组索引不能为负数: {index_value}",
                    location=self._get_location(index_expr),
                    severity="error"
                )
            
            # 检查上界
            dim_size = array_info.dimensions[0]
            if dim_size is not None and index_value >= dim_size:
                return BoundsError(
                    message=f"数组越界: 索引 {index_value} 超出范围 [0, {dim_size-1}]",
                    location=self._get_location(index_expr),
                    severity="error"
                )
        
        # 运行时检查：生成边界检查代码
        return self._generate_runtime_check(array_name, index_expr, array_info)
    
    def _generate_runtime_check(
        self,
        array_name: str,
        index_expr: ASTNode,
        array_info: ArrayTypeInfo
    ) -> Optional[ASTNode]:
        """生成运行时边界检查"""
        
        if array_info.is_complete:
            # 静态大小，生成静态检查
            dim_size = array_info.dimensions[0]
            
            check = IfStmtNode(
                condition=BinaryExprNode(
                    left=index_expr,
                    operator=">=",
                    right=IntLiteralNode(value=dim_size)
                ),
                then_branch=BlockStmtNode(statements=[
                    ExprStmtNode(
                        value=CallExprNode(
                            callee="运行时错误",
                            args=[StringLiteralNode(
                                value=f"数组 '{array_name}' 越界访问"
                            )]
                        )
                    )
                ])
            )
            
            return check
        
        return None  # 动态大小，在运行时检查
```

#### 2.2.3 数组类型推导器

```python
class ArrayTypeInferrer:
    """数组类型推导"""
    
    def infer_array_type(
        self,
        node: ArrayLiteralNode,
        context: InferenceContext
    ) -> ArrayTypeInfo:
        """
        从数组字面量推导类型
        
        Example:
            [1, 2, 3]      -> 整数型[3]
            [1.0, 2.0]     -> 浮点型[2]
            ["a", "b"]     -> 字符串型[2]
        """
        
        if not node.elements:
            # 空数组，默认为整数型
            return ArrayTypeInfo(
                element_type="整数型",
                dimensions=[0]
            )
        
        # 推导元素类型
        element_types = []
        for elem in node.elements:
            elem_type = self.engine.infer(elem, context.env)
            element_types.append(elem_type)
        
        # 统一元素类型（取最通用类型）
        unified_type = self._unify_types(element_types)
        
        return ArrayTypeInfo(
            element_type=unified_type,
            dimensions=[len(node.elements)]
        )
    
    def infer_multidim_type(
        self,
        node: ArrayLiteralNode,
        context: InferenceContext
    ) -> ArrayTypeInfo:
        """推导多维数组类型"""
        
        if not node.elements:
            return ArrayTypeInfo(
                element_type="整数型",
                dimensions=[0]
            )
        
        # 递归推导每个子数组
        sub_types = []
        for sub_array in node.elements:
            if isinstance(sub_array, ArrayLiteralNode):
                sub_type = self.infer_multidim_type(sub_array, context)
                sub_types.append(sub_type)
            else:
                # 第一行，单独处理
                elem_type = self.engine.infer(sub_array, context.env)
                sub_types.append(ArrayTypeInfo(
                    element_type=elem_type,
                    dimensions=[1]
                ))
        
        # 统一子数组类型
        unified_sub = self._unify_array_types(sub_types)
        
        # 添加当前维度
        return ArrayTypeInfo(
            element_type=unified_sub.element_type,
            dimensions=[len(node.elements)] + unified_sub.dimensions
        )
    
    def infer_subscript_type(
        self,
        array_type: ArrayTypeInfo,
        indices: List[ASTNode],
        context: InferenceContext
    ) -> str:
        """
        推导下标访问后的类型
        
        Example:
            整数型[3][4] 下标访问一次 -> 整数型[4]
            整数型[3] 下标访问一次 -> 整数型
        """
        
        if len(indices) > len(array_type.dimensions):
            raise SemanticError(
                f"数组维度错误: 访问了 {len(indices)} 维，但数组只有 {len(array_type.dimensions)} 维"
            )
        
        remaining_dims = array_type.dimensions[len(indices):]
        
        if not remaining_dims:
            # 完全下标访问，返回元素类型
            return array_type.element_type
        
        # 部分下标访问，返回子数组
        return ArrayTypeInfo(
            element_type=array_type.element_type,
            dimensions=remaining_dims
        ).to_string()
    
    def _unify_types(self, types: List[str]) -> str:
        """统一类型列表（类型提升）"""
        
        # 类型层次
        type_hierarchy = {
            "布尔型": 0,
            "字符型": 1,
            "整数型": 2,
            "浮点型": 3,
            "字符串型": 4,
        }
        
        max_level = 0
        result_type = "整数型"
        
        for t in types:
            level = type_hierarchy.get(t, -1)
            if level > max_level:
                max_level = level
                result_type = t
        
        return result_type
```

#### 2.2.4 数组语义分析器

```python
class ArraySemanticAnalyzer:
    """数组语义分析"""
    
    def __init__(self, bounds_checker: ArrayBoundsChecker):
        self.bounds_checker = bounds_checker
    
    def analyze_array_decl(
        self,
        node: VariableDeclNode,
        context: SemanticContext
    ) -> ArrayTypeInfo:
        """分析数组声明"""
        
        array_type = self._parse_array_type(node.var_type)
        
        # 检查大小
        if node.initializer:
            # 从初始化推导大小
            if isinstance(node.initializer, ArrayLiteralNode):
                inferred_type = self.inferrer.infer_array_type(
                    node.initializer,
                    context
                )
                
                # 合并类型信息
                if array_type.dimensions[0] is None:
                    array_type.dimensions[0] = inferred_type.dimensions[0]
                    array_type.element_type = inferred_type.element_type
                
                # 检查类型一致性
                if array_type.element_type != inferred_type.element_type:
                    raise SemanticError(
                        f"数组元素类型不匹配: 声明为 '{array_type.element_type}', "
                        f"但初始化为 '{inferred_type.element_type}'"
                    )
        
        # 检查 VLA
        if array_type.is_dynamic:
            if not context.options.support_vla:
                raise SemanticError(
                    "变长数组(VLA)需要启用 VLA 支持",
                    suggestion="使用 --enable-vla 启用"
                )
        
        # 注册到边界检查器
        self.bounds_checker.register_array(node.name, array_type)
        
        return array_type
    
    def analyze_subscript(
        self,
        node: SubscriptExprNode,
        context: SemanticContext
    ) -> str:
        """分析下标访问表达式"""
        
        # 分析数组
        array_type_str = self.analyzer.analyze(node.array, context)
        array_type = self._parse_array_type(array_type_str)
        
        # 分析索引
        index_type = self.analyzer.analyze(node.index, context)
        
        # 检查索引类型
        if index_type not in ("整数型", "字符型"):
            raise SemanticError(
                f"数组索引必须为整数类型，实际为 '{index_type}'",
                location=SourceLocation(line=node.line, column=node.column)
            )
        
        # 检查边界（如果可能）
        bounds_error = self.bounds_checker.check_access(
            self._get_array_name(node.array),
            node.index,
            index_type,
            context
        )
        
        if bounds_error:
            context.errors.add(bounds_error)
        
        # 推导结果类型
        result_type = self.inferrer.infer_subscript_type(
            array_type,
            [node.index],
            context
        )
        
        node.inferred_type = result_type
        return result_type
    
    def analyze_array_param(
        self,
        node: ParameterNode,
        context: SemanticContext
    ) -> ArrayTypeInfo:
        """分析数组参数（退化为指针）"""
        
        # 数组参数自动退化为指针
        array_type = self._parse_array_type(node.param_type)
        array_type.is_pointer = True
        
        return array_type
```

### 2.3 IR 生成支持

```python
class IRGenerator:
    """IR 生成器 - 数组支持"""
    
    def visit_array_literal(
        self,
        node: ArrayLiteralNode
    ) -> IRValue:
        """生成数组字面量 IR"""
        
        # 推导数组类型
        array_type = self.inferrer.infer_array_type(node, self.context)
        
        # 分配内存
        size = self._calculate_array_size(array_type)
        ir_array = self.builder.alloca(
            array_type.element_type,
            size,
            name=f"array_literal_{node.id}"
        )
        
        # 初始化元素
        for i, elem in enumerate(node.elements):
            elem_ir = self.visit(elem)
            elem_ptr = self.builder.gep(ir_array, [i])
            self.builder.store(elem_ir, elem_ptr)
        
        return ir_array
    
    def visit_subscript(
        self,
        node: SubscriptExprNode
    ) -> IRValue:
        """生成下标访问 IR"""
        
        # 获取数组指针
        array_ptr = self.visit(node.array)
        
        # 获取索引
        index = self.visit(node.index)
        
        # 生成边界检查（如果启用）
        if self.options.enable_bounds_check:
            bounds_check = self._generate_bounds_check(
                array_ptr, index, node
            )
            self.builder.add_instruction(bounds_check)
        
        # GEP 获取元素指针
        elem_ptr = self.builder.gep(array_ptr, [index])
        
        # 加载元素值
        return self.builder.load(elem_ptr)
    
    def _calculate_array_size(self, array_type: ArrayTypeInfo) -> int:
        """计算数组总大小（字节）"""
        
        elem_size = self.type_size(array_type.element_type)
        
        total = elem_size
        for dim in array_type.dimensions:
            total *= dim
        
        return total
```
