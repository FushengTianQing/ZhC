| 字段 | 值 |

| **优先级** | P2 |

## 1. 开发内容分析

### 1.1 目标概述

实现从 LLVM IR 到目标文件 (.o/.exe) 的直接生成功能，替代当前依赖外部 LLVM 工具链的做法，提升编译器的独立性和构建速度。

### 1.2 技术背景

#### 传统编译流程
```
源文件 → 前端 → IR → Pass → asm → 链接器 → 可执行文件
                              ↑
                        依赖外部 llc/ld
```

#### 目标架构
```
源文件 → 前端 → IR → Pass → 直接生成 .o/.exe
                              ↑
                        内置代码生成器
```

#### 目标文件格式
| 平台 | 格式 | 工具链 |
|------|------|--------|
| Linux x64 | ELF + MachO | GCC/Clang |
| macOS | MachO | Apple clang |
| Windows | PE/COFF | MSVC/MinGW |
| WebAssembly | Wasm | Emscripten |

### 1.3 需求分析

#### 核心需求
1. **直接对象文件生成**：生成标准 .o 文件，无需外部汇编器
2. **多目标支持**：x86_64, AArch64, ARM, RISC-V, WebAssembly
3. **链接信息生成**：生成符号表、重定位信息
4. **ABI 遵循**：完整的函数调用约定实现

#### 用户场景
```bash
# 单文件编译
zhc input.zhc -o output.o

# 多文件编译 + 链接
zhc file1.zhc -c -o file1.o
zhc file2.zhc -c -o file2.o
ld file1.o file2.o -o program

# 直接生成可执行文件
zhc main.zhc -o program
```

### 2.3 核心接口设计

#### CodeGenerator 类
```python
class CodeGenerator:
    """目标代码生成器"""

    def __init__(self, target: Target):
        self.target = target
        self.module = None
        self.mc_context = None
        self.asm_printer = None

    def generate(self, module: ll.Module, output_path: str,
                 file_type: FileType = FileType.OBJECT) -> bytes:
        """生成目标文件

        Args:
            module: LLVM 模块
            output_path: 输出路径
            file_type: 输出文件类型 (OBJECT, EXECUTABLE, LIBRARY)

        Returns:
            生成的文件内容
        """
        self.module = module
        self._initialize_context()
        self._run_isel_and_emit(output_path, file_type)

    def _initialize_context(self):
        """初始化汇编上下文"""
        self.mc_context = MCContext(
            target=self.target,
            tm=TargetMachine.get_target(self.target)
        )
        self.asm_printer = AsmPrinter(self.mc_context)

    def _run_isel_and_emit(self, output_path, file_type):
        """执行指令选择并输出"""
        for func in self.module.functions:
            lowered_func = self.target_lower.lower_function(func)
            selected = self.instruction_selector.select(lowered_func)
            allocated = self.register_allocator.allocate(selected)
            self.asm_printer.emit_function(allocated)
```

class TargetRegistry:
    """目标平台注册表"""

    _targets: Dict[str, Target] = {}

    @classmethod
    def register(cls, target: Target):
        cls._targets[target.name] = target
        cls._targets[target.triple] = target

    @classmethod
    def get(cls, name_or_triple: str) -> Target:
        if name_or_triple in cls._targets:
            return cls._targets[name_or_triple]
        raise ValueError(f"Unknown target: {name_or_triple}")

    @classmethod
    def available_targets(cls) -> List[Target]:
        return list(cls._targets.values())

# src/zhc/codegen/instruction_selector.py
class InstructionSelector:
    """指令选择器 - 从 DAG 选择目标指令"""

    def __init__(self, target: Target):
        self.target = target
        self.isel = self._create_target_isel()

    def select(self, dag: SelectionDAG) -> List[MachineInstr]:
        """选择目标指令"""
        for node in dag.nodes:
            if node.opcode == ISD.SELECT:
                # 根据条件选择最优指令
                if self.target.has_feature('cmov'):
                    self._select_cmov(node)
                else:
                    self._select_branch(node)
            elif node.opcode == ISD.ADD:
                self._select_add(node)
            # ...

    def _select_add(self, node: SDNode):
        """选择加法指令"""
        if node.value_type == MVT.i32:
            return self._emit(MCInst(MCOperand.create ImmReg(0)))
        elif node.value_type == MVT.i64:
            # x86_64: 使用 REX 前缀
            return self._emit_64bit_add(node)

```python
# src/zhc/codegen/x86_codegen.py
class X86InstructionSelector(InstructionSelector):
    """x86_64 专用指令选择"""

    def select_call(self, node: SDNode):
        """选择调用指令"""
        # 系统 V AMD64 ABI:
        # 参数: rdi, rsi, rdx, rcx, r8, r9, 然后栈
        # 返回值: rax, rdx
        call = MCInst(X86Insts.CALL)
        call.add_operand(MCOperand.createImm(call_target))
        return self._emit(call)

    def select_prologue(self, func: MachineFunction):
        """生成函数序言"""
        frame = func.frame
        insts = []

        # push rbp
        insts.append(MCInst(X86Insts.PUSH, [MCReg.RBP]))

        # mov rbp, rsp
        insts.append(MCInst(X86Insts.MOV, [MCReg.RBP, MCReg.RSP]))

        # 分配栈帧
        if frame.local_size > 0:
            insts.append(MCInst(X86Insts.SUB,
                [MCReg.RSP, MCImm(frame.local_size)]))

        # 保存 callee-saved 寄存器
        for reg in frame.saved_regs:
            offset = frame.get_spill_offset(reg)
            insts.append(MCInst(X86Insts.MOV,
                [MCMem(offset, MCReg.RBP), reg]))

        return insts

    def select_epilogue(self, func: MachineFunction):
        """生成函数尾声"""
        frame = func.frame
        insts = []

        # 恢复 callee-saved 寄存器
        for reg in reversed(frame.saved_regs):
            offset = frame.get_spill_offset(reg)
            insts.append(MCInst(X86Insts.MOV,
                [reg, MCMem(offset, MCReg.RBP)]))

        # 释放栈帧
        if frame.local_size > 0:
            insts.append(MCInst(X86Insts.ADD,
                [MCReg.RSP, MCImm(frame.local_size)]))

        # pop rbp; ret
        insts.append(MCInst(X86Insts.POP, [MCReg.RBP]))
        insts.append(MCInst(X86Insts.RET))

        return insts
```

# src/zhc/codegen/register_allocator.py
class LinearScanRegisterAllocator:
    """线性扫描寄存器分配器"""

    def __init__(self, target: Target):
        self.target = target
        self.allocated = {}      # vreg -> preg
        self.spill_slots = {}   # vreg -> stack_offset
        self.active = []        # 活跃区间

    def allocate(self, function: MachineFunction) -> MachineFunction:
        """分配寄存器"""
        self._calculate_live_intervals(function)

        for interval in sorted(function.intervals, key=lambda x: x.start):
            self._allocate_interval(interval, function)

        return function

    def _allocate_interval(self, interval, function):
        """为单个区间分配"""
        self._expire_old_intervals(interval)

        # 尝试分配空闲寄存器
        for preg in self._get_allocatable_regs(interval.reg_class):
            if not self._is_reg_busy(preg):
                self.allocated[interval.vreg] = preg
                interval.reg = preg
                self.active.append((interval.end, preg))
                return

        # 需要溢出
        self._spill_at_interval(interval)
```

# src/zhc/codegen/frame_lower.py
class FrameLowering:
    """栈帧 lowering"""

    def __init__(self, target: Target):
        self.target = target

    def lower_frame(self, function: MachineFunction):
        """Lower 栈帧信息"""
        frame = function.frame
        frame.align = self._compute_frame_alignment(function)
        frame.local_size = self._compute_local_size(function)
        frame.outgoing_size = self._compute_outgoing_size(function)

        # 分配溢出槽
        for vreg in frame.vregs_to_spill:
            frame.allocate_spill_slot(vreg, size=self._get_reg_size(vreg))

    def _compute_frame_alignment(self, function) -> int:
        """计算栈帧对齐"""
        max_align = self.target.stack_alignment
        for local in function.locals:
            max_align = max(max_align, local.alignment)
        return max_align
```

# src/zhc/codegen/elf_object_writer.py
class ELFObjectWriter(ObjectWriter):
    """ELF 目标文件写入器"""

    def write(self, mc_context: MCContext, output_path: str):
        """生成 ELF 文件"""
        sections = self._layout_sections(mc_context)

        elf_header = ELFHeader(
            e_ident=self._make_elf_ident(),
            e_type=ET.REL,  # 可重定位对象
            e_machine=self._get_elf_machine(),
            e_entry=0,
            e_phoff=0,
            e_shoff=self._compute_shstrtab_offset(),
            e_flags=0,
            e_ehsize=ELFHeader.SIZE,
            e_phentsize=0,
            e_phnum=0,
            e_shentsize=SectionHeader.SIZE,
            e_shnum=len(sections) + 1,  # +1 for null section
            e_shstrndx=len(sections),
        )

        with open(output_path, 'wb') as f:
            # 写入 ELF 头
            f.write(elf_header.pack())

            # 写入节内容
            for section in sections:
                f.write(section.content)

            # 写入节头表
            for section in sections:
                f.write(section.header.pack())

    def _make_elf_ident(self) -> bytes:
        """创建 ELF 标识"""
        return bytearray([
            0x7f, ord('E'), ord('L'), ord('F'),  # magic
            2,       # 64-bit
            1,       # little-endian
            1,       # ELF version
            0,       # OS/ABI
            0, 0, 0, 0, 0, 0, 0, 0  # padding
        ])
```

class MachOObjectWriter(ObjectWriter):
    """MachO 目标文件写入器"""

    def write(self, mc_context: MCContext, output_path: str):
        """生成 MachO 文件"""
        segments = self._layout_segments(mc_context)

        # MachO 头
        header = MachHeader(
            magic=MACHO_MAGIC_64,
            cputype=self._get_cpu_type(),
            cpusubtype=CPU_SUBTYPE_ALL,
            filetype=MH_OBJECT,
            ncmds=len(segments),
            sizeofcmds=self._compute_cmd_size(segments),
            flags=0,
            reserved=0,
        )

        with open(output_path, 'wb') as f:
            f.write(header.pack())

            # 写入 segment 命令和内容
            for seg in segments:
                f.write(seg.cmd.pack())
                f.write(seg.content)

# src/zhc/codegen/wasm_codegen.py
class WasmObjectWriter(ObjectWriter):
    """WebAssembly 目标文件写入器"""

    MAGIC = b'\\x00asm'
    VERSION = 1

    def write(self, mc_context: MCContext, output_path: str):
        """生成 WebAssembly 二进制"""
        wasm_module = self._build_wasm_module(mc_context)

        with open(output_path, 'wb') as f:
            f.write(self.MAGIC)
            f.write(self._uleb128(self.VERSION))
            self._write_sections(f, wasm_module)

    def _write_sections(self, f, module):
        """写入各个 section"""
        for section in module.sections:
            f.write(self._uleb128(section.id))
            f.write(self._encode_section(section))
```

## 6. 风险与挑战

### 6.1 技术风险
| 风险 | 影响 | 缓解策略 |
|------|------|----------|
| 目标文件格式复杂 | 高 | 参考 LLVM MC 层实现 |
| ABI 实现错误 | 高 | 严格测试 ABI 合规性 |
| 寄存器分配效率 | 中 | 实现多种分配策略 |
| 跨平台测试困难 | 中 | 使用 Docker/QEMU 进行测试 |

### 6.2 已知限制
1. **初期仅支持有限目标**：x86_64, AArch64, WebAssembly
2. **调试信息需单独生成**：与 DWARF 功能配合
3. **链接器依赖**：初期仍需外部链接器
