> **优先级**: P5
> **功能模块**: 函数式
> **功能名称**: 闭包支持
> **创建日期**: 2026-04-10
> **状态**: 规划中

## 1. 功能概述

实现闭包（Closure）和 lambda 表达式支持，允许函数捕获并访问其定义环境中的变量，支持函数式编程范式。

### 1.1 目标

- 支持 lambda 表达式语法
- 支持闭包捕获外部变量
- 支持闭包作为函数参数和返回值
- 支持闭包的内存管理（upvalue）

### 1.2 语法设计

```zhc
// Lambda 表达式
函数型(整数型, 整数型) -> 整数型 加法 = (整数型 a, 整数型 b) -> a + b;

// 闭包捕获外部变量
函数型(整数型) -> 整数型 创建计数器() {
    整数型 计数 = 0;
    
    // 返回一个闭包，捕获计数变量
    返回 () -> {
        计数 = 计数 + 1;
        返回 计数;
    };
}

// 使用
整数型 计数器 = 创建计数器();
打印(计数器());  // 1
打印(计数器());  // 2

// 高阶函数
函数型(整数型, 整数型) -> 整数型 应用(函数型(整数型, 整数型) -> 整数型 f, 整数型 x, 整数型 y) {
    返回 f(x, y);
}

打印(应用(加法, 3, 4));  // 7
```

### 3.2 核心数据结构

```python
# src/zhc/functional/closure.py

from dataclasses import dataclass, field
from typing import List, Optional, Set
from enum import Enum

class CaptureMode(Enum):
    """捕获模式"""
    REFERENCE = "reference"   # 引用捕获（可修改）
    VALUE = "value"           # 值捕获（不可修改）
    CONST_REF = "const_ref"   # 常量引用

@dataclass
class Upvalue:
    """Upvalue 定义"""
    name: str                    # 变量名
    type_name: str               # 类型
    mode: CaptureMode            # 捕获模式
    index: int                   # 在环境中的索引
    is_mutable: bool = False     # 是否可变

@dataclass
class ClosureType:
    """闭包类型"""
    name: str                        # 类型名称
    param_types: List[str]           # 参数类型
    return_type: str                 # 返回类型
    upvalues: List[Upvalue]          # 捕获的变量
    environment_size: int            # 环境大小
    
    def get_signature(self) -> str:
        """获取函数签名"""
        params = ", ".join(self.param_types)
        return f"({params}) -> {self.return_type}"

@dataclass
class ClosureEnvironment:
    """闭包环境"""
    upvalues: List[Upvalue]
    values: dict  # 运行时值
    
    def get(self, name: str):
        """获取 upvalue 值"""
        for uv in self.upvalues:
            if uv.name == name:
                return self.values.get(name)
        return None
    
    def set(self, name: str, value):
        """设置 upvalue 值"""
        for uv in self.upvalues:
            if uv.name == name and uv.is_mutable:
                self.values[name] = value
                return
        raise ValueError(f"Cannot modify {name}")
```

### 3.3 Upvalue 分析器

```python
# src/zhc/functional/analyzer.py

class UpvalueAnalyzer:
    """Upvalue 分析器"""
    
    def __init__(self):
        self.captured_vars: Set[str] = set()
        self.local_vars: Set[str] = set()
    
    def analyze(self, lambda_expr, outer_scope_vars: Set[str]) -> List[Upvalue]:
        """分析 lambda 表达式捕获的变量"""
        self.captured_vars = set()
        self.local_vars = set()
        
        # 分析 lambda 体中的变量引用
        self._analyze_expression(lambda_expr.body)
        
        # 找出需要捕获的外部变量
        upvalues = []
        for var in self.captured_vars:
            if var in outer_scope_vars and var not in self.local_vars:
                mode = self._determine_capture_mode(var, lambda_expr)
                upvalues.append(Upvalue(
                    name=var,
                    type_name=self._get_var_type(var),
                    mode=mode,
                    index=len(upvalues),
                    is_mutable=(mode == CaptureMode.REFERENCE),
                ))
        
        return upvalues
    
    def _analyze_expression(self, expr):
        """递归分析表达式中的变量引用"""
        if isinstance(expr, Identifier):
            if expr.name not in self.local_vars:
                self.captured_vars.add(expr.name)
        elif isinstance(expr, Assignment):
            self._analyze_expression(expr.value)
        elif isinstance(expr, BinaryOp):
            self._analyze_expression(expr.left)
            self._analyze_expression(expr.right)
        # ... 其他表达式类型
    
    def _determine_capture_mode(self, var_name: str, lambda_expr) -> CaptureMode:
        """确定捕获模式"""
        # 如果变量在 lambda 中被修改，使用引用捕获
        if self._is_modified_in_lambda(var_name, lambda_expr):
            return CaptureMode.REFERENCE
        # 否则使用值捕获
        return CaptureMode.VALUE
    
    def _is_modified_in_lambda(self, var_name: str, lambda_expr) -> bool:
        """检查变量是否在 lambda 中被修改"""
        # 检查是否有赋值语句
        for stmt in self._collect_statements(lambda_expr.body):
            if isinstance(stmt, Assignment) and stmt.target.name == var_name:
                return True
        return False
```

typedef struct ZhcUpvalue {

    ZhcUpvalue* upvalues;
    int upvalue_count;

    void* func_ptr,
    int upvalue_count,
    ...
