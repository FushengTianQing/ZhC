# P5-类型系统-复数类型

> **优先级**: P5
> **功能模块**: 类型系统
> **功能名称**: 复数类型 _Complex
> **创建日期**: 2026-04-10
> **状态**: 规划中

## 1. 功能概述

实现 C99 标准的复数类型支持，包括 `float _Complex`、`double _Complex`、`long double _Complex`，以及复数的算术运算和数学函数。

### 1.1 目标

- 支持 C99 复数类型语法
- 支持复数算术运算（加、减、乘、除）
- 支持复数数学函数（sqrt、exp、log、sin、cos 等）
- 支持实部/虚部访问

### 1.2 语法设计

```zhc
// 复数类型声明
浮点数复数 z1 = 3.0 + 4.0i;
双精度复数 z2 = 1.0 - 2.0i;
长双精度复数 z3 = 2.5 + 1.5I;

// 复数运算
双精度复数 和 = z1 + z2;
双精度复数 差 = z1 - z2;
双精度复数 积 = z1 * z2;
双精度复数 商 = z1 / z2;

// 实部和虚部
双精度 实部 = 复数_实部(z1);  // 或 z1._M_re
双精度 虚部 = 复数_虚部(z1);  // 或 z1._M_im

// 复数函数
双精度复数 平方根 = 复数_平方根(z1);
双精度复数 指数 = 复数_指数(z1);
双精度复数 对数 = 复数_对数(z1);
双精度复数 正弦 = 复数_正弦(z1);
双精度复数 余弦 = 复数_余弦(z1);

// 极坐标转换
双精度 模 = 复数_模(z1);      // |z|
双精度 幅角 = 复数_幅角(z1);  // arg(z)

// 共轭复数
双精度复数 共轭 = 复数_共轭(z1);
```

### 3.2 核心数据结构

```python
# src/zhc/type_system/complex.py

from dataclasses import dataclass
from enum import Enum
from typing import Union

class ComplexElementType(Enum):
    """复数元素类型"""
    FLOAT = "float"
    DOUBLE = "double"
    LONG_DOUBLE = "long double"

@dataclass
class ComplexType:
    """复数类型"""
    element_type: ComplexElementType
    
    @property
    def name(self) -> str:
        return f"{self.element_type.value} _Complex"
    
    @property
    def size(self) -> int:
        """复数大小（字节）"""
        sizes = {
            ComplexElementType.FLOAT: 8,      # 2 * 4
            ComplexElementType.DOUBLE: 16,    # 2 * 8
            ComplexElementType.LONG_DOUBLE: 32, # 2 * 16
        }
        return sizes[self.element_type]
    
    @property
    def alignment(self) -> int:
        """对齐要求"""
        return self.size // 2

@dataclass
class ComplexValue:
    """复数值"""
    real: float
    imag: float
    element_type: ComplexElementType = ComplexElementType.DOUBLE
    
    def __add__(self, other: 'ComplexValue') -> 'ComplexValue':
        return ComplexValue(
            self.real + other.real,
            self.imag + other.imag,
            self.element_type,
        )
    
    def __sub__(self, other: 'ComplexValue') -> 'ComplexValue':
        return ComplexValue(
            self.real - other.real,
            self.imag - other.imag,
            self.element_type,
        )
    
    def __mul__(self, other: 'ComplexValue') -> 'ComplexValue':
        # (a+bi)(c+di) = (ac-bd) + (ad+bc)i
        return ComplexValue(
            self.real * other.real - self.imag * other.imag,
            self.real * other.imag + self.imag * other.real,
            self.element_type,
        )
    
    def __truediv__(self, other: 'ComplexValue') -> 'ComplexValue':
        # (a+bi)/(c+di) = [(ac+bd) + (bc-ad)i] / (c²+d²)
        denom = other.real ** 2 + other.imag ** 2
        return ComplexValue(
            (self.real * other.real + self.imag * other.imag) / denom,
            (self.imag * other.real - self.real * other.imag) / denom,
            self.element_type,
        )
    
    def conjugate(self) -> 'ComplexValue':
        """共轭复数"""
        return ComplexValue(self.real, -self.imag, self.element_type)
    
    def magnitude(self) -> float:
        """模"""
        return (self.real ** 2 + self.imag ** 2) ** 0.5
    
    def phase(self) -> float:
        """幅角"""
        import math
        return math.atan2(self.imag, self.real)
```

### 3.3 复数数学函数

```python
# src/zhc/type_system/complex_math.py

import math
from typing import Union
from .complex import ComplexValue, ComplexElementType

def complex_sqrt(z: ComplexValue) -> ComplexValue:
    """复数平方根"""
    r = z.magnitude()
    theta = z.phase()
    sqrt_r = math.sqrt(r)
    return ComplexValue(
        sqrt_r * math.cos(theta / 2),
        sqrt_r * math.sin(theta / 2),
        z.element_type,
    )

def complex_exp(z: ComplexValue) -> ComplexValue:
    """复数指数 e^z"""
    exp_real = math.exp(z.real)
    return ComplexValue(
        exp_real * math.cos(z.imag),
        exp_real * math.sin(z.imag),
        z.element_type,
    )

def complex_log(z: ComplexValue) -> ComplexValue:
    """复数对数 ln(z)"""
    return ComplexValue(
        math.log(z.magnitude()),
        z.phase(),
        z.element_type,
    )

def complex_sin(z: ComplexValue) -> ComplexValue:
    """复数正弦"""
    return ComplexValue(
        math.sin(z.real) * math.cosh(z.imag),
        math.cos(z.real) * math.sinh(z.imag),
        z.element_type,
    )

def complex_cos(z: ComplexValue) -> ComplexValue:
    """复数余弦"""
    return ComplexValue(
        math.cos(z.real) * math.cosh(z.imag),
        -math.sin(z.real) * math.sinh(z.imag),
        z.element_type,
    )

def complex_tan(z: ComplexValue) -> ComplexValue:
    """复数正切"""
    cos_z = complex_cos(z)
    sin_z = complex_sin(z)
    return sin_z / cos_z

def complex_polar(r: float, theta: float, 
                  etype: ComplexElementType = ComplexElementType.DOUBLE) -> ComplexValue:
    """极坐标转直角坐标"""
    return ComplexValue(
        r * math.cos(theta),
        r * math.sin(theta),
        etype,
    )
```

### 3.4 IR 设计

```python
# src/zhc/ir/nodes/complex.py

class IRComplexType(IRType):
    """复数类型 IR"""
    element_type: IRType
    
    @property
    def llvm_type(self):
        # LLVM 中复数用两个元素的 struct 表示
        return ir.StructType([self.element_type, self.element_type])

class IRComplexLiteral(IRNode):
    """复数字面量 IR"""
    real: float
    imag: float
    complex_type: ComplexType

class IRComplexOp(IRNode):
    """复数运算 IR"""
    op: str  # add, sub, mul, div
    left: IRNode
    right: IRNode
    result_type: ComplexType
```
