## 1. 功能概述

实现在运行时检查对象类型、判断类型兼容性、执行类型断言，支持多态和动态类型验证。

### 1.1 目标

- 支持类型兼容性检查
- 支持类型断言（cast）
- 支持接口实现检查
- 支持继承关系检查

### 1.2 语法设计

```zhc
// 类型检查
类 动物 {
    函数型 无 -> 无 叫声;
}

类 狗 继承 动物 {
    函数 叫() {
        打印("汪汪!");
    }
}

类 猫 继承 动物 {
    函数 叫() {
        打印("喵喵!");
    }
}

// 类型兼容性检查
动物 a = 新 狗();
如果 (是类型(狗, a)) {
    打印("a 是狗");
}

// 类型断言
狗 d = 转换为[狗](a);  // 或 (狗) a
如果 (d != 空) {
    d.叫();
}

// 接口检查
接口 可飞行 {
    函数型 无 -> 无 飞;
}

类 鸟 继承 可飞行 {
    函数 飞() {
        打印("鸟在飞");
    }
}

如果 (实现接口(可飞行, 对象)) {
    可飞行 f = 转换为[可飞行](对象);
    f.飞();
}

// 类型相等检查
如果 (类型相同(对象1, 对象2)) {
    打印("相同类型");
}

// 类型名称获取
字符串 类型名 = 类型名称(对象);
打印(类型名);  // "狗"
```

### 3.2 核心数据结构

```python
# src/zhc/reflection/type_check.py

from dataclasses import dataclass, field
from typing import List, Optional, Set, Any, TypeVar, Generic

T = TypeVar('T')

@dataclass
class TypeHierarchy:
    """类型层次结构"""
    _parents: dict = field(default_factory=dict)  # type_name -> parent_name
    _children: dict = field(default_factory=dict)  # type_name -> [child_names]
    _interfaces: dict = field(default_factory=dict)  # type_name -> [interface_names]
    
    @classmethod
    def instance(cls) -> 'TypeHierarchy':
        """获取单例"""
        if not hasattr(cls, '_instance'):
            cls._instance = cls()
        return cls._instance
    
    def register_parent(self, type_name: str, parent_name: str) -> None:
        """注册父类关系"""
        self._parents[type_name] = parent_name
        if parent_name not in self._children:
            self._children[parent_name] = []
        self._children[parent_name].append(type_name)
    
    def register_interface(self, type_name: str, interface_name: str) -> None:
        """注册接口实现"""
        if type_name not in self._interfaces:
            self._interfaces[type_name] = []
        self._interfaces[type_name].append(interface_name)
    
    def get_parent(self, type_name: str) -> Optional[str]:
        """获取父类"""
        return self._parents.get(type_name)
    
    def get_children(self, type_name: str) -> List[str]:
        """获取所有子类"""
        return self._children.get(type_name, [])
    
    def get_interfaces(self, type_name: str) -> List[str]:
        """获取实现的接口"""
        return self._interfaces.get(type_name, [])
    
    def is_a(self, subtype: str, supertype: str) -> bool:
        """检查 subtype 是否是 supertype 的子类型"""
        if subtype == supertype:
            return True
        
        # 检查父类链
        current = subtype
        while current:
            parent = self._parents.get(current)
            if parent == supertype:
                return True
            current = parent
        
        # 检查接口
        interfaces = self._interfaces.get(subtype, [])
        for iface in interfaces:
            if self.is_a(iface, supertype):
                return True
        
        return False

class TypeChecker:
    """类型检查器"""
    
    def __init__(self):
        self.hierarchy = TypeHierarchy.instance()
    
    def is_type(self, obj: Any, type_name: str) -> bool:
        """检查对象是否为指定类型"""
        obj_type = type(obj).__name__
        return self.hierarchy.is_a(obj_type, type_name)
    
    def is_subtype(self, subtype: str, supertype: str) -> bool:
        """检查子类型关系"""
        return self.hierarchy.is_a(subtype, supertype)
    
    def implements_interface(self, type_name: str, interface_name: str) -> bool:
        """检查是否实现接口"""
        return self.hierarchy.is_a(type_name, interface_name)
    
    def get_common_base(self, type1: str, type2: str) -> Optional[str]:
        """获取两个类型的公共基类"""
        # 收集 type1 的所有祖先
        ancestors1 = self._get_ancestors(type1)
        
        # 检查 type2 的祖先是否在 type1 的祖先中
        current = type2
        while current:
            if current in ancestors1:
                return current
            current = self.hierarchy.get_parent(current)
        
        return None
    
    def _get_ancestors(self, type_name: str) -> Set[str]:
        """获取所有祖先类型"""
        ancestors = set()
        current = type_name
        while current:
            ancestors.add(current)
            current = self.hierarchy.get_parent(current)
        return ancestors

class TypeCast:
    """类型转换"""
    
    def __init__(self):
        self.checker = TypeChecker()
    
    def safe_cast(self, obj: Any, target_type: str) -> Optional[Any]:
        """安全类型转换"""
        if self.checker.is_type(obj, target_type):
            return obj
        return None
    
    def dynamic_cast(self, obj: Any, target_type: str) -> Optional[Any]:
        """动态类型转换（运行时检查）"""
        if obj is None:
            return None
        if self.checker.is_type(obj, target_type):
            return obj
        raise TypeError(f"无法将 {type(obj).__name__} 转换为 {target_type}")
    
    def require_type(self, obj: Any, type_name: str) -> Any:
        """要求对象为指定类型，否则抛出异常"""
        if not self.checker.is_type(obj, type_name):
            raise TypeError(f"期望类型 {type_name}，实际为 {type(obj).__name__}")
        return obj
    
    def cast_to_interface(self, obj: Any, interface_name: str) -> Optional[Any]:
        """转换为接口类型"""
        if self.checker.implements_interface(type(obj).__name__, interface_name):
            return obj
        return None

# 全局实例
_type_checker = TypeChecker()
_type_cast = TypeCast()

def is_type(obj: Any, type_name: str) -> bool:
    """检查对象类型"""
    return _type_checker.is_type(obj, type_name)

def is_subtype(subtype: str, supertype: str) -> bool:
    """检查子类型关系"""
    return _type_checker.is_subtype(subtype, supertype)

def safe_cast(obj: Any, target_type: str) -> Optional[Any]:
    """安全类型转换"""
    return _type_cast.safe_cast(obj, target_type)

def dynamic_cast(obj: Any, target_type: str) -> Any:
    """动态类型转换"""
    return _type_cast.dynamic_cast(obj, target_type)

def type_name(obj: Any) -> str:
    """获取类型名称"""
    return type(obj).__name__

def type_equals(obj1: Any, obj2: Any) -> bool:
    """检查类型是否相同"""
    return type(obj1) is type(obj2)
```
