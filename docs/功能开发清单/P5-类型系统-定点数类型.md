# P5-类型系统-定点数类型

> **优先级**: P5
> **功能模块**: 类型系统
> **功能名称**: 定点数类型
> **创建日期**: 2026-04-10
> **状态**: 规划中

## 1. 功能概述

实现 C99 定点数类型支持，包括 `_Fract`（小数）和 `_Accum`（累加器）类型，用于嵌入式系统和数字信号处理中的定点运算。

### 1.1 目标

- 支持 `_Fract` 小数类型
- 支持 `_Accum` 累加器类型
- 支持定点数算术运算
- 支持定点数与整数的转换

### 1.2 语法设计

```zhc
// _Fract 类型（小数）
短定点小数 f1 = 0.5;        // _Fract
长定点小数 f2 = 0.25;       // _Long_Fract
无符号定点小数 uf1 = 0.75;   // _Fract_U

// _Accum 类型（累加器）
短定点累加 a1 = 100;         // _Accum
长定点累加 a2 = 1000;        // _Long_Accum
无符号定点累加 ua1 = 200;    // _Accum_U

// 定点数运算
短定点累加 结果 = a1 + a2;
短定点小数 商 = f1 * f2;

// 定点数转换
整数型 整数值 = (整数型) f1;  // 定点数转整数
短定点小数 从整数 = (短定点小数) 5;  // 整数转定点数

// 定点数移位（模拟小数位移）
短定点小数 缩放 = f1 << 8;  // 定点数乘以 2^8
```

### 3.2 核心数据结构

```python
# src/zhc/type_system/fixed_point.py

from dataclasses import dataclass
from enum import Enum
from typing import Union

class FixedPointFormat(Enum):
    """定点数格式"""
    # _Fract (小数)
    FRACT_HALF = ("_Fract", 1, 7, True)      # 1.7 格式
    FRACT = ("_Fract", 1, 15, True)           # 1.15 格式
    LONG_FRACT = ("_Long_Fract", 1, 31, True) # 1.31 格式
    
    # _Accum (累加器)
    ACCUM_SHORT = ("_Accum", 8, 8, True)      # 8.8 格式
    ACCUM = ("_Accum", 16, 16, True)          # 16.16 格式
    LONG_ACCUM = ("_Long_Accum", 32, 32, True) # 32.32 格式
    
    # 无符号版本
    FRACT_U = ("_Fract_U", 0, 8, False)       # 0.8 格式
    ACCUM_U = ("_Accum_U", 8, 8, False)       # 8.8 无符号

@dataclass
class FixedPointType:
    """定点数类型"""
    name: str
    total_bits: int
    int_bits: int
    frac_bits: int
    is_signed: bool
    
    @classmethod
    def from_format(cls, format: FixedPointFormat) -> 'FixedPointType':
        """从格式创建类型"""
        name, int_bits, frac_bits, is_signed = format.value
        total_bits = int_bits + frac_bits
        return cls(name, total_bits, int_bits, frac_bits, is_signed)
    
    @classmethod
    def fract(cls, signed: bool = True) -> 'FixedPointType':
        """_Fract 类型"""
        if signed:
            return cls.from_format(FixedPointFormat.FRACT)
        return cls.from_format(FixedPointFormat.FRACT_U)
    
    @classmethod
    def accum(cls, signed: bool = True) -> 'FixedPointType':
        """_Accum 类型"""
        if signed:
            return cls.from_format(FixedPointFormat.ACCUM)
        return cls.from_format(FixedPointFormat.ACCUM_U)
    
    @property
    def scale_factor(self) -> float:
        """缩放因子：2^frac_bits"""
        return 2 ** self.frac_bits
    
    @property
    def min_value(self) -> float:
        """最小值"""
        if self.is_signed:
            return -2 ** (self.int_bits - 1)
        return 0
    
    @property
    def max_value(self) -> float:
        """最大值"""
        if self.is_signed:
            return 2 ** (self.int_bits - 1) - (1 / self.scale_factor)
        return 2 ** self.total_bits - (1 / self.scale_factor)
    
    @property
    def lsb(self) -> float:
        """最小精度单位"""
        return 1 / self.scale_factor

@dataclass
class FixedPointValue:
    """定点数值"""
    raw: int                    # 原始整数值
    ftype: FixedPointType       # 类型
    
    @classmethod
    def from_float(cls, value: float, ftype: FixedPointType) -> 'FixedPointValue':
        """从浮点数创建定点数"""
        scaled = round(value * ftype.scale_factor)
        # 限制范围
        max_val = 2 ** ftype.total_bits - 1
        min_val = 0 if not ftype.is_signed else -(2 ** (ftype.total_bits - 1))
        scaled = max(min_val, min(max_val, scaled))
        return cls(scaled, ftype)
    
    def to_float(self) -> float:
        """转换为浮点数"""
        return self.raw / self.scale_factor
    
    def to_int(self) -> int:
        """转换为整数（截断小数部分）"""
        if self.ftype.is_signed:
            return self.raw >> self.ftype.frac_bits
        return self.raw >> self.ftype.frac_bits
    
    def __add__(self, other: 'FixedPointValue') -> 'FixedPointValue':
        """加法"""
        # 对齐小数位
        result = self._align_frac_bits(other)
        return FixedPointValue(self.raw + result.raw, self.ftype)
    
    def __sub__(self, other: 'FixedPointValue') -> 'FixedPointValue':
        """减法"""
        result = self._align_frac_bits(other)
        return FixedPointValue(self.raw - result.raw, self.ftype)
    
    def __mul__(self, other: 'FixedPointValue') -> 'FixedPointType':
        """乘法（返回更高精度类型）"""
        # 乘积需要更多小数位
        product = self.raw * other.raw
        new_frac_bits = self.ftype.frac_bits + other.ftype.frac_bits
        new_int_bits = self.ftype.int_bits + other.ftype.int_bits
        new_ftype = FixedPointType(
            name=f"临时定点数",
            total_bits=new_int_bits + new_frac_bits,
            int_bits=new_int_bits,
            frac_bits=new_frac_bits,
            is_signed=self.ftype.is_signed and other.ftype.is_signed,
        )
        return FixedPointValue(product >> new_frac_bits, new_ftype)
    
    def shift_left(self, bits: int) -> 'FixedPointValue':
        """左移（乘以 2^bits）"""
        return FixedPointValue(self.raw << bits, self.ftype)
    
    def shift_right(self, bits: int) -> 'FixedPointValue':
        """右移（除以 2^bits）"""
        if self.ftype.is_signed:
            # 算术右移
            result = self.raw >> bits
            return FixedPointValue(result, self.ftype)
        return FixedPointValue(self.raw >> bits, self.ftype)
    
    def _align_frac_bits(self, other: 'FixedPointValue') -> 'FixedPointValue':
        """对齐小数位"""
        if self.ftype.frac_bits == other.ftype.frac_bits:
            return other
        # 需要对齐
        diff = self.ftype.frac_bits - other.ftype.frac_bits
        if diff > 0:
            # other 需要左移
            return FixedPointValue(other.raw << diff, self.ftype)
        else:
            # self 需要左移
            return FixedPointValue(self.raw << (-diff), other.ftype)
```
