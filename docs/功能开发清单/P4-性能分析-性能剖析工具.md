## 1. 需求分析

### 1.1 功能目标

实现性能剖析工具：

1. **函数调用追踪**: 记录函数调用关系
2. **执行时间测量**: 测量函数执行时间
3. **内存使用统计**: 跟踪内存分配释放
4. **热点识别**: 识别性能瓶颈

### 1.2 功能特性

| 功能 | 说明 |
|-----|------|
| 调用栈追踪 | 记录函数调用关系 |
| 执行时间 | 测量函数执行时间（纳秒精度） |
| 调用次数 | 统计函数被调用次数 |
| 调用关系图 | 生成函数调用关系图 |

## 2. 实现方案

### 2.1 性能剖析器架构

```c
// src/lib/zhc_profiler.h

#ifndef ZHC_PROFILER_H
#define ZHC_PROFILER_H

#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <stdint.h>

// 函数记录
typedef struct {
    const char* name;
    uint64_t call_count;
    uint64_t total_time_ns;
    uint64_t min_time_ns;
    uint64_t max_time_ns;
} FunctionProfile;

// 调用关系
typedef struct {
    const char* caller;
    const char* callee;
    uint64_t call_count;
} CallRelation;

// 剖析器状态
typedef struct {
    int enabled;
    FILE* output;
    FunctionProfile* functions;
    int function_count;
    int max_functions;
} ProfilerState;

extern ProfilerState g_profiler;

// 剖析器控制
void profiler_init(int max_functions);
void profiler_shutdown(void);
void profiler_enable(void);
void profiler_disable(void);

// 函数剖析宏
#define PROFILE_FUNCTION() \
    ProfileAuto _profile_auto(__func__)

#define PROFILE_SCOPE(name) \
    ProfileAuto _profile_auto(name)

typedef struct {
    const char* name;
    uint64_t start_time;
} ProfileAuto;

// 剖析器实现
void profiler_enter(const char* func_name);
void profiler_exit(const char* func_name);
void profiler_report(void);

#endif
```

### 2.2 剖析器实现

```c
// src/lib/zhc_profiler.c

#include "zhc_profiler.h"
#include <string.h>
#include <sys/time.h>

ProfilerState g_profiler = {0};

static uint64_t get_time_ns(void) {
    struct timespec ts;
    clock_gettime(CLOCK_MONOTONIC, &ts);
    return (uint64_t)ts.tv_sec * 1000000000ULL + ts.tv_nsec;
}

void profiler_init(int max_functions) {
    g_profiler.enabled = 0;
    g_profiler.function_count = 0;
    g_profiler.max_functions = max_functions;
    g_profiler.functions = calloc(max_functions, sizeof(FunctionProfile));
}

void profiler_enable(void) {
    g_profiler.enabled = 1;
}

void profiler_disable(void) {
    g_profiler.enabled = 0;
}

void profiler_enter(const char* func_name) {
    if (!g_profiler.enabled) return;
    
    // 查找或创建函数记录
    FunctionProfile* profile = NULL;
    for (int i = 0; i < g_profiler.function_count; i++) {
        if (strcmp(g_profiler.functions[i].name, func_name) == 0) {
            profile = &g_profiler.functions[i];
            break;
        }
    }
    
    if (!profile && g_profiler.function_count < g_profiler.max_functions) {
        profile = &g_profiler.functions[g_profiler.function_count++];
        profile->name = func_name;
        profile->call_count = 0;
        profile->total_time_ns = 0;
        profile->min_time_ns = UINT64_MAX;
        profile->max_time_ns = 0;
    }
    
    if (profile) {
        profile->call_count++;
        profile->start_time = get_time_ns();
    }
}

void profiler_exit(const char* func_name) {
    if (!g_profiler.enabled) return;
    
    uint64_t elapsed = get_time_ns() - 0; // 需要传入 start_time
    
    // 更新函数统计
    for (int i = 0; i < g_profiler.function_count; i++) {
        if (strcmp(g_profiler.functions[i].name, func_name) == 0) {
            FunctionProfile* profile = &g_profiler.functions[i];
            profile->total_time_ns += elapsed;
            if (elapsed < profile->min_time_ns) profile->min_time_ns = elapsed;
            if (elapsed > profile->max_time_ns) profile->max_time_ns = elapsed;
            break;
        }
    }
}

void profiler_report(void) {
    printf("\n=== 性能剖析报告 ===\n\n");
    printf("%-30s %10s %15s %12s %12s\n", 
           "函数", "调用次数", "总时间", "平均时间", "最大时间");
    printf("------------------------------------------------------------------------\n");
    
    for (int i = 0; i < g_profiler.function_count; i++) {
        FunctionProfile* p = &g_profiler.functions[i];
        uint64_t avg = p->call_count > 0 ? p->total_time_ns / p->call_count : 0;
        
        printf("%-30s %10lu %12.3fms %10.3fus %10.3fus\n",
               p->name,
               p->call_count,
               p->total_time_ns / 1000000.0,
               avg / 1000.0,
               p->max_time_ns / 1000.0);
    }
}
```
