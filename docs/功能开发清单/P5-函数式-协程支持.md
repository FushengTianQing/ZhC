> **优先级**: P5

> **创建日期**: 2026-04-10
> **状态**: 规划中

## 1. 功能概述

实现异步函数和协程支持，允许函数在执行过程中暂停和恢复，实现非阻塞 I/O 和并发处理。

### 1.1 目标

- 支持协程函数定义语法
- 支持协程的创建、暂停、恢复
- 支持异步/等待语法
- 支持协程间的通信（channel）

### 1.2 语法设计

```zhc
// 协程定义
协程 函数 获取数据(字符串 url) -> 字符串 {
    字符串 数据 = 等待 HTTP请求(url);
    返回 数据;
}

// 异步调用
异步 函数 主函数() {
    字符串 结果 = 等待 获取数据("https://api.example.com");
    打印(结果);
}

// 并发执行
函数 并行示例() {
    任务 任务1 = 启动 协程 计算任务1();
    任务 任务2 = 启动 协程 计算任务2();
    
    整数型 结果1 = 等待 任务1;
    整数型 结果2 = 等待 任务2;
    
    返回 结果1 + 结果2;
}

// Channel 通信
函数 管道示例() {
    通道[整数型] 通道1 = 创建通道[整数型]();
    
    启动 协程 生产者(通道1);
    启动 协程 消费者(通道1);
}

// 定时器
函数 定时器示例() {
    等待 1秒;
    打印("1秒后执行");
}
```

### 3.2 核心数据结构

```python
# src/zhc/functional/coroutine.py

from dataclasses import dataclass, field
from typing import List, Optional, Any, Callable
from enum import Enum

class CoroutineState(Enum):
    """协程状态"""
    CREATED = "created"      # 创建
    RUNNING = "running"      # 运行中
    SUSPENDED = "suspended"  # 暂停
    WAITING = "waiting"      # 等待（I/O等）
    COMPLETED = "completed"  # 完成
    CANCELLED = "cancelled" # 取消

@dataclass
class CoroutineStack:
    """协程栈帧"""
    function_name: str
    return_address: int
    local_variables: dict = field(default_factory=dict)
    saved_registers: List[Any] = field(default_factory=list)

@dataclass
class CoroutineContext:
    """协程上下文"""
    stack: List[CoroutineStack] = field(default_factory=list)
    current_frame: Optional[CoroutineStack] = None
    upvalues: dict = field(default_factory=dict)
    
    def push_frame(self, frame: CoroutineStack) -> None:
        self.stack.append(frame)
        self.current_frame = frame
    
    def pop_frame(self) -> Optional[CoroutineStack]:
        if self.stack:
            frame = self.stack.pop()
            self.current_frame = self.stack[-1] if self.stack else None
            return frame
        return None

@dataclass
class Coroutine:
    """协程对象"""
    id: int
    name: str
    function: Callable
    state: CoroutineState
    context: CoroutineContext
    result: Optional[Any] = None
    exception: Optional[Exception] = None
    
    def resume(self) -> Any:
        """恢复协程"""
        if self.state not in (CoroutineState.CREATED, CoroutineState.SUSPENDED):
            raise RuntimeError(f"Cannot resume {self.state}")
        self.state = CoroutineState.RUNNING
        # ... 执行到下一个 yield 点
        return self.result
    
    def suspend(self, value: Any = None) -> Any:
        """暂停协程"""
        self.state = CoroutineState.SUSPENDED
        self.result = value
        # ... 保存状态并让出控制权
        return value
    
    def cancel(self) -> None:
        """取消协程"""
        self.state = CoroutineState.CANCELLED
    
    def is_done(self) -> bool:
        """检查是否完成"""
        return self.state in (
            CoroutineState.COMPLETED,
            CoroutineState.CANCELLED,
        )

@dataclass
class Channel:
    """通道（用于协程间通信）"""
    buffer_size: int
    buffer: List[Any] = field(default_factory=list)
    senders: List[Coroutine] = field(default_factory=list)
    receivers: List[Coroutine] = field(default_factory=list)
    closed: bool = False
    
    def send(self, value: Any, sender: Coroutine) -> None:
        """发送值"""
        if self.closed:
            raise RuntimeError("Channel is closed")
        if len(self.buffer) >= self.buffer_size:
            self.senders.append(sender)
            sender.suspend()
        self.buffer.append(value)
        # 唤醒等待的接收者
        self._wake_receiver()
    
    def recv(self, receiver: Coroutine) -> Any:
        """接收值"""
        if self.closed and not self.buffer:
            raise RuntimeError("Channel is closed and empty")
        if not self.buffer:
            self.receivers.append(receiver)
            receiver.suspend()
            return None
        value = self.buffer.pop(0)
        # 唤醒等待的发送者
        self._wake_sender()
        return value
    
    def close(self) -> None:
        """关闭通道"""
        self.closed = True
        # 唤醒所有等待的协程
        for coroutine in self.senders + self.receivers:
            coroutine.resume()
    
    def _wake_receiver(self) -> None:
        if self.receivers:
            receiver = self.receivers.pop(0)
            receiver.resume()
    
    def _wake_sender(self) -> None:
        if self.senders:
            sender = self.senders.pop(0)
            sender.resume()
```

### 3.3 调度器

```python
# src/zhc/functional/scheduler.py

class Scheduler:
    """协程调度器"""
    
    _instance: Optional['Scheduler'] = None
    
    def __init__(self):
        self.coroutines: Dict[int, Coroutine] = {}
        self.ready_queue: List[Coroutine] = []
        self.wait_queue: Dict[Coroutine, Any] = {}
        self.current: Optional[Coroutine] = None
        self.next_id: int = 1
    
    @classmethod
    def instance(cls) -> 'Scheduler':
        if cls._instance is None:
            cls._instance = cls()
        return cls._instance
    
    def create_coroutine(self, func: Callable, name: str = None) -> Coroutine:
        """创建协程"""
        coro = Coroutine(
            id=self.next_id,
            name=name or f"coro_{self.next_id}",
            function=func,
            state=CoroutineState.CREATED,
            context=CoroutineContext(),
        )
        self.coroutines[self.next_id] = coro
        self.next_id += 1
        return coro
    
    def start(self, coro: Coroutine) -> None:
        """启动协程"""
        if coro.state != CoroutineState.CREATED:
            raise RuntimeError("Can only start created coroutine")
        coro.state = CoroutineState.RUNNING
        self.ready_queue.append(coro)
        if self.current is None:
            self.run()
    
    def resume(self, coro: Coroutine) -> Any:
        """恢复协程"""
        if coro.is_done():
            return coro.result
        coro.state = CoroutineState.RUNNING
        self.ready_queue.append(coro)
        if self.current != coro:
            self._switch_to(coro)
        return coro.result
    
    def suspend(self, coro: Coroutine, wait_for: Any = None) -> None:
        """挂起协程"""
        coro.state = CoroutineState.SUSPENDED if wait_for is None else CoroutineState.WAITING
        if wait_for is not None:
            self.wait_queue[coro] = wait_for
    
    def schedule(self) -> Optional[Coroutine]:
        """调度下一个协程"""
        if self.ready_queue:
            return self.ready_queue.pop(0)
        return None
    
    def run(self) -> None:
        """运行调度器"""
        while self.ready_queue or self.wait_queue:
            # 调度就绪协程
            coro = self.schedule()
            if coro:
                self.current = coro
                try:
                    coro.resume()
                except Exception as e:
                    coro.exception = e
                    coro.state = CoroutineState.CANCELLED
                
                if coro.is_done():
                    self._on_coroutine_done(coro)
                elif coro.state == CoroutineState.SUSPENDED:
                    # 保持挂起状态
                    pass
                else:
                    self.ready_queue.append(coro)
            
            # 处理等待队列
            self._process_wait_queue()
    
    def _switch_to(self, target: Coroutine) -> None:
        """切换到目标协程"""
        # 保存当前协程状态
        # 恢复目标协程状态
        pass
    
    def _process_wait_queue(self) -> None:
        """处理等待队列"""
        for coro, condition in list(self.wait_queue.items()):
            if self._check_condition(condition):
                del self.wait_queue[coro]
                coro.state = CoroutineState.RUNNING
                self.ready_queue.append(coro)
    
    def _check_condition(self, condition) -> bool:
        """检查条件是否满足"""
        return True
    
    def _on_coroutine_done(self, coro: Coroutine) -> None:
        """协程完成回调"""
        if coro.exception:
            raise coro.exception
```

## 5. API 设计

### 5.1 Python API

```python
# src/zhc/functional/__init__.py

from .coroutine import (
    Coroutine,
    CoroutineState,
    Channel,
)
from .scheduler import Scheduler

# 获取调度器
scheduler = Scheduler.instance()

# 创建并启动协程
async def my_task():
    await asyncio.sleep(1)
    return 42

coro = scheduler.create_coroutine(my_task, "my_task")
scheduler.start(coro)

# 创建通道
ch = Channel(buffer_size=10)

# 使用 Channel
async def producer(ch):
    for i in range(10):
        await ch.send(i)

async def consumer(ch):
    for _ in range(10):
        value = await ch.recv()
        print(f"Received: {value}")

# 运行
scheduler.start(scheduler.create_coroutine(producer(ch), "producer"))
scheduler.start(scheduler.create_coroutine(consumer(ch), "consumer"))
```
