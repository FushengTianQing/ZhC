## 1. 需求分析

### 1.1 功能目标

实现内存使用分析工具：

1. **内存分配追踪**: 记录所有内存分配
2. **内存释放验证**: 验证内存是否正确释放
3. **泄漏检测**: 检测内存泄漏
4. **使用统计**: 统计内存使用情况

### 1.2 分析功能

| 功能 | 说明 |
|-----|------|
| 分配追踪 | 记录 malloc/申请 分配 |
| 释放追踪 | 记录 free/释放 释放 |
| 泄漏检测 | 检测未释放的内存 |
| 使用统计 | 峰值内存、分配次数统计 |

## 2. 实现方案

### 2.1 内存分析器架构

```c
// src/lib/zhc_memcheck.h

#ifndef ZHC_MEMCHECK_H
#define ZHC_MEMCHECK_H

#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>
#include <string.h>

// 内存块信息
typedef struct MemBlock {
    void* ptr;
    size_t size;
    const char* file;
    int line;
    const char* alloc_func;
    uint64_t alloc_time;
    struct MemBlock* next;
} MemBlock;

// 内存统计
typedef struct {
    uint64_t total_alloc;
    uint64_t total_free;
    uint64_t current_used;
    uint64_t peak_used;
    uint64_t alloc_count;
    uint64_t leak_count;
} MemStats;

extern MemStats g_mem_stats;
extern MemBlock* g_mem_blocks;

// 初始化
void memcheck_init(void);
void memcheck_shutdown(void);

// 追踪函数
void* memcheck_alloc(size_t size, const char* file, int line, const char* func);
void memcheck_free(void* ptr, const char* file, int line);
void* memcheck_realloc(void* ptr, size_t new_size, const char* file, int line);

// 分配宏（替换标准函数）
#define 申请(size) memcheck_alloc(size, __FILE__, __LINE__, "申请")
#define 释放(ptr) memcheck_free(ptr, __FILE__, __LINE__)

// 报告函数
void memcheck_report(void);
void memcheck_print_stats(void);
int memcheck_check_leaks(void);

#endif
```

### 2.2 内存追踪实现

```c
// src/lib/zhc_memcheck.c

#include "zhc_memcheck.h"
#include <time.h>

MemStats g_mem_stats = {0};
MemBlock* g_mem_blocks = NULL;
static uint64_t g_alloc_id = 0;

// 获取时间戳
static uint64_t get_timestamp(void) {
    struct timespec ts;
    clock_gettime(CLOCK_MONOTONIC, &ts);
    return (uint64_t)ts.tv_sec * 1000000000ULL + ts.tv_nsec;
}

// 记录内存块
static void record_block(MemBlock* block) {
    block->next = g_mem_blocks;
    g_mem_blocks = block;
}

// 查找内存块
static MemBlock* find_block(void* ptr) {
    for (MemBlock* block = g_mem_blocks; block; block = block->next) {
        if (block->ptr == ptr) {
            return block;
        }
    }
    return NULL;
}

// 移除内存块
static void remove_block(void* ptr) {
    MemBlock** pprev = &g_mem_blocks;
    for (MemBlock* block = g_mem_blocks; block; block = block->next) {
        if (block->ptr == ptr) {
            *pprev = block->next;
            free(block);
            return;
        }
        pprev = &block->next;
    }
}

void* memcheck_alloc(size_t size, const char* file, int line, const char* func) {
    // 分配实际内存
    void* ptr = malloc(size);
    if (!ptr) return NULL;
    
    // 创建块记录
    MemBlock* block = malloc(sizeof(MemBlock));
    block->ptr = ptr;
    block->size = size;
    block->file = file;
    block->line = line;
    block->alloc_func = func;
    block->alloc_time = get_timestamp();
    
    // 记录
    record_block(block);
    
    // 更新统计
    g_mem_stats.total_alloc += size;
    g_mem_stats.current_used += size;
    g_mem_stats.alloc_count++;
    
    if (g_mem_stats.current_used > g_mem_stats.peak_used) {
        g_mem_stats.peak_used = g_mem_stats.current_used;
    }
    
    return ptr;
}

void memcheck_free(void* ptr, const char* file, int line) {
    if (!ptr) return;
    
    MemBlock* block = find_block(ptr);
    if (block) {
        // 更新统计
        g_mem_stats.total_free += block->size;
        g_mem_stats.current_used -= block->size;
        
        // 移除记录
        remove_block(ptr);
        
        // 释放内存
        free(ptr);
    } else {
        fprintf(stderr, "警告: 尝试释放未分配的内存 %p (%s:%d)\n", ptr, file, line);
    }
}

void memcheck_report(void) {
    printf("\n=== 内存使用报告 ===\n\n");
    
    // 打印统计
    printf("总分配: %lu 字节\n", g_mem_stats.total_alloc);
    printf("总释放: %lu 字节\n", g_mem_stats.total_free);
    printf("当前使用: %lu 字节\n", g_mem_stats.current_used);
    printf("峰值使用: %lu 字节\n", g_mem_stats.peak_used);
    printf("分配次数: %lu\n", g_mem_stats.alloc_count);
    printf("\n");
    
    // 打印泄漏
    if (g_mem_blocks) {
        printf("=== 内存泄漏 ===\n\n");
        for (MemBlock* block = g_mem_blocks; block; block = block->next) {
            printf("泄漏: %p (大小: %zu 字节)\n", block->ptr, block->size);
            printf("  位置: %s:%d\n", block->file, block->line);
            printf("  函数: %s\n", block->alloc_func);
            
            g_mem_stats.leak_count++;
        }
        
        printf("\n总计泄漏: %lu 处, %lu 字节\n", 
               g_mem_stats.leak_count, g_mem_stats.current_used);
    } else {
        printf("未检测到内存泄漏！\n");
    }
}

int memcheck_check_leaks(void) {
    memcheck_report();
    return g_mem_blocks != NULL;
}
```
