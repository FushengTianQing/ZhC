> **创建日期**: 2026-04-10

## 1. 功能概述

实现异常在调用栈中的传播机制，支持跨函数传播、栈展开、资源清理，确保异常能够正确地从抛出点传递到捕获点。

### 1.1 目标

- 实现异常跨函数传播
- 实现栈展开（stack unwinding）
- 支持 finally 块的自动执行
- 支持异常嵌套和重新抛出

### 1.2 传播规则

```zhc
函数 内层函数() {
    抛出 新 运行时错误("内层错误");
}

函数 中间函数() {
    内层函数();  // 异常向上传播
}

函数 外层函数() {
    尝试 {
        中间函数();
    } 捕获 (运行时错误 e) {
        打印("捕获: " + e.消息);
    }
}
```

### 1.3 传播流程

```
外层函数()
    │
    ├─ 尝试块开始
    │
    ├─ 中间函数()
    │       │
    │       ├─ 内层函数()
    │       │       │
    │       │       └─ 抛出异常 ──────────┐
    │       │                             │
    │       └─ (栈展开) ◄─────────────────┘
    │
    ├─ (栈展开)
    │
    └─ 捕获块处理异常
```

### 3.2 核心数据结构

```python
# src/zhc/exception/propagation.py

from dataclasses import dataclass, field
from typing import List, Optional, Callable
from enum import Enum

class ExceptionState(Enum):
    """异常状态"""
    NONE = "none"           # 无异常
    THROWING = "throwing"   # 正在抛出
    UNWINDING = "unwinding" # 栈展开中
    CAUGHT = "caught"       # 已捕获
    HANDLED = "handled"     # 已处理

@dataclass
class StackFrame:
    """栈帧信息"""
    function_name: str
    file_name: str
    line_number: int
    has_finally: bool = False
    finally_block: Optional[Callable] = None
    cleanup_handlers: List[Callable] = field(default_factory=list)

@dataclass
class ExceptionContext:
    """异常上下文"""
    current_exception: Optional[ExceptionObject] = None
    state: ExceptionState = ExceptionState.NONE
    stack_frames: List[StackFrame] = field(default_factory=list)
    handler_stack: List['ExceptionHandler'] = field(default_factory=list)
    
    def push_frame(self, frame: StackFrame) -> None:
        """压入栈帧"""
        self.stack_frames.append(frame)
    
    def pop_frame(self) -> Optional[StackFrame]:
        """弹出栈帧"""
        return self.stack_frames.pop() if self.stack_frames else None
    
    def find_handler(self, exc_type: str) -> Optional['ExceptionHandler']:
        """查找匹配的处理器"""
        for handler in reversed(self.handler_stack):
            if handler.matches(exc_type):
                return handler
        return None

@dataclass
class ExceptionHandler:
    """异常处理器"""
    catch_types: List[str]        # 捕获的异常类型
    catch_block: Callable         # catch 代码块
    finally_block: Optional[Callable] = None  # finally 代码块
    frame_index: int = 0          # 所在栈帧索引
    variable_name: Optional[str] = None  # 异常变量名
    
    def matches(self, exc_type: str) -> bool:
        """检查是否匹配异常类型"""
        return exc_type in self.catch_types
```

### 3.3 栈展开算法

```python
class StackUnwinder:
    """栈展开器"""
    
    def __init__(self, context: ExceptionContext):
        self.context = context
    
    def unwind_to_handler(self, handler: ExceptionHandler) -> None:
        """展开到指定处理器"""
        self.context.state = ExceptionState.UNWINDING
        
        while self.context.stack_frames:
            frame = self.context.stack_frames[-1]
            
            # 执行 finally 块
            if frame.has_finally and frame.finally_block:
                frame.finally_block()
            
            # 执行清理处理器
            for cleanup in frame.cleanup_handlers:
                cleanup()
            
            # 弹出栈帧
            self.context.pop_frame()
            
            # 如果到达处理器所在帧，停止展开
            if len(self.context.stack_frames) <= handler.frame_index:
                break
    
    def unwind_to_top(self) -> None:
        """展开到顶层（未捕获异常）"""
        self.context.state = ExceptionState.UNWINDING
        
        while self.context.stack_frames:
            frame = self.context.pop_frame()
            
            if frame.has_finally and frame.finally_block:
                frame.finally_block()
            
            for cleanup in frame.cleanup_handlers:
                cleanup()
        
        # 调用顶层异常处理器
        self._handle_uncaught()
    
    def _handle_uncaught(self) -> None:
        """处理未捕获异常"""
        exc = self.context.current_exception
        if exc:
            # 输出错误信息并终止程序
            print(f"未捕获异常: {exc.type_name}: {exc.message}")
            # ... 输出堆栈跟踪
            import sys
            sys.exit(1)
```

### 3.4 异常传播核心

```python
class ExceptionPropagator:
    """异常传播器"""
    
    def __init__(self):
        self.context = ExceptionContext()
        self.unwinder = StackUnwinder(self.context)
    
    def throw(self, exc: ExceptionObject) -> None:
        """抛出异常"""
        self.context.current_exception = exc
        self.context.state = ExceptionState.THROWING
        
        # 捕获堆栈跟踪
        exc.stack_trace = self._capture_stack_trace()
        
        # 查找处理器
        handler = self.context.find_handler(exc.type_name)
        
        if handler:
            # 找到处理器，展开到该位置
            self.unwinder.unwind_to_handler(handler)
            # 执行 catch 块
            self._execute_catch(handler, exc)
        else:
            # 未找到处理器，继续向上传播
            self._propagate_to_parent(exc)
    
    def _execute_catch(self, handler: ExceptionHandler, exc: ExceptionObject) -> None:
        """执行 catch 块"""
        self.context.state = ExceptionState.CAUGHT
        
        # 绑定异常变量
        if handler.variable_name:
            # ... 绑定变量到当前作用域
            pass
        
        # 执行 catch 代码块
        try:
            handler.catch_block()
        except Exception as e:
            # catch 块中又抛出异常
            self.throw(e)
        finally:
            # 执行 finally 块
            if handler.finally_block:
                handler.finally_block()
        
        self.context.state = ExceptionState.HANDLED
    
    def _propagate_to_parent(self, exc: ExceptionObject) -> None:
        """向上传播异常"""
        # 弹出当前帧
        self.context.pop_frame()
        
        if self.context.stack_frames:
            # 继续查找处理器
            handler = self.context.find_handler(exc.type_name)
            if handler:
                self.unwinder.unwind_to_handler(handler)
                self._execute_catch(handler, exc)
            else:
                self._propagate_to_parent(exc)
        else:
            # 到达顶层
            self.unwinder.unwind_to_top()
    
    def _capture_stack_trace(self) -> List[str]:
        """捕获堆栈跟踪"""
        import traceback
        return traceback.format_stack()
```

from .propagation import (
    ExceptionContext,
    ExceptionPropagator,
    StackUnwinder,
    ExceptionHandler,
)

# 获取全局传播器
_global_propagator = ExceptionPropagator()

def throw(exc: ExceptionObject) -> None:
    """抛出异常"""
    _global_propagator.throw(exc)

def get_context() -> ExceptionContext:
    """获取当前异常上下文"""
    return _global_propagator.context

# 使用上下文管理器
with exception_context() as ctx:
    ctx.register_handler(
        catch_types=["运行时错误"],

        finally_block=lambda: print("清理")
