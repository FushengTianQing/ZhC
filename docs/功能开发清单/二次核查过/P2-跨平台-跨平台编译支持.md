## 1. 开发内容分析

### 1.1 目标概述

实现完整的交叉编译支持，允许在单一平台上为多个目标平台生成代码。这包括目标平台检测、交叉编译工具链配置、平台特定优化等。

### 1.2 技术背景

#### 交叉编译场景
```
┌─────────────────────────────────────────────────────────┐
│                    Cross-Compile Matrix                  │
├─────────────────────────────────────────────────────────┤
│          │   x86_64   │  aarch64  │   arm    │  riscv  │
│  Host    │            │           │          │         │
├──────────┼────────────┼───────────┼──────────┼─────────┤
│  x86_64  │  Native    │  Cross    │  Cross   │  Cross  │
│ aarch64  │  Cross     │  Native   │  Cross   │  Cross  │
│   arm    │  Cross     │  Cross    │  Native  │  Cross  │
│  riscv   │  Cross     │  Cross    │  Cross   │  Native │
└─────────────────────────────────────────────────────────┘
```

#### 目标三元组 (Target Triple)
```
格式: <arch>-<vendor>-<os>[-<environment>]

示例:
- x86_64-unknown-linux-gnu      # Linux x86_64
- aarch64-apple-darwin          # macOS Apple Silicon
- arm-none-eabi                 # 嵌入式 ARM
- riscv64-unknown-linux-gnu     # Linux RISC-V
- wasm32-unknown-unknown        # WebAssembly
```

### 1.3 需求分析

#### 核心需求
1. **目标平台检测**：自动检测主机和目标平台
2. **交叉编译工具链**：配置和管理交叉编译工具链
3. **平台特定配置**：处理不同平台的差异
4. **多目标构建**：支持同时为多个目标构建

#### 用户场景
```bash
# 在 macOS 上为 Linux 编译
zhc --target=x86_64-unknown-linux-gnu input.zhc -o linux_binary

# 在 Linux 上为 Windows 编译
zhc --target=x86_64-pc-windows-gnu input.zhc -o windows.exe

# 在 x86 上为 ARM 编译
zhc --target=aarch64-unknown-linux-gnu input.zhc -o arm_binary

# 同时为多个平台编译
zhc --targets=x86_64,aarch64,wasm32 input.zhc -o output/
```

### 2.3 核心接口设计

#### CrossCompileManager 类
```python
class CrossCompileManager:
    """交叉编译管理器"""

    def __init__(self):
        self.host = HostDetector.detect()
        self.target_manager = TargetManager()
        self.toolchain_manager = ToolchainManager()
        self.sysroot_manager = SysrootManager()

    def configure_target(self, triple: str,
                         toolchain_path: str = None,
                         sysroot: str = None) -> TargetConfig:
        """配置目标平台

        Args:
            triple: 目标三元组
            toolchain_path: 工具链路径（可选）
            sysroot: 系统根路径（可选）

        Returns:
            目标配置对象
        """
        target = self.target_manager.parse_triple(triple)

        if toolchain_path:
            self.toolchain_manager.register_toolchain(target, toolchain_path)

        if sysroot:
            self.sysroot_manager.set_sysroot(target, sysroot)

        return self._build_target_config(target)

    def compile_for_target(self, source: str, target: str,
                           output: str, options: CompileOptions = None):
        """为指定目标编译"""
        config = self.configure_target(target)
        compiler = self._create_compiler(config)
        return compiler.compile(source, output, options)
```

#### TargetConfig 类
```python
@dataclass
class TargetConfig:
    """目标平台配置"""

    # 基本信息
    triple: str                          # 目标三元组
    arch: Architecture                   # 架构
    vendor: Vendor                       # 厂商
    os: OperatingSystem                  # 操作系统
    environment: EnvironmentType          # 环境

    # 工具链配置
    toolchain: Optional[Toolchain] = None
    sysroot: Optional[str] = None
    linker: Optional[str] = None

    # 编译选项
    cpu: str = "generic"
    features: List[str] = field(default_factory=list)
    abi: str = ""

    # 链接选项
    link_type: LinkType = LinkType.DYNAMIC
    runtime_library: str = ""
    crt_files: List[str] = field(default_factory=list)

    def is_native(self) -> bool:
        """是否为本机编译"""
        return self.triple == HostDetector.detect().triple

    def is_cross(self) -> bool:
        """是否为交叉编译"""
        return not self.is_native()
```

```python
# src/zhc/cross/host_detector.py
class HostDetector:
    """主机平台检测器"""

    @classmethod
    def detect(cls) -> HostInfo:
        """检测主机平台信息"""
        return HostInfo(
            triple=cls._detect_triple(),
            arch=cls._detect_arch(),
            os=cls._detect_os(),
            vendor=cls._detect_vendor(),
            cpu=cls._detect_cpu(),
            features=cls._detect_features(),
        )

    @classmethod
    def _detect_triple(cls) -> str:
        """检测主机三元组"""
        import platform

        arch = platform.machine().lower()
        system = platform.system().lower()

        # 映射到 LLVM triple
        arch_map = {
            'x86_64': 'x86_64',
            'amd64': 'x86_64',
            'aarch64': 'aarch64',
            'arm64': 'aarch64',
            'armv7l': 'arm',
            'riscv64': 'riscv64',
        }

        os_map = {
            'linux': 'unknown-linux-gnu',
            'darwin': 'apple-darwin',
            'windows': 'pc-windows-msvc',
        }

        return f"{arch_map.get(arch, arch)}-{os_map.get(system, system)}"

    @classmethod
    def _detect_cpu(cls) -> str:
        """检测 CPU 型号"""
        import platform
        cpu = platform.processor()

        # Linux: 从 /proc/cpuinfo 获取
        if platform.system() == 'Linux':
            try:
                with open('/proc/cpuinfo') as f:
                    for line in f:
                        if line.startswith('model name'):
                            return line.split(':')[1].strip()
            except:
                pass

        # macOS: 使用 sysctl
        elif platform.system() == 'Darwin':
            result = subprocess.run(
                ['sysctl', '-n', 'machdep.cpu.brand_string'],
                capture_output=True, text=True
            )
            if result.returncode == 0:
                return result.stdout.strip()

        return cpu or "generic"
```

#### 任务 1.2: Triple 解析器
```python
# src/zhc/cross/triple_parser.py
class TripleParser:
    """目标三元组解析器"""

    @classmethod
    def parse(cls, triple: str) -> TargetTriple:
        """解析三元组字符串

        格式: <arch>-<vendor>-<os>[-<environment>]

        示例:
        - x86_64-unknown-linux-gnu
        - aarch64-apple-darwin
        - arm-none-eabi
        """
        parts = triple.split('-')

        if len(parts) < 3:
            raise ValueError(f"Invalid triple: {triple}")

        arch = cls._parse_arch(parts[0])
        vendor = cls._parse_vendor(parts[1])
        os = cls._parse_os(parts[2])
        environment = cls._parse_environment(parts[3]) if len(parts) > 3 else None

        return TargetTriple(
            arch=arch,
            vendor=vendor,
            os=os,
            environment=environment,
            original=triple,
        )

    @classmethod
    def _parse_arch(cls, arch_str: str) -> Architecture:
        """解析架构"""
        arch_map = {
            'x86_64': Architecture.X86_64,
            'amd64': Architecture.X86_64,
            'i386': Architecture.I386,
            'i686': Architecture.I386,
            'aarch64': Architecture.AARCH64,
            'arm64': Architecture.AARCH64,
            'arm': Architecture.ARM,
            'armv7': Architecture.ARM,
            'armv7a': Architecture.ARM,
            'riscv64': Architecture.RISCV64,
            'riscv32': Architecture.RISCV32,
            'wasm32': Architecture.WASM32,
            'wasm64': Architecture.WASM64,
        }
        return arch_map.get(arch_str.lower(), Architecture.UNKNOWN)
```

```python
# src/zhc/cross/toolchain_manager.py
class ToolchainManager:
    """工具链管理器"""

    def __init__(self):
        self.toolchains: Dict[str, Toolchain] = {}

    def detect_toolchain(self, target: TargetTriple) -> Optional[Toolchain]:
        """检测目标工具链"""
        # 1. 检查内置 LLVM 工具链
        if self._has_builtin_llvm_toolchain(target):
            return self._create_builtin_toolchain(target)

        # 2. 检查系统工具链
        system_toolchain = self._detect_system_toolchain(target)
        if system_toolchain:
            return system_toolchain

        # 3. 检查已注册的工具链
        return self.toolchains.get(target.triple)

    def _detect_system_toolchain(self, target: TargetTriple) -> Optional[Toolchain]:
        """检测系统工具链"""
        if target.os == OperatingSystem.LINUX:
            # 检查交叉编译工具链
            prefix = self._get_cross_compile_prefix(target)
            for tool in ['gcc', 'g++', 'ld']:
                if not shutil.which(f"{prefix}-{tool}"):
                    return None
            return Toolchain(
                prefix=prefix,
                path=f"/usr/{prefix}",
                tools={
                    'cc': f"{prefix}-gcc",
                    'cxx': f"{prefix}-g++",
                    'ld': f"{prefix}-ld",
                    'ar': f"{prefix}-ar",
                    'as': f"{prefix}-as",
                }
            )

        elif target.os == OperatingSystem.DARWIN:
            # macOS 使用 Xcode 工具链
            return self._detect_xcode_toolchain()

        elif target.os == OperatingSystem.WINDOWS:
            # Windows 使用 MSVC 或 MinGW
            return self._detect_msvc_toolchain() or self._detect_mingw_toolchain()

        return None
```

#### 任务 2.2: Sysroot 管理
```python
# src/zhc/cross/sysroot_manager.py
class SysrootManager:
    """系统根管理器"""

    def __init__(self):
        self.sysroots: Dict[str, str] = {}

    def set_sysroot(self, target: TargetTriple, path: str):
        """设置 sysroot"""
        if not os.path.isdir(path):
            raise ValueError(f"Sysroot path does not exist: {path}")

        self.sysroots[target.triple] = path

    def get_sysroot(self, target: TargetTriple) -> Optional[str]:
        """获取 sysroot"""
        # 1. 检查显式设置的 sysroot
        if target.triple in self.sysroots:
            return self.sysroots[target.triple]

        # 2. 尝试自动检测
        return self._auto_detect_sysroot(target)

    def _auto_detect_sysroot(self, target: TargetTriple) -> Optional[str]:
        """自动检测 sysroot"""
        if target.os == OperatingSystem.LINUX:
            # 检查常见路径
            common_paths = [
                f"/usr/{target.arch}-linux-gnu",
                f"/usr/{target.arch}-unknown-linux-gnu",
                f"/opt/sysroot/{target.triple}",
            ]
            for path in common_paths:
                if os.path.isdir(path):
                    return path

        elif target.os == OperatingSystem.DARWIN:
            # macOS SDK
            result = subprocess.run(
                ["xcrun", "--show-sdk-path"],
                capture_output=True, text=True
            )
            if result.returncode == 0:
                return result.stdout.strip()

        return None
```

```python
# src/zhc/cross/linker_manager.py
class LinkerManager:
    """链接器管理器"""

    # 支持的链接器
    LINKERS = {
        'ld.lld': LinkerInfo(name='ld.lld', type=LinkerType.LLD, cross_platform=True),
        'ld64': LinkerInfo(name='ld64', type=LinkerType.LD64, platforms=['darwin']),
        'lld-link': LinkerInfo(name='lld-link', type=LinkerType.LLD_LINK, platforms=['windows']),
        'ld': LinkerInfo(name='ld', type=LinkerType.GNU_LD, platforms=['linux']),
        'lld': LinkerInfo(name='lld', type=LinkerType.LLD, cross_platform=True),
    }

    def get_linker(self, target: TargetTriple) -> LinkerInfo:
        """获取适合目标的链接器"""
        # 优先使用 LLD（跨平台）
        if self._is_lld_available():
            return self._get_lld_for_target(target)

        # 回退到平台默认链接器
        return self._get_platform_linker(target)

    def _get_lld_for_target(self, target: TargetTriple) -> LinkerInfo:
        """获取 LLD 链接器"""
        if target.os == OperatingSystem.LINUX:
            return self.LINKERS['ld.lld']
        elif target.os == OperatingSystem.DARWIN:
            return self.LINKERS['ld64']
        elif target.os == OperatingSystem.WINDOWS:
            return self.LINKERS['lld-link']
        else:
            return self.LINKERS['lld']
```

#### 任务 3.2: 链接命令生成
```python
def generate_link_command(self, objects: List[str], output: str,
                          target: TargetTriple, options: LinkOptions) -> List[str]:
    """生成链接命令"""
    linker = self.get_linker(target)
    cmd = [linker.name]

    # 添加目标特定选项
    if target.os == OperatingSystem.LINUX:
        cmd.extend([
            '-dynamic-linker', self._get_dynamic_linker(target),
            '-L', '/lib',
            '-L', '/usr/lib',
        ])
        if options.sysroot:
            cmd.extend(['--sysroot', options.sysroot])

    elif target.os == OperatingSystem.DARWIN:
        cmd.extend([
            '-syslibroot', options.sysroot or self._get_sdk_path(),
            '-lSystem',
        ])

    elif target.os == OperatingSystem.WINDOWS:
        cmd.extend([
            '/subsystem:console',
            '/entry:mainCRTStartup',
        ])

    # 添加对象文件
    cmd.extend(objects)

    # 添加输出文件
    cmd.extend(['-o', output])

    # 添加库
    for lib in options.libraries:
        cmd.extend(['-l', lib])

    return cmd
```

```python
class RuntimeManager:
    """运行时管理器"""

    def __init__(self):
        self.runtimes: Dict[str, RuntimeLibrary] = {}

    def get_runtime(self, target: TargetTriple,
                    runtime_type: str = "default") -> RuntimeLibrary:
        """获取运行时库"""
        key = f"{target.triple}:{runtime_type}"
        if key in self.runtimes:
            return self.runtimes[key]

        return self._detect_runtime(target, runtime_type)

    def _detect_runtime(self, target: TargetTriple,
                        runtime_type: str) -> RuntimeLibrary:
        """检测运行时库"""
        if target.os == OperatingSystem.LINUX:
            if runtime_type == "musl":
                return self._get_musl_runtime(target)
            else:
                return self._get_glibc_runtime(target)

        elif target.os == OperatingSystem.DARWIN:
            return self._get_darwin_runtime(target)

        elif target.os == OperatingSystem.WINDOWS:
            if runtime_type == "mingw":
                return self._get_mingw_runtime(target)
            else:
                return self._get_msvc_runtime(target)

        elif target.os == OperatingSystem.UNKNOWN:
            # WebAssembly 或嵌入式
            return self._get_minimal_runtime(target)

    def get_crt_files(self, target: TargetTriple,
                      link_type: LinkType) -> List[str]:
        """获取 CRT 文件"""
        runtime = self.get_runtime(target)
        return runtime.get_crt_files(link_type)
```

```python
class MultiTargetCompiler:
    """多目标编译器"""

    def __init__(self, targets: List[str]):
        self.targets = [TargetManager.parse_triple(t) for t in targets]
        self.managers = {
            t.triple: CrossCompileManager().configure_target(t.triple)
            for t in self.targets
        }

    def compile_all(self, source: str, output_dir: str,
                    options: CompileOptions = None):
        """为所有目标编译"""
        results = {}
        for target in self.targets:
            output_name = self._get_output_name(source, target, output_dir)
            result = self.compile_for_target(source, target, output_name, options)
            results[target.triple] = result
        return results

    def compile_for_target(self, source: str, target: TargetTriple,
                           output: str, options: CompileOptions = None):
        """为单个目标编译"""
        manager = self.managers[target.triple]
        compiler = manager.create_compiler()
        return compiler.compile(source, output, options)
```

```python
        """设置目标平台"""
        self.cross_manager.configure_target(target)

    def set_sysroot(self, path: str):
        """设置 sysroot"""
        self.cross_manager.sysroot_manager.set_sysroot(self.target, path)

    def set_toolchain(self, path: str):
        """设置工具链"""
        self.cross_manager.toolchain_manager.register_toolchain(
            self.target, path
        )

    def get_available_targets(self) -> List[str]:
        """获取可用目标列表"""
        return TargetManager.list_available_targets()
```

### 4.2 命令行接口
```bash
# 指定目标
zhc --target=aarch64-unknown-linux-gnu input.zhc -o output

# 设置 sysroot
zhc --target=arm-linux-gnueabihf --sysroot=/opt/arm-sysroot input.zhc

# 设置工具链
zhc --target=riscv64-unknown-linux-gnu --toolchain=/opt/riscv-toolchain input.zhc

# 列出可用目标
zhc --list-targets

# 显示目标信息
zhc --target-info=x86_64-unknown-linux-gnu

# 多目标编译
zhc --targets=x86_64,aarch64,wasm32 input.zhc -o dist/
```

## 6. 风险与挑战

### 6.1 技术风险
| 风险 | 影响 | 缓解策略 |
|------|------|----------|
| 工具链兼容性 | 高 | 使用 LLD 作为默认链接器 |
| Sysroot 缺失 | 中 | 提供自动下载功能 |
| ABI 不兼容 | 高 | 严格遵循目标 ABI |

### 6.2 已知限制
1. **Windows 交叉编译**：需要 MSVC SDK
2. **嵌入式目标**：可能需要额外的运行时支持
3. **调试困难**：交叉编译的调试需要远程调试器
