## 功能概述

**功能名称**：泛型类型参数推导  
**优先级**：P1  
**功能模块**：类型系统  
**创建日期**：2026-04-09

---

## 1. 需求分析

### 1.1 当前状态

ZhC 编译器已有基础泛型支持：
- `src/zhc/generics/` 模块
- 泛型函数定义语法
- 基础泛型实例化

**现有问题**：
1. 泛型类型参数不能自动推导
2. 调用泛型函数必须显式指定类型参数
3. 泛型约束不完整
4. 泛型特化支持有限

### 1.2 目标状态

实现完整的泛型类型参数自动推导：

```zhc
// 无需显式指定类型参数
泛型<T> T 取最大值(T a, T b) {
    如果 (a > b) {
        返回 a;
    }
    返回 b;
}

整数型 主函数() {
    整数型 x = 取最大值(10, 20);      // T 自动推导为 整数型
    浮点型 y = 取最大值(3.14, 2.71);  // T 自动推导为 浮点型
    
    // 复杂推导
    自动 z = 取最大值(x, y);          // 错误：类型不一致
    
    返回 0;
}

// 泛型结构体
泛型<T> 结构体 容器 {
    T 元素;
    整数型 长度;
}

整数型 主函数() {
    容器<整数型> 整型容器;
    容器<字符串型> 字符串容器;
    
    返回 0;
}
```

#### 2.2.1 泛型类型参数推导器

```python
class GenericTypeParamInferrer:
    """泛型类型参数推导器"""
    
    def __init__(self, type_engine: TypeInferenceEngine):
        self.engine = type_engine
        self.instantiation_cache: Dict[str, "GenericInstance"] = {}
    
    def infer_type_params(
        self,
        generic_func: GenericFunctionDecl,
        call_args: List[ASTNode],
        call_arg_types: List[str],
        explicit_params: Optional[List[str]] = None
    ) -> List[str]:
        """
        推导泛型函数调用的类型参数
        
        Args:
            generic_func: 泛型函数定义
            call_args: 调用参数 AST 节点
            call_arg_types: 调用参数类型列表
            explicit_params: 显式指定的类型参数（可选）
            
        Returns:
            推导出的类型参数列表
            
        Example:
            // 函数定义
            泛型<T, U> U 转换(T 值, U 默认值) { ... }
            
            // 调用
            整数型 x = 转换("42", 0);  // T=字符串型, U=整数型
        """
        
        type_params = generic_func.type_params
        inferred_params = []
        
        # 如果有显式类型参数，先填入
        if explicit_params:
            inferred_params = list(explicit_params)
        else:
            inferred_params = [None] * len(type_params)
        
        # 建立参数到类型参数的映射
        param_mapping: Dict[str, List[int]] = {}  # 参数类型 -> 类型参数索引
        
        for i, (type_param, param_type) in enumerate(zip(generic_func.parameters, call_arg_types)):
            if i < len(inferred_params) and inferred_params[i] is not None:
                # 显式指定的参数已确定
                continue
            
            # 尝试推导
            if self._is_concrete_type(param_type):
                inferred_params[i] = param_type
            elif self._contains_type_param(param_type, type_params):
                # 记录参数约束
                inferred_params[i] = self._infer_from_constraint(
                    param_type, type_params, call_arg_types
                )
        
        # 验证所有类型参数都已推导
        for i, param in enumerate(inferred_params):
            if param is None:
                # 尝试从返回值推导
                if generic_func.return_type:
                    inferred_params[i] = self._infer_from_return(
                        generic_func, call_arg_types, i
                    )
        
        # 填充未推导的类型参数为默认值
        for i, param in enumerate(inferred_params):
            if param is None:
                inferred_params[i] = type_params[i].default_type or "整数型"
        
        return inferred_params
    
    def _is_concrete_type(self, type_str: str) -> bool:
        """判断是否为具体类型（非泛型参数）"""
        # 排除包含泛型参数的表示
        return not type_str.startswith("T") or type_str in CONCRETE_TYPES
    
    def _contains_type_param(
        self,
        type_str: str,
        type_params: List[TypeParam]
    ) -> bool:
        """检查类型是否包含泛型参数"""
        param_names = {tp.name for tp in type_params}
        return any(p in param_names for p in type_str.split())
    
    def _infer_from_constraint(
        self,
        param_type: str,
        type_params: List[TypeParam],
        arg_types: List[str]
    ) -> Optional[str]:
        """从约束推导类型"""
        
        for tp in type_params:
            if tp.name == param_type:
                constraints = tp.constraints
                
                # 如果只有一个具体约束，直接使用
                if len(constraints) == 1:
                    return constraints[0]
                
                # 从参数上下文推导
                for arg_type in arg_types:
                    if arg_type != param_type and arg_type in constraints:
                        return arg_type
                
                return None
        
        return None
    
    def _infer_from_return(
        self,
        generic_func: GenericFunctionDecl,
        arg_types: List[str],
        param_index: int
    ) -> Optional[str]:
        """从返回值推导类型参数"""
        
        if not generic_func.return_type:
            return None
        
        return_type = generic_func.return_type
        
        # 如果返回类型包含该类型参数
        if return_type == type_params[param_index].name:
            # 从参数推导返回类型
            for i, arg_type in enumerate(arg_types):
                if i < len(generic_func.parameters):
                    param_type = generic_func.parameters[i].param_type
                    if param_type == return_type:
                        return arg_type
        
        return None
```

#### 2.2.2 泛型约束系统

```python
@dataclass
class TypeConstraint:
    """泛型类型约束"""
    
    type_param: str                    # 类型参数名
    constraint_type: str               # 约束类型
    methods: List[str] = field(default_factory=list)  # 必须支持的方法
    operators: List[str] = field(default_factory=list)  # 必须支持的运算符


@dataclass 
class TypeParam:
    """泛型类型参数"""
    
    name: str                          # 参数名，如 T, U
    constraints: List[str] = field(default_factory=list)  # 类型约束
    default_type: Optional[str] = None # 默认类型
    variance: str = "不变"            # 协变/逆变/不变


class GenericConstraintSolver:
    """泛型约束求解器"""
    
    # 标准约束
    STANDARD_CONSTRAINTS = {
        "可比较": ["整数型", "浮点型", "字符型"],
        "可相等": ["整数型", "浮点型", "字符型", "布尔型", "字符串型"],
        "可运算": ["整数型", "浮点型"],
    }
    
    def solve(
        self,
        type_params: List[TypeParam],
        call_arg_types: List[str]
    ) -> Dict[str, str]:
        """
        求解泛型约束
        
        Returns:
            类型参数到具体类型的映射
        """
        
        solutions = {}
        
        for tp in type_params:
            if tp.name in solutions:
                continue
            
            # 检查约束
            valid_types = self._get_valid_types(tp.constraints)
            
            if not valid_types:
                # 无约束，使用第一个兼容的调用参数类型
                for arg_type in call_arg_types:
                    if arg_type not in ("未知", "空"):
                        solutions[tp.name] = arg_type
                        break
            else:
                # 有约束，选择最具体的匹配
                for arg_type in call_arg_types:
                    if arg_type in valid_types:
                        solutions[tp.name] = arg_type
                        break
        
        return solutions
    
    def _get_valid_types(self, constraints: List[str]) -> List[str]:
        """获取满足约束的类型列表"""
        valid = set()
        
        for constraint in constraints:
            if constraint in self.STANDARD_CONSTRAINTS:
                valid.update(self.STANDARD_CONSTRAINTS[constraint])
            else:
                valid.add(constraint)
        
        return list(valid)
    
    def validate_constraint(
        self,
        type_param: TypeParam,
        actual_type: str
    ) -> bool:
        """验证类型是否满足约束"""
        
        if not type_param.constraints:
            return True
        
        valid_types = self._get_valid_types(type_param.constraints)
        
        return actual_type in valid_types or self._has_all_methods(
            actual_type, type_param.methods
        )
```

#### 2.2.3 泛型函数实例化器增强

```python
class GenericInstantiator:
    """泛型实例化器 - 增强版"""
    
    def __init__(self):
        self.cache: Dict[str, GenericFunctionInstance] = {}
        self.type_param_inferrer = GenericTypeParamInferrer()
    
    def instantiate_call(
        self,
        generic_func: GenericFunctionDecl,
        call_args: List[ASTNode],
        location: SourceLocation
    ) -> GenericFunctionInstance:
        """
        实例化泛型函数调用
        
        自动推导类型参数并生成实例化后的函数
        """
        
        # 1. 推导参数类型
        call_arg_types = []
        for arg in call_args:
            arg_type = self._infer_arg_type(arg)
            call_arg_types.append(arg_type)
        
        # 2. 推导泛型类型参数
        inferred_params = self.type_param_inferrer.infer_type_params(
            generic_func,
            call_args,
            call_arg_types
        )
        
        # 3. 生成缓存键
        cache_key = self._generate_cache_key(generic_func.name, inferred_params)
        
        # 4. 检查缓存
        if cache_key in self.cache:
            return self.cache[cache_key]
        
        # 5. 实例化
        instance = self._create_instance(
            generic_func,
            inferred_params,
            call_arg_types
        )
        
        # 6. 缓存
        self.cache[cache_key] = instance
        
        return instance
    
    def _create_instance(
        self,
        generic_func: GenericFunctionDecl,
        type_params: List[str],
        arg_types: List[str]
    ) -> GenericFunctionInstance:
        """创建实例化后的函数"""
        
        # 1. 替换类型参数
        type_mapping = dict(zip(
            [tp.name for tp in generic_func.type_params],
            type_params
        ))
        
        # 2. 复制并替换类型
        new_parameters = []
        for param, arg_type in zip(generic_func.parameters, arg_types):
            new_param = copy.deepcopy(param)
            new_param.param_type = type_mapping.get(param.param_type, param.param_type)
            new_parameters.append(new_param)
        
        # 3. 处理返回类型
        return_type = type_mapping.get(
            generic_func.return_type,
            generic_func.return_type
        )
        
        # 4. 复制并替换函数体
        new_body = self._replace_types_in_body(
            generic_func.body,
            type_mapping
        )
        
        # 5. 创建实例
        instance = GenericFunctionInstance(
            original=generic_func,
            type_params=type_params,
            parameters=new_parameters,
            return_type=return_type,
            body=new_body,
            is_inferred=True
        )
        
        return instance
    
    def _replace_types_in_body(
        self,
        body: ASTNode,
        type_mapping: Dict[str, str]
    ) -> ASTNode:
        """替换函数体中的类型"""
        
        def replace_node(node: ASTNode) -> ASTNode:
            new_node = copy.deepcopy(node)
            
            # 替换节点类型
            if hasattr(new_node, 'node_type'):
                if new_node.node_type in type_mapping:
                    new_node.node_type = type_mapping[new_node.node_type]
            
            # 递归处理子节点
            for child in new_node.get_children():
                replace_node(child)
            
            return new_node
        
        return replace_node(body)
```

### 2.3 编译流程集成

```python
class SemanticAnalyzer:
    """语义分析器 - 集成泛型推导"""
    
    def __init__(self):
        self.generic_instantiator = GenericInstantiator()
    
    def analyze_call_expr(
        self,
        node: CallExprNode
    ) -> str:
        """分析函数调用（包含泛型推导）"""
        
        callee = node.callee
        
        # 检查是否为泛型函数
        if isinstance(callee, IdentifierExprNode):
            symbol = self.symbol_table.lookup(callee.name)
            
            if symbol and symbol.is_generic:
                # 泛型函数调用，需要推导
                instance = self.generic_instantiator.instantiate_call(
                    symbol.generic_func,
                    node.args,
                    SourceLocation(line=node.line, column=node.column)
                )
                
                # 替换调用目标
                node.callee = instance.get_call_target()
                node.inferred_return_type = instance.return_type
                
                # 注册实例（如果需要）
                self._register_generic_instance(instance)
                
                print(f"[泛型推导] {callee.name}<{', '.join(instance.type_params)}>")
                
                return instance.return_type
        
        # 普通函数调用
        return super().analyze_call_expr(node)
    
    def _register_generic_instance(
        self,
        instance: GenericFunctionInstance
    ) -> None:
        """注册泛型实例"""
        
        instance_key = f"{instance.original.name}_{'_'.join(instance.type_params)}"
        
        # 添加到符号表
        instance_symbol = Symbol(
            name=instance_key,
            symbol_type="函数",
            data_type="泛型实例",
            is_defined=True
        )
        
        self.symbol_table.global_scope.add_symbol(instance_symbol)
```
