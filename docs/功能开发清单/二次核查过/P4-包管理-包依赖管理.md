## 1. 需求分析

### 1.1 功能目标

实现现代化的包依赖管理系统，支持：

1. **依赖声明**: 在项目配置文件中声明依赖
2. **依赖解析**: 自动解析依赖关系和版本约束
3. **依赖下载**: 从仓库下载依赖包
4. **依赖锁定**: 锁定依赖版本，确保可重复构建
5. **依赖更新**: 检查和更新依赖版本

**示例配置文件** (`zhc.json`):
```json
{
  "name": "我的项目",
  "version": "1.0.0",
  "dependencies": {
    "网络库": "^2.1.0",
    "数据库驱动": ">=1.5.0 <2.0.0",
    "日志工具": "~1.2.3"
  },
  "devDependencies": {
    "测试框架": "^1.0.0",
    "文档生成器": "^0.5.0"
  }
}
```

### 1.2 依赖管理规范

参考主流包管理器设计：

| 特性 | 说明 |
|-----|------|
| 版本约束 | 支持语义化版本约束（^, ~, >=, <等） |
| 依赖解析 | 解析传递依赖，处理版本冲突 |
| 锁定文件 | `zhc.lock` 锁定确切版本 |
| 缓存机制 | 本地缓存已下载的包 |
| 私有仓库 | 支持配置私有包仓库 |
| 离线模式 | 优先使用本地缓存 |

### 1.3 边界情况

| 边界情况 | 处理方式 |
|---------|---------|
| 版本冲突 | 报告冲突详情，提供解决建议 |
| 循环依赖 | 检测并报错 |
| 网络错误 | 使用本地缓存，重试机制 |
| 包不存在 | 清晰的错误提示 |
| 版本不兼容 | 列出兼容版本供选择 |
| 损坏的包 | 校验哈希，重新下载 |

## 3. 实现方案

### 3.1 设计决策

**问题**: 包管理器应该如何集成到编译流程？

**选择**: 分离设计 - 包管理器作为独立工具

**理由**:
1. 包管理和编译是独立的阶段
2. 便于独立测试和调试
3. 支持离线编译
4. 与现有工具链集成

**工作流程**:
```
开发阶段:
zhc init          # 初始化项目
zhc add 网络库    # 添加依赖
zhc install       # 安装依赖
zhc build         # 编译项目

编译阶段:
zhc build         # 检查依赖 → 编译
```

from typing import Dict, Optional
from dataclasses import dataclass, field
from pathlib import Path
import json

@dataclass
class DependencySpec:
    """依赖规格"""
    name: str
    version_constraint: str
    source: Optional[str] = None  # 包来源（仓库URL）
    
@dataclass
class ProjectConfig:
    """项目配置"""
    name: str
    version: str
    dependencies: Dict[str, str] = field(default_factory=dict)
    dev_dependencies: Dict[str, str] = field(default_factory=dict)
    source: Optional[str] = None  # 默认包仓库
    
    @classmethod
    def from_file(cls, path: Path) -> 'ProjectConfig':
        """从配置文件加载"""
        with open(path, 'r', encoding='utf-8') as f:
            data = json.load(f)
        
        return cls(
            name=data['name'],
            version=data['version'],
            dependencies=data.get('dependencies', {}),
            dev_dependencies=data.get('devDependencies', {}),
            source=data.get('source')
        )
    
    def to_file(self, path: Path):
        """保存到配置文件"""
        data = {
            'name': self.name,
            'version': self.version,
            'dependencies': self.dependencies,
            'devDependencies': self.dev_dependencies,
        }
        
        if self.source:
            data['source'] = self.source
        
        with open(path, 'w', encoding='utf-8') as f:
            json.dump(data, f, indent=2, ensure_ascii=False)

```python
# src/package/version.py

"""
语义化版本约束处理

支持格式：
- ^1.2.3  兼容版本（>=1.2.3 <2.0.0）
- ~1.2.3  近似版本（>=1.2.3 <1.3.0）
- >=1.2.3 范围约束
- 1.2.3   精确版本
"""

import re
from typing import List, Tuple, Optional
from dataclasses import dataclass
from functools import total_ordering

@dataclass
@total_ordering
class Version:
    """语义化版本"""
    major: int
    minor: int
    patch: int
    prerelease: Optional[str] = None
    
    @classmethod
    def parse(cls, version_str: str) -> 'Version':
        """解析版本字符串"""
        # 格式: 1.2.3 或 1.2.3-beta
        match = re.match(r'(\d+)\.(\d+)\.(\d+)(?:-(.+))?', version_str)
        if not match:
            raise ValueError(f"无效的版本格式: {version_str}")
        
        return cls(
            major=int(match.group(1)),
            minor=int(match.group(2)),
            patch=int(match.group(3)),
            prerelease=match.group(4)
        )
    
    def __str__(self) -> str:
        base = f"{self.major}.{self.minor}.{self.patch}"
        if self.prerelease:
            base += f"-{self.prerelease}"
        return base
    
    def __lt__(self, other: 'Version') -> bool:
        if not isinstance(other, Version):
            return NotImplemented
        
        # 比较主版本号
        if self.major != other.major:
            return self.major < other.major
        
        # 比较次版本号
        if self.minor != other.minor:
            return self.minor < other.minor
        
        # 比较补丁版本号
        if self.patch != other.patch:
            return self.patch < other.patch
        
        # 预发布版本 < 正式版本
        if self.prerelease and not other.prerelease:
            return True
        if not self.prerelease and other.prerelease:
            return False
        
        return (self.prerelease or '') < (other.prerelease or '')
    
    def __eq__(self, other: object) -> bool:
        if not isinstance(other, Version):
            return NotImplemented
        return (self.major, self.minor, self.patch, self.prerelease) == \
               (other.major, other.minor, other.patch, other.prerelease)

class VersionConstraint:
    """版本约束"""
    
    def __init__(self, constraint_str: str):
        self.original = constraint_str
        self.constraints = self._parse(constraint_str)
    
    def _parse(self, constraint_str: str) -> List[Tuple[str, Version]]:
        """解析约束字符串"""
        constraints = []
        
        # ^1.2.3 -> >=1.2.3 <2.0.0
        if constraint_str.startswith('^'):
            version = Version.parse(constraint_str[1:])
            constraints.append(('>=', version))
            constraints.append(('<', Version(version.major + 1, 0, 0)))
        
        # ~1.2.3 -> >=1.2.3 <1.3.0
        elif constraint_str.startswith('~'):
            version = Version.parse(constraint_str[1:])
            constraints.append(('>=', version))
            constraints.append(('<', Version(version.major, version.minor + 1, 0)))
        
        # >=1.2.3, <2.0.0
        elif any(constraint_str.startswith(op) for op in ['>=', '<=', '>', '<', '=']):
            match = re.match(r'(>=|<=|>|<|=)(.+)', constraint_str)
            if match:
                op, ver_str = match.groups()
                version = Version.parse(ver_str.strip())
                constraints.append((op, version))
        
        # 精确版本
        else:
            version = Version.parse(constraint_str)
            constraints.append(('=', version))
        
        return constraints
    
    def matches(self, version: Version) -> bool:
        """检查版本是否满足约束"""
        for op, constraint_version in self.constraints:
            if op == '>=' and not (version >= constraint_version):
                return False
            elif op == '>' and not (version > constraint_version):
                return False
            elif op == '<=' and not (version <= constraint_version):
                return False
            elif op == '<' and not (version < constraint_version):
                return False
            elif op == '=' and not (version == constraint_version):
                return False
        
        return True
```

```python
# src/package/dependency.py

"""
依赖解析器

使用回溯算法解析依赖关系，处理版本冲突
"""

from typing import Dict, List, Set, Optional
from dataclasses import dataclass
from collections import defaultdict

@dataclass
class ResolvedDependency:
    """已解析的依赖"""
    name: str
    version: Version
    source: Optional[str]
    dependencies: Dict[str, 'ResolvedDependency']

class DependencyResolver:
    """依赖解析器"""
    
    def __init__(self, repository: 'PackageRepository'):
        self.repository = repository
        self.resolved: Dict[str, ResolvedDependency] = {}
        self.conflicts: List[str] = []
    
    def resolve(
        self,
        dependencies: Dict[str, str]
    ) -> Dict[str, ResolvedDependency]:
        """解析依赖关系
        
        Args:
            dependencies: 依赖字典 {包名: 版本约束}
            
        Returns:
            解析后的依赖树
            
        Raises:
            DependencyConflictError: 版本冲突
        """
        # 清空之前的结果
        self.resolved.clear()
        self.conflicts.clear()
        
        # 解析每个依赖
        for name, constraint in dependencies.items():
            self._resolve_dependency(name, constraint, set())
        
        # 检查是否有冲突
        if self.conflicts:
            raise DependencyConflictError(
                f"依赖版本冲突: {', '.join(self.conflicts)}"
            )
        
        return self.resolved
    
    def _resolve_dependency(
        self,
        name: str,
        constraint: str,
        visited: Set[str]
    ):
        """递归解析依赖"""
        
        # 检测循环依赖
        if name in visited:
            raise CyclicDependencyError(f"检测到循环依赖: {name}")
        
        visited.add(name)
        
        # 检查是否已经解析
        if name in self.resolved:
            existing = self.resolved[name]
            version_constraint = VersionConstraint(constraint)
            
            # 检查版本是否兼容
            if not version_constraint.matches(existing.version):
                self.conflicts.append(
                    f"{name}: 需要 {constraint}, 已解析 {existing.version}"
                )
            return
        
        # 从仓库查询可用版本
        available_versions = self.repository.get_versions(name)
        if not available_versions:
            raise PackageNotFoundError(f"包 '{name}' 不存在")
        
        # 选择满足约束的最新版本
        version_constraint = VersionConstraint(constraint)
        selected_version = None
        
        for version in sorted(available_versions, reverse=True):
            if version_constraint.matches(version):
                selected_version = version
                break
        
        if not selected_version:
            raise VersionNotFoundError(
                f"包 '{name}' 没有满足约束 {constraint} 的版本"
            )
        
        # 创建解析结果
        resolved_dep = ResolvedDependency(
            name=name,
            version=selected_version,
            source=self.repository.get_source(name, selected_version),
            dependencies={}
        )
        
        # 注册到已解析列表
        self.resolved[name] = resolved_dep
        
        # 获取该包的依赖
        package_deps = self.repository.get_dependencies(name, selected_version)
        
        # 递归解析传递依赖
        for dep_name, dep_constraint in package_deps.items():
            self._resolve_dependency(dep_name, dep_constraint, visited.copy())
            resolved_dep.dependencies[dep_name] = self.resolved[dep_name]
```

# src/package/manager.py

"""
包管理器

提供完整的包管理功能：安装、更新、卸载
"""

from typing import Dict, Optional
from pathlib import Path
import shutil
import hashlib

class PackageManager:
    """包管理器"""
    
    def __init__(
        self,
        project_root: Path,
        repository: Optional['PackageRepository'] = None
    ):
        self.project_root = project_root
        self.config_path = project_root / 'zhc.json'
        self.lock_path = project_root / 'zhc.lock'
        self.packages_dir = project_root / 'packages'
        self.cache_dir = Path.home() / '.zhc' / 'cache'
        
        # 初始化仓库
        self.repository = repository or DefaultRepository()
        
        # 加载配置
        self.config: Optional[ProjectConfig] = None
        self.lockfile: Optional[Lockfile] = None
        
        if self.config_path.exists():
            self.config = ProjectConfig.from_file(self.config_path)
        
        if self.lock_path.exists():
            self.lockfile = Lockfile.from_file(self.lock_path)
    
    def init(self, name: str, version: str = "0.1.0"):
        """初始化项目"""
        
        if self.config_path.exists():
            raise PackageError(f"项目已存在: {self.config_path}")
        
        # 创建配置文件
        self.config = ProjectConfig(
            name=name,
            version=version
        )
        self.config.to_file(self.config_path)
        
        # 创建目录结构
        self.packages_dir.mkdir(exist_ok=True)
        
        print(f"✓ 初始化项目 '{name}' 成功")
    
    def add(self, package_name: str, version: str = "latest"):
        """添加依赖"""
        
        if not self.config:
            raise PackageError("未找到项目配置文件，请先运行 'zhc init'")
        
        # 查询最新版本
        if version == "latest":
            versions = self.repository.get_versions(package_name)
            if not versions:
                raise PackageNotFoundError(f"包 '{package_name}' 不存在")
            version = str(max(versions))
        
        # 添加到配置
        self.config.dependencies[package_name] = version
        self.config.to_file(self.config_path)
        
        print(f"✓ 添加依赖: {package_name} {version}")
        
        # 安装依赖
        self.install()
    
    def install(self, offline: bool = False):
        """安装所有依赖"""
        
        if not self.config:
            raise PackageError("未找到项目配置文件")
        
        print("正在解析依赖...")
        
        # 解析依赖
        resolver = DependencyResolver(self.repository)
        resolved = resolver.resolve(self.config.dependencies)
        
        print(f"解析完成，共 {len(resolved)} 个依赖")
        
        # 创建锁定文件
        self.lockfile = Lockfile.from_resolved(resolved)
        self.lockfile.to_file(self.lock_path)
        
        # 下载并安装包
        for name, dep in resolved.items():
            self._install_package(dep, offline=offline)
        
        print(f"✓ 安装完成")
    
    def _install_package(
        self,
        dep: ResolvedDependency,
        offline: bool = False
    ):
        """安装单个包"""
        
        target_dir = self.packages_dir / dep.name
        
        # 检查是否已安装
        if target_dir.exists():
            # 验证版本
            installed_version = self._get_installed_version(target_dir)
            if installed_version == dep.version:
                print(f"  ✓ {dep.name} {dep.version} (已安装)")
                return
        
        print(f"  → 安装 {dep.name} {dep.version}...")
        
        # 检查缓存
        cached_path = self._get_cached_path(dep.name, dep.version)
        
        if cached_path.exists():
            # 从缓存安装
            print(f"    从缓存安装")
            shutil.copytree(cached_path, target_dir)
        elif not offline:
            # 从仓库下载
            print(f"    从仓库下载")
            self.repository.download(dep.name, dep.version, target_dir)
            
            # 添加到缓存
            self._add_to_cache(dep.name, dep.version, target_dir)
        else:
            raise PackageError(
                f"离线模式下无法安装 {dep.name} {dep.version}，"
                f"请先在线安装或检查缓存"
            )
        
        print(f"  ✓ {dep.name} {dep.version} 安装成功")
    
    def update(self, package_name: Optional[str] = None):
        """更新依赖"""
        
        if not self.config:
            raise PackageError("未找到项目配置文件")
        
        if package_name:
            # 更新单个包
            if package_name not in self.config.dependencies:
                raise PackageError(f"依赖 '{package_name}' 不在配置中")
            
            # 查询最新版本
            versions = self.repository.get_versions(package_name)
            latest = max(versions)
            
            # 更新版本
            self.config.dependencies[package_name] = str(latest)
            self.config.to_file(self.config_path)
            
            print(f"✓ 更新 {package_name} 到 {latest}")
        else:
            # 更新所有依赖
            print("检查依赖更新...")
            
            for name in self.config.dependencies:
                versions = self.repository.get_versions(name)
                latest = max(versions)
                current = self.config.dependencies[name]
                
                if str(latest) != current:
                    self.config.dependencies[name] = str(latest)
                    print(f"  {name}: {current} → {latest}")
        
        self.config.to_file(self.config_path)
        
        # 重新安装
        self.install()
    
    def remove(self, package_name: str):
        """移除依赖"""
        
        if not self.config:
            raise PackageError("未找到项目配置文件")
        
        if package_name not in self.config.dependencies:
            raise PackageError(f"依赖 '{package_name}' 不在配置中")
        
        # 从配置中移除
        del self.config.dependencies[package_name]
        self.config.to_file(self.config_path)
        
        # 删除包文件
        target_dir = self.packages_dir / package_name
        if target_dir.exists():
            shutil.rmtree(target_dir)
        
        print(f"✓ 移除依赖: {package_name}")
    
    def _get_cached_path(self, name: str, version: Version) -> Path:
        """获取缓存路径"""
        return self.cache_dir / name / str(version)
    
    def _add_to_cache(self, name: str, version: Version, source: Path):
        """添加到缓存"""
        cache_path = self._get_cached_path(name, version)
        cache_path.parent.mkdir(parents=True, exist_ok=True)
        shutil.copytree(source, cache_path)
    
    def _get_installed_version(self, package_dir: Path) -> Optional[Version]:
        """获取已安装包的版本"""
        manifest_path = package_dir / 'zhc.json'
        if not manifest_path.exists():
            return None
        
        with open(manifest_path, 'r', encoding='utf-8') as f:
            import json
            data = json.load(f)
            return Version.parse(data.get('version', '0.0.0'))
```

# tests/package/test_version.py

def test_version_parsing():
    """测试版本解析"""
    v1 = Version.parse("1.2.3")
    assert v1.major == 1
    assert v1.minor == 2
    assert v1.patch == 3
    
    v2 = Version.parse("2.0.0-beta")
    assert v2.prerelease == "beta"

def test_version_comparison():
    """测试版本比较"""
    v1 = Version.parse("1.2.3")
    v2 = Version.parse("1.2.4")
    v3 = Version.parse("2.0.0")
    
    assert v1 < v2
    assert v2 < v3
    assert v1 < v3

def test_version_constraint_caret():
    """测试 ^ 约束"""
    constraint = VersionConstraint("^1.2.3")
    
    assert constraint.matches(Version.parse("1.2.3"))
    assert constraint.matches(Version.parse("1.3.0"))
    assert constraint.matches(Version.parse("1.9.9"))
    assert not constraint.matches(Version.parse("2.0.0"))
    assert not constraint.matches(Version.parse("1.2.2"))

def test_version_constraint_tilde():
    """测试 ~ 约束"""
    constraint = VersionConstraint("~1.2.3")
    
    assert constraint.matches(Version.parse("1.2.3"))
    assert constraint.matches(Version.parse("1.2.9"))
    assert not constraint.matches(Version.parse("1.3.0"))
    assert not constraint.matches(Version.parse("2.0.0"))
```
