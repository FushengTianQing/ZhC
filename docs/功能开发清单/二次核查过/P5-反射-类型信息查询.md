# P5-反射-类型信息查询

> **优先级**: P5
> **功能模块**: 反射
> **功能名称**: 类型信息查询
> **创建日期**: 2026-04-10
> **状态**: ✅ 已完成 (2026-04-11)

## 1. 功能概述

实现运行时类型信息（RTTI）查询，允许程序在运行时获取类型名称、大小、字段、方法等元信息。

### 1.1 目标

- 支持类型名称查询
- 支持类型大小查询
- 支持字段列表查询
- 支持方法列表查询
- 支持类型继承关系查询

### 1.2 语法设计

```zhc
// 类型信息查询
类型信息 info = 获取类型信息(变量);
打印(info.名称);           // "整数型"
打印(info.大小);           // 4
打印(info.是基本类型);     // true

// 对象类型查询
类 动物 {
    字符串 名称;
    函数型 无 -> 无 叫声;
}

类 狗 继承 动物 {
    整数型 年龄;
    
    函数 叫() {
        打印("汪汪!");
    }
}

狗 小狗 = 新 狗();
类型信息 类型 = 获取类型信息(小狗);
打印(类型.名称);           // "狗"
打印(类型.父类);           // "动物"
打印(类型.字段);           // ["名称", "叫声", "年龄"]
print(类型.方法);          // ["叫"]

// 字段查询
字段信息 字段 = 类型.获取字段("年龄");
打印(字段.名称);           // "年龄"
打印(字段.类型);           // "整数型"
打印(字段.偏移量);         // 16

// 动态访问字段
设置字段值(小狗, "年龄", 3);
整数型 年龄 = 获取字段值(小狗, "年龄");
```

### 3.2 核心数据结构

```python
# src/zhc/reflection/type_info.py

from dataclasses import dataclass, field
from typing import List, Optional, Dict, Any

@dataclass
class FieldInfo:
    """字段信息"""
    name: str
    type_name: str
    offset: int
    size: int
    is_public: bool = False
    is_static: bool = False
    is_const: bool = False
    
    def to_dict(self) -> Dict:
        return {
            "name": self.name,
            "type": self.type_name,
            "offset": self.offset,
            "size": self.size,
            "is_public": self.is_public,
            "is_static": self.is_static,
            "is_const": self.is_const,
        }

@dataclass
class MethodInfo:
    """方法信息"""
    name: str
    return_type: str
    params: List[Dict[str, str]]  # [{"name": "x", "type": "int"}, ...]
    is_static: bool = False
    is_virtual: bool = False
    vtable_index: Optional[int] = None
    
    def to_dict(self) -> Dict:
        return {
            "name": self.name,
            "return_type": self.return_type,
            "params": self.params,
            "is_static": self.is_static,
            "is_virtual": self.is_virtual,
        }

@dataclass
class TypeInfo:
    """类型信息"""
    name: str
    size: int
    alignment: int
    is_class: bool = False
    is_struct: bool = False
    is_union: bool = False
    is_enum: bool = False
    is_primitive: bool = False
    base_class: Optional[str] = None
    interfaces: List[str] = field(default_factory=list)
    fields: List[FieldInfo] = field(default_factory=list)
    methods: List[MethodInfo] = field(default_factory=list)
    constants: Dict[str, Any] = field(default_factory=dict)
    
    def get_field(self, name: str) -> Optional[FieldInfo]:
        """获取字段信息"""
        for field in self.fields:
            if field.name == name:
                return field
        return None
    
    def get_method(self, name: str) -> Optional[MethodInfo]:
        """获取方法信息"""
        for method in self.methods:
            if method.name == name:
                return method
        return None
    
    def is_assignable_from(self, other: 'TypeInfo') -> bool:
        """检查是否可以从 other 类型赋值"""
        if self.name == other.name:
            return True
        # 检查继承关系
        if other.base_class:
            return self.is_assignable_from(TypeRegistry.lookup(other.base_class))
        return False
    
    def to_dict(self) -> Dict:
        return {
            "name": self.name,
            "size": self.size,
            "alignment": self.alignment,
            "is_class": self.is_class,
            "is_struct": self.is_struct,
            "is_union": self.is_union,
            "is_enum": self.is_enum,
            "is_primitive": self.is_primitive,
            "base_class": self.base_class,
            "interfaces": self.interfaces,
            "fields": [f.to_dict() for f in self.fields],
            "methods": [m.to_dict() for m in self.methods],
            "constants": self.constants,
        }

class TypeRegistry:
    """类型注册表"""
    
    _types: Dict[str, TypeInfo] = {}
    
    @classmethod
    def register(cls, type_info: TypeInfo) -> None:
        """注册类型"""
        cls._types[type_info.name] = type_info
    
    @classmethod
    def lookup(cls, name: str) -> Optional[TypeInfo]:
        """查找类型"""
        return cls._types.get(name)
    
    @classmethod
    def is_registered(cls, name: str) -> bool:
        """检查类型是否已注册"""
        return name in cls._types
    
    @classmethod
    def get_all_types(cls) -> List[TypeInfo]:
        """获取所有类型"""
        return list(cls._types.values())
    
    @classmethod
    def get_subclasses(cls, base_name: str) -> List[TypeInfo]:
        """获取所有子类"""
        result = []
        for type_info in cls._types.values():
            if type_info.base_class == base_name:
                result.append(type_info)
        return result

def get_type_info(obj: Any) -> TypeInfo:
    """获取对象的类型信息"""
    type_name = type(obj).__name__
    type_info = TypeRegistry.lookup(type_name)
    if type_info is None:
        # 动态创建
        type_info = TypeInfo(
            name=type_name,
            size=0,  # 需要从实际对象计算
            alignment=0,
            is_primitive=type_name in ("int", "float", "double", "bool", "str"),
        )
    return type_info

def get_type_name(obj: Any) -> str:
    """获取对象类型名称"""
    return type(obj).__name__

def get_type_size(type_name: str) -> int:
    """获取类型大小"""
    type_info = TypeRegistry.lookup(type_name)
    return type_info.size if type_info else 0

def get_fields(obj: Any) -> List[FieldInfo]:
    """获取对象的所有字段"""
    type_info = get_type_info(obj)
    return type_info.fields

def get_field_value(obj: Any, field_name: str) -> Any:
    """获取字段值"""
    if hasattr(obj, field_name):
        return getattr(obj, field_name)
    return None

def set_field_value(obj: Any, field_name: str, value: Any) -> None:
    """设置字段值"""
    if hasattr(obj, field_name):
        setattr(obj, field_name, value)
```
