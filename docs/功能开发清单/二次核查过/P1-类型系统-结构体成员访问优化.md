# P1 类型系统 - 结构体成员访问优化

## 功能概述

**功能名称**：结构体成员访问优化  
**优先级**：P1  
**功能模块**：类型系统  
**创建日期**：2026-04-09

---

## 1. 需求分析

### 1.1 当前状态

ZhC 编译器已有基础结构体/类支持（`src/zhc/parser/class_.py`）：
- 类声明解析（`ClassParser`）
- 属性和方法定义
- 访问控制（公开/私有/保护）
- 构造函数和析构函数

LLVM 后端已有基础 GEP 支持（`src/zhc/backend/llvm_instruction_strategy.py`）：
- `GepStrategy`：处理 `Opcode.GEP` 指令
- 基础指针运算

**现有问题**：
1. 结构体成员访问生成的 LLVM IR 不够优化
2. 缺少结构体类型的 LLVM 类型定义
3. 成员偏移量计算未优化
4. 缺少结构体布局信息管理
5. 未利用 LLVM 的结构体优化特性（如 `inbounds` GEP）

### 1.2 目标状态

优化 LLVM IR 中的结构体访问：

```zhc
// 结构体定义
结构体 学生 {
    字符串型 姓名;
    整数型 年龄;
    浮点型 成绩;
}

// 成员访问
整数型 主函数() {
    学生 s;
    s.姓名 = "张三";
    s.年龄 = 20;
    s.成绩 = 85.5;
    
    自动 年龄 = s.年龄;  // 优化的 GEP 指令
    返回 年龄;
}
```

**优化目标**：
1. 生成正确的 LLVM 结构体类型
2. 使用 `inbounds` GEP 提升性能
3. 缓存成员偏移量信息
4. 支持嵌套结构体访问
5. 支持结构体数组访问

### 2.2 核心组件设计

#### 2.2.1 结构体布局计算器

**新增文件**：`src/zhc/semantic/struct_layout.py`

```python
from dataclasses import dataclass
from typing import Dict, List, Optional

@dataclass
class MemberLayout:
    """成员布局信息"""
    name: str           # 成员名
    type_name: str      # 类型名
    offset: int         # 字节偏移量
    size: int           # 字节大小
    index: int          # LLVM GEP 索引

@dataclass
class StructLayout:
    """结构体布局信息"""
    name: str           # 结构体名
    size: int           # 总字节大小
    alignment: int      # 对齐要求
    members: List[MemberLayout]
    member_map: Dict[str, MemberLayout]  # 成员名 -> 布局信息

class StructLayoutCalculator:
    """结构体布局计算器"""
    
    # 类型大小映射（字节）
    TYPE_SIZES = {
        "整数型": 4,
        "短整型": 2,
        "长整型": 8,
        "字符型": 1,
        "浮点型": 4,
        "双精度浮点型": 8,
        "布尔型": 4,
        "字符串型": 8,  # 指针大小
    }
    
    # 类型对齐要求
    TYPE_ALIGNMENTS = {
        "整数型": 4,
        "短整型": 2,
        "长整型": 8,
        "字符型": 1,
        "浮点型": 4,
        "双精度浮点型": 8,
        "布尔型": 4,
        "字符串型": 8,
    }
    
    def __init__(self):
        self._layouts: Dict[str, StructLayout] = {}
    
    def calculate_layout(self, struct_info: 'ClassInfo') -> StructLayout:
        """计算结构体布局"""
        members = []
        current_offset = 0
        max_alignment = 1
        
        for i, attr in enumerate(struct_info.attributes):
            # 获取类型大小和对齐
            type_size = self.TYPE_SIZES.get(attr.type_name, 4)
            type_align = self.TYPE_ALIGNMENTS.get(attr.type_name, 4)
            
            # 对齐当前偏移
            if current_offset % type_align != 0:
                current_offset += type_align - (current_offset % type_align)
            
            # 创建成员布局
            member_layout = MemberLayout(
                name=attr.name,
                type_name=attr.type_name,
                offset=current_offset,
                size=type_size,
                index=i  # LLVM GEP 使用成员索引
            )
            members.append(member_layout)
            
            # 更新偏移和最大对齐
            current_offset += type_size
            max_alignment = max(max_alignment, type_align)
        
        # 结构体尾部对齐
        if current_offset % max_alignment != 0:
            current_offset += max_alignment - (current_offset % max_alignment)
        
        # 创建布局信息
        layout = StructLayout(
            name=struct_info.name,
            size=current_offset,
            alignment=max_alignment,
            members=members,
            member_map={m.name: m for m in members}
        )
        
        self._layouts[struct_info.name] = layout
        return layout
    
    def get_layout(self, struct_name: str) -> Optional[StructLayout]:
        """获取结构体布局"""
        return self._layouts.get(struct_name)
    
    def get_member_offset(self, struct_name: str, member_name: str) -> Optional[int]:
        """获取成员偏移量"""
        layout = self.get_layout(struct_name)
        if layout:
            member = layout.member_map.get(member_name)
            if member:
                return member.offset
        return None
    
    def get_member_index(self, struct_name: str, member_name: str) -> Optional[int]:
        """获取成员 LLVM GEP 索引"""
        layout = self.get_layout(struct_name)
        if layout:
            member = layout.member_map.get(member_name)
            if member:
                return member.index
        return None
```

#### 2.2.2 LLVM 结构体类型映射

**修改文件**：`src/zhc/backend/llvm_type_mapper.py`

```python
import llvmlite.ir as ll
from typing import Dict, Optional, Any
from zhc.semantic.struct_layout import StructLayout, StructLayoutCalculator

class LLVMStructTypeMapper:
    """LLVM 结构体类型映射器"""
    
    def __init__(self, layout_calculator: StructLayoutCalculator):
        self.layout_calc = layout_calculator
        self._struct_types: Dict[str, ll.LiteralStructType] = {}
    
    def get_llvm_struct_type(
        self, 
        struct_name: str,
        module: Optional[ll.Module] = None
    ) -> Optional[ll.LiteralStructType]:
        """获取 LLVM 结构体类型"""
        
        # 检查缓存
        if struct_name in self._struct_types:
            return self._struct_types[struct_name]
        
        # 获取布局信息
        layout = self.layout_calc.get_layout(struct_name)
        if not layout:
            return None
        
        # 构建 LLVM 成员类型列表
        member_types = []
        for member in layout.members:
            member_type = self._get_member_llvm_type(member.type_name)
            member_types.append(member_type)
        
        # 创建 LLVM 结构体类型
        struct_type = ll.LiteralStructType(member_types, name=struct_name)
        
        # 缓存
        self._struct_types[struct_name] = struct_type
        
        return struct_type
    
    def _get_member_llvm_type(self, type_name: str) -> ll.Type:
        """获取成员 LLVM 类型"""
        from .type_system import get_type_mapper
        
        mapper = get_type_mapper()
        llvm_type = mapper.to_llvm(type_name)
        
        if llvm_type:
            return llvm_type
        
        # 默认返回 i32
        return ll.IntType(32)
    
    def create_struct_global(
        self,
        module: ll.Module,
        struct_name: str,
        struct_type: ll.LiteralStructType
    ) -> ll.GlobalVariable:
        """创建结构体全局变量模板"""
        # 创建未初始化的全局变量
        gv = ll.GlobalVariable(module, struct_type, f"_{struct_name}_template")
        gv.linkage = 'external'
        return gv
```

#### 2.2.3 优化 GEP 策略

**修改文件**：`src/zhc/backend/llvm_instruction_strategy.py`

```python
class StructGepStrategy(InstructionStrategy):
    """
    结构体成员访问 GEP 策略 - 优化版本
    
    使用方式：
        %member_ptr = struct_gep %struct_ptr, "成员名"
        或
        %member_ptr = struct_gep %struct_ptr, %index  (动态索引)
    
    优化：
    1. 使用 inbounds GEP 提升性能
    2. 缓存结构体布局信息
    3. 支持嵌套结构体访问
    """
    
    opcode = Opcode.STRUCT_GEP
    
    def compile(self, builder, instr, context):
        import llvmlite.ir as ll
        
        # 获取结构体指针
        struct_ptr = context.get_value(instr.operands[0])
        
        # 获取结构体类型信息
        struct_type_name = instr.metadata.get('struct_type')
        if not struct_type_name:
            # 从指针类型推断
            if hasattr(struct_ptr.type, 'pointee'):
                pointee = struct_ptr.type.pointee
                if hasattr(pointee, 'name'):
                    struct_type_name = pointee.name
        
        # 获取成员索引
        if len(instr.operands) > 1:
            # 动态索引
            member_index = context.get_value(instr.operands[1])
            indices = [ll.Constant(ll.IntType(32), 0), member_index]
        else:
            # 静态成员名
            member_name = instr.metadata.get('member_name')
            if struct_type_name and member_name:
                # 从布局计算器获取索引
                layout_calc = context.struct_layout_calculator
                index = layout_calc.get_member_index(struct_type_name, member_name)
                if index is not None:
                    indices = [
                        ll.Constant(ll.IntType(32), 0),  # 结构体指针解引用
                        ll.Constant(ll.IntType(32), index)  # 成员索引
                    ]
                else:
                    # 默认索引 0
                    indices = [ll.Constant(ll.IntType(32), 0), ll.Constant(ll.IntType(32), 0)]
            else:
                indices = [ll.Constant(ll.IntType(32), 0), ll.Constant(ll.IntType(32), 0)]
        
        # 使用 inbounds GEP（优化）
        result = builder.gep(
            struct_ptr, 
            indices, 
            inbounds=True,  # 关键优化：假设指针在边界内
            name=context.get_result_name(instr)
        )
        
        context.store_result(instr, result)
        return result


class NestedStructGepStrategy(InstructionStrategy):
    """
    嵌套结构体访问策略
    
    使用方式：
        %nested_member = nested_gep %struct_ptr, "外层成员", "内层成员"
    
    例如：
        结构体 学校 { 学生 s; }
        %age_ptr = nested_gep %school_ptr, "s", "年龄"
    """
    
    opcode = Opcode.NESTED_GEP
    
    def compile(self, builder, instr, context):
        import llvmlite.ir as ll
        
        struct_ptr = context.get_value(instr.operands[0])
        
        # 获取访问路径
        access_path = instr.metadata.get('access_path', [])
        # 例如: ["s", "年龄"]
        
        # 构建 GEP 索引链
        indices = [ll.Constant(ll.IntType(32), 0)]  # 解引用
        
        current_struct_type = struct_type_name = instr.metadata.get('struct_type')
        layout_calc = context.struct_layout_calculator
        
        for member_name in access_path:
            if current_struct_type:
                layout = layout_calc.get_layout(current_struct_type)
                if layout:
                    member = layout.member_map.get(member_name)
                    if member:
                        indices.append(ll.Constant(ll.IntType(32), member.index))
                        # 更新当前结构体类型（如果是嵌套结构体）
                        current_struct_type = member.type_name
                    else:
                        indices.append(ll.Constant(ll.IntType(32), 0))
                else:
                    indices.append(ll.Constant(ll.IntType(32), 0))
            else:
                indices.append(ll.Constant(ll.IntType(32), 0))
        
        # 使用 inbounds GEP
        result = builder.gep(
            struct_ptr,
            indices,
            inbounds=True,
            name=context.get_result_name(instr)
        )
        
        context.store_result(instr, result)
        return result
```

### 2.3 编译流程集成

```python
class LLVMBackend:
    """LLVM 后端 - 集成结构体优化"""
    
    def __init__(self):
        # ... 现有初始化 ...
        
        # 初始化结构体布局计算器
        self.struct_layout_calc = StructLayoutCalculator()
        self.struct_type_mapper = LLVMStructTypeMapper(self.struct_layout_calc)
        
        # 注册到编译上下文
        self.context.struct_layout_calculator = self.struct_layout_calc
    
    def compile_to_module(self, ir: IRProgram, module_name: str) -> ll.Module:
        """编译到 LLVM Module"""
        
        # 1. 先处理结构体定义
        for struct_def in ir.struct_definitions:
            # 计算布局
            layout = self.struct_layout_calc.calculate_layout(struct_def)
            
            # 创建 LLVM 类型
            struct_type = self.struct_type_mapper.get_llvm_struct_type(
                struct_def.name, 
                self.module
            )
            
            # 注册到上下文
            self.context.struct_types[struct_def.name] = struct_type
        
        # 2. 编译函数（使用优化后的结构体访问）
        for func in ir.functions:
            self._compile_function(func)
        
        return self.module
```
