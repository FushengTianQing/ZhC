# P1 类型系统 - 自动类型推导

## 功能概述

**功能名称**：自动类型推导（类似 C++ auto）  
**优先级**：P1  
**功能模块**：类型系统  
**创建日期**：2026-04-09

---

## 1. 需求分析

### 1.1 当前状态

ZhC 编译器已有基础类型推导（`src/zhc/typeinfer/engine.py`）：
- Hindley-Milner 类型推导算法
- 基础类型推导（字面量、二元表达式）
- 类型变量和约束系统

**现有问题**：
1. 不支持 `auto` 关键字（或中文等价关键字）
2. 变量声明必须显式指定类型
3. 函数返回类型不支持自动推导
4. 泛型类型推导不完整

### 1.2 目标状态

实现完整的自动类型推导：

```zhc
// 变量类型推导
自动 x = 42;              // 推导为 整数型
自动 y = 3.14;            // 推导为 浮点型
自动 s = "hello";         // 推导为 字符串型
自动 arr = [1, 2, 3];     // 推导为 整数型数组

// 函数返回类型推导
自动型 加法(整数型 a, 整数型 b) {
    返回 a + b;           // 自动推导返回类型为 整数型
}

// 复杂表达式推导
自动 结果 = 计算(10, 20) * 2;  // 根据函数返回类型推导

// 泛型推导
自动 容器 = 创建容器();        // 推导泛型参数
```

@dataclass
class VariableDeclNode(ASTNode):
    """变量声明节点 - 支持自动类型"""
    
    name: str
    var_type: Optional[str] = None      # None 表示自动推导
    initializer: Optional[ASTNode] = None
    is_auto: bool = False               # 是否为自动类型
    inferred_type: Optional[str] = None # 推导出的类型


@dataclass
class FunctionDeclNode(ASTNode):
    """函数声明节点 - 支持自动返回类型"""
    
    name: str
    parameters: List[ParameterNode]
    return_type: Optional[str] = None   # None 表示自动推导
    is_auto_return: bool = False        # 是否为自动返回类型
    inferred_return_type: Optional[str] = None
    body: Optional[BlockStmtNode] = None


@dataclass
class AutoTypeNode(ASTNode):
    """自动类型节点"""
    
    node_type = ASTNodeType.AUTO_TYPE
    
    def __init__(self, line=0, column=0):
        super().__init__(ASTNodeType.AUTO_TYPE, line, column)
        self.resolved_type = None  # 解析后的实际类型

#### 2.2.3 AutoTypeInferencer（新增）

```python
class AutoTypeInferencer:
    """自动类型推导器"""
    
    def __init__(self, type_engine: TypeInferenceEngine):
        self.engine = type_engine
        self.type_env = TypeEnv()
    
    def infer_variable_type(
        self,
        node: VariableDeclNode,
        context: InferenceContext
    ) -> str:
        """推导变量类型"""
        
        if not node.is_auto:
            return node.var_type
        
        if node.initializer is None:
            raise SemanticError(
                "自动类型变量必须有初始化表达式",
                location=SourceLocation(line=node.line, column=node.column)
            )
        
        # 推导初始化表达式类型
        init_type = self.engine.infer(node.initializer, self.type_env)
        
        # 解析类型变量
        if isinstance(init_type, TypeVariable):
            # 尝试求解约束
            if not self.engine.solve_constraints():
                raise SemanticError(
                    "无法推导变量类型",
                    location=SourceLocation(line=node.line, column=node.column)
                )
            init_type = self.engine._prune(init_type)
        
        # 转换为具体类型字符串
        resolved_type = self._type_to_string(init_type)
        
        # 缓存推导结果
        node.inferred_type = resolved_type
        
        return resolved_type
    
    def infer_function_return_type(
        self,
        node: FunctionDeclNode,
        context: InferenceContext
    ) -> str:
        """推导函数返回类型"""
        
        if not node.is_auto_return:
            return node.return_type
        
        # 收集所有返回语句的类型
        return_types = []
        
        for stmt in self._find_return_statements(node.body):
            if stmt.value:
                ret_type = self.engine.infer(stmt.value, self.type_env)
                return_types.append(ret_type)
        
        if not return_types:
            return "空型"
        
        # 统一所有返回类型
        unified_type = return_types[0]
        for t in return_types[1:]:
            if not self.engine.unify(unified_type, t):
                raise SemanticError(
                    f"函数返回类型不一致: {unified_type} 和 {t}",
                    location=SourceLocation(line=node.line, column=node.column)
                )
        
        # 解析类型变量
        if isinstance(unified_type, TypeVariable):
            self.engine.solve_constraints()
            unified_type = self.engine._prune(unified_type)
        
        resolved_type = self._type_to_string(unified_type)
        node.inferred_return_type = resolved_type
        
        return resolved_type
    
    def infer_expression_type(
        self,
        expr: ASTNode,
        expected_type: Optional[str] = None
    ) -> str:
        """推导表达式类型"""
        
        inferred = self.engine.infer(expr, self.type_env)
        
        if isinstance(inferred, TypeVariable):
            if expected_type:
                # 添加约束
                expected = self._string_to_type(expected_type)
                self.engine.add_constraint(inferred, expected)
                self.engine.solve_constraints()
            inferred = self.engine._prune(inferred)
        
        return self._type_to_string(inferred)
    
    def _type_to_string(self, type_: Type) -> str:
        """将类型对象转换为字符串"""
        if isinstance(type_, BaseType):
            return type_.value
        elif isinstance(type_, ArrayType):
            elem = self._type_to_string(type_.element_type)
            if type_.size:
                return f"{elem}[{type_.size}]"
            return f"{elem}[]"
        elif isinstance(type_, FunctionType):
            params = ", ".join(self._type_to_string(p) for p in type_.param_types)
            ret = self._type_to_string(type_.return_type)
            return f"({params}) -> {ret}"
        elif isinstance(type_, TypeVariable):
            return "未知"
        else:
            return str(type_)
    
    def _string_to_type(self, type_str: str) -> Type:
        """将字符串转换为类型对象"""
        # 解析类型字符串
        type_map = {
            "整数型": BaseType.INT,
            "浮点型": BaseType.FLOAT,
            "字符型": BaseType.CHAR,
            "字符串型": BaseType.STRING,
            "布尔型": BaseType.BOOL,
            "空型": BaseType.VOID,
        }
        
        if type_str in type_map:
            return type_map[type_str]
        
        # 数组类型
        if type_str.endswith("]"):
            match = re.match(r"(.+)\[(\d*)\]$", type_str)
            if match:
                elem_type = self._string_to_type(match.group(1))
                size = int(match.group(2)) if match.group(2) else None
                return ArrayType(elem_type, size)
        
        return self.engine.fresh_type_var()
    
    def _find_return_statements(
        self,
        node: ASTNode
    ) -> List[ReturnStmtNode]:
        """递归查找所有返回语句"""
        returns = []
        
        if isinstance(node, ReturnStmtNode):
            returns.append(node)
        elif hasattr(node, 'statements'):
            for stmt in node.statements:
                returns.extend(self._find_return_statements(stmt))
        elif hasattr(node, 'body'):
            returns.extend(self._find_return_statements(node.body))
        
        return returns
```

#### 2.2.4 类型推导规则

```python
class TypeInferenceRules:
    """类型推导规则"""
    
    # 字面量类型映射
    LITERAL_TYPES = {
        ASTNodeType.INT_LITERAL: "整数型",
        ASTNodeType.FLOAT_LITERAL: "浮点型",
        ASTNodeType.STRING_LITERAL: "字符串型",
        ASTNodeType.CHAR_LITERAL: "字符型",
        ASTNodeType.BOOL_LITERAL: "布尔型",
        ASTNodeType.NULL_LITERAL: "空",
    }
    
    # 二元运算符结果类型
    BINARY_RESULT_TYPES = {
        ("+", "整数型", "整数型"): "整数型",
        ("+", "浮点型", "浮点型"): "浮点型",
        ("+", "整数型", "浮点型"): "浮点型",
        ("+", "浮点型", "整数型"): "浮点型",
        ("+", "字符串型", "字符串型"): "字符串型",
        ("-", "整数型", "整数型"): "整数型",
        ("-", "浮点型", "浮点型"): "浮点型",
        ("*", "整数型", "整数型"): "整数型",
        ("/", "整数型", "整数型"): "整数型",
        ("/", "浮点型", "浮点型"): "浮点型",
        ("<", "整数型", "整数型"): "布尔型",
        (">", "整数型", "整数型"): "布尔型",
        ("==", "整数型", "整数型"): "布尔型",
        ("并且", "布尔型", "布尔型"): "布尔型",
        ("或者", "布尔型", "布尔型"): "布尔型",
    }
    
    @classmethod
    def infer_binary_type(
        cls,
        operator: str,
        left_type: str,
        right_type: str
    ) -> Optional[str]:
        """推导二元表达式类型"""
        
        # 精确匹配
        key = (operator, left_type, right_type)
        if key in cls.BINARY_RESULT_TYPES:
            return cls.BINARY_RESULT_TYPES[key]
        
        # 通用规则
        if operator in ["<", ">", "<=", ">=", "==", "!="]:
            return "布尔型"
        
        if operator in ["并且", "或者"]:
            return "布尔型"
        
        # 数值运算
        if operator in ["+", "-", "*", "/"]:
            if "浮点型" in [left_type, right_type]:
                return "浮点型"
            return "整数型"
        
        return None
```

### 2.3 编译流程集成

```python
class SemanticAnalyzer:
    """语义分析器 - 集成自动类型推导"""
    
    def __init__(self):
        self.type_inferencer = AutoTypeInferencer(TypeInferenceEngine())
    
    def analyze_variable_decl(
        self,
        node: VariableDeclNode
    ) -> None:
        """分析变量声明"""
        
        if node.is_auto:
            # 自动类型推导
            inferred_type = self.type_inferencer.infer_variable_type(
                node,
                context=self._create_context()
            )
            
            # 更新符号表
            symbol = Symbol(
                name=node.name,
                symbol_type="变量",
                data_type=inferred_type,
                is_defined=True
            )
            self.symbol_table.current_scope.add_symbol(symbol)
            
            # 记录推导结果
            node.var_type = inferred_type
            node.inferred_type = inferred_type
            
            print(f"[类型推导] 变量 '{node.name}' 推导为 '{inferred_type}'")
        
        else:
            # 显式类型声明
            super().analyze_variable_decl(node)
    
    def analyze_function_decl(
        self,
        node: FunctionDeclNode
    ) -> None:
        """分析函数声明"""
        
        # 先分析参数（建立类型环境）
        for param in node.parameters:
            self.type_inferencer.type_env.extend(
                param.name,
                self.type_inferencer._string_to_type(param.param_type)
            )
        
        # 分析函数体
        self.analyze_block(node.body)
        
        # 自动返回类型推导
        if node.is_auto_return:
            inferred_return = self.type_inferencer.infer_function_return_type(
                node,
                context=self._create_context()
            )
            node.return_type = inferred_return
            node.inferred_return_type = inferred_return
            
            print(f"[类型推导] 函数 '{node.name}' 返回类型推导为 '{inferred_return}'")
```
