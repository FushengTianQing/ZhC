## 1. 需求分析

### 1.1 当前状态

ZhC 编译器当前遇到错误时会：
- 抛出异常并终止编译
- 只报告第一个错误
- 无法收集所有错误

**现有问题**：
1. 遇到第一个错误就停止，无法发现其他问题
2. 用户需要多次编译才能修复所有错误
3. 错误之间可能存在关联，但无法一起展示
4. 缺少错误恢复策略，无法继续分析后续代码

### 1.2 目标状态

实现错误恢复机制，收集所有错误：

```
编译发现 5 个错误，2 个警告:

错误[E001]: 类型不匹配
   --> test.zhc:10:15

错误[E002]: 未定义的变量 '计数器'
   --> test.zhc:12:8

错误[E003]: 函数 '计算' 缺少返回语句
   --> test.zhc:20:1

警告[W001]: 变量 '临时' 未使用
   --> test.zhc:8:5

警告[W002]: 可能的整数溢出
   --> test.zhc:15:20

编译失败。请修复上述错误后重新编译。
```

#### 2.2.1 ErrorRecoveryStrategy（新增）

```python
from enum import Enum

class RecoveryAction(Enum):
    """恢复动作类型"""
    SKIP_TOKEN = "跳过当前 Token"
    SKIP_TO_SYNC = "跳到同步点"
    INSERT_TOKEN = "插入缺失 Token"
    REPLACE_TOKEN = "替换错误 Token"
    CREATE_PLACEHOLDER = "创建占位节点"
    ABORT = "中止当前分析"


class ErrorRecoveryStrategy:
    """错误恢复策略"""
    
    # 同步点 Token（用于恢复）
    SYNC_TOKENS = {
        'statement': ['分号', '右花括号'],
        'declaration': ['整数型', '浮点型', '字符型', '字符串型', '布尔型', '空型'],
        'expression': ['分号', '右括号', '右花括号', '逗号'],
        'block': ['右花括号'],
        'function': ['整数型', '浮点型', '字符型', '字符串型', '布尔型', '空型', '右花括号']
    }
    
    def __init__(self, error_collection: ErrorCollection):
        self.errors = error_collection
        self.recovery_points = []
        self.max_errors = 100  # 最大错误数量限制
    
    def recover(
        self,
        error: ZHCError,
        context: RecoveryContext
    ) -> RecoveryAction:
        """根据错误类型选择恢复策略"""
        
        # 检查是否达到最大错误数
        if self.errors.error_count() >= self.max_errors:
            return RecoveryAction.ABORT
        
        # 根据错误类型选择策略
        if error.error_code.startswith('PARSER_'):
            return self._recover_parser_error(error, context)
        elif error.error_code.startswith('SEMANTIC_'):
            return self._recover_semantic_error(error, context)
        else:
            return RecoveryAction.SKIP_TOKEN
    
    def _recover_parser_error(
        self,
        error: ZHCError,
        context: RecoveryContext
    ) -> RecoveryAction:
        """语法错误恢复"""
        
        # 缺失 Token：尝试插入
        if error.error_code == 'PARSER_MISSING_TOKEN':
            return RecoveryAction.INSERT_TOKEN
        
        # 意外 Token：跳到同步点
        if error.error_code == 'PARSER_UNEXPECTED_TOKEN':
            return RecoveryAction.SKIP_TO_SYNC
        
        # 不平衡括号：跳到匹配点
        if error.error_code in ['PARSER_UNBALANCED_PARENS', 'PARSER_UNBALANCED_BRACES']:
            return RecoveryAction.SKIP_TO_SYNC
        
        return RecoveryAction.SKIP_TOKEN
    
    def _recover_semantic_error(
        self,
        error: ZHCError,
        context: RecoveryContext
    ) -> RecoveryAction:
        """语义错误恢复"""
        
        # 未定义符号：创建占位符
        if error.error_code == 'SEMANTIC_UNDEFINED_VARIABLE':
            return RecoveryAction.CREATE_PLACEHOLDER
        
        # 类型不匹配：继续分析
        if error.error_code == 'SEMANTIC_TYPE_MISMATCH':
            return RecoveryAction.SKIP_TOKEN
        
        return RecoveryAction.SKIP_TOKEN
    
    def find_sync_point(
        self,
        tokens: List[Token],
        current_idx: int,
        sync_type: str
    ) -> int:
        """查找同步点"""
        sync_tokens = self.SYNC_TOKENS.get(sync_type, ['分号', '右花括号'])
        
        for i in range(current_idx, len(tokens)):
            if tokens[i].type in sync_tokens:
                return i
        
        return len(tokens) - 1
```

#### 2.2.2 ParserErrorRecovery（新增）

```python
class ParserErrorRecovery:
    """Parser 错误恢复"""
    
    def __init__(self, parser, recovery_strategy: ErrorRecoveryStrategy):
        self.parser = parser
        self.strategy = recovery_strategy
    
    def handle_error(
        self,
        error: ParserError,
        tokens: List[Token],
        current_idx: int
    ) -> Tuple[Optional[ASTNode], int]:
        """处理语法错误并恢复"""
        
        # 记录错误
        self.strategy.errors.add(error)
        
        # 选择恢复策略
        context = RecoveryContext(
            tokens=tokens,
            current_idx=current_idx,
            parser_state=self.parser.state
        )
        action = self.strategy.recover(error, context)
        
        # 执行恢复动作
        if action == RecoveryAction.INSERT_TOKEN:
            return self._insert_token(error, tokens, current_idx)
        
        elif action == RecoveryAction.SKIP_TO_SYNC:
            sync_idx = self.strategy.find_sync_point(
                tokens, current_idx, self._get_sync_type(error)
            )
            return None, sync_idx + 1
        
        elif action == RecoveryAction.CREATE_PLACEHOLDER:
            return self._create_placeholder(error), current_idx + 1
        
        elif action == RecoveryAction.ABORT:
            raise CompilationAbortedError("错误数量过多，编译中止")
        
        else:
            return None, current_idx + 1
    
    def _insert_token(
        self,
        error: ParserError,
        tokens: List[Token],
        current_idx: int
    ) -> Tuple[Optional[ASTNode], int]:
        """插入缺失的 Token"""
        missing_token_type = error.context.get('expected_token')
        
        if missing_token_type:
            # 创建虚拟 Token
            virtual_token = Token(
                type=missing_token_type,
                value='',
                line=tokens[current_idx].line,
                column=tokens[current_idx].column
            )
            
            # 插入并继续解析
            self.parser.tokens.insert(current_idx, virtual_token)
            return None, current_idx
        
        return None, current_idx + 1
    
    def _create_placeholder(self, error: ParserError) -> ASTNode:
        """创建占位节点"""
        placeholder = ASTNode(
            node_type=ASTNodeType.ERROR_NODE,
            line=error.location.line,
            column=error.location.column
        )
        placeholder.error = error
        placeholder.is_placeholder = True
        return placeholder
    
    def _get_sync_type(self, error: ParserError) -> str:
        """根据错误类型获取同步点类型"""
        if 'declaration' in error.error_code.lower():
            return 'declaration'
        elif 'statement' in error.error_code.lower():
            return 'statement'
        elif 'expression' in error.error_code.lower():
            return 'expression'
        else:
            return 'block'
```

#### 2.2.3 SemanticErrorRecovery（新增）

```python
class SemanticErrorRecovery:
    """语义分析错误恢复"""
    
    def __init__(self, analyzer, recovery_strategy: ErrorRecoveryStrategy):
        self.analyzer = analyzer
        self.strategy = recovery_strategy
    
    def handle_undefined_symbol(
        self,
        name: str,
        node: ASTNode,
        symbol_type: str
    ) -> Symbol:
        """处理未定义符号"""
        
        # 创建占位符号
        placeholder = Symbol(
            name=name,
            symbol_type=symbol_type,
            data_type='未知',
            is_placeholder=True,
            definition_location=f"{node.line}:{node.column}"
        )
        
        # 添加到符号表（临时）
        self.analyzer.symbol_table.current_scope.add_symbol(placeholder)
        
        # 记录错误
        error = SemanticError(
            message=f"未定义的符号 '{name}'",
            location=SourceLocation(
                file_path=self.analyzer.current_file,
                line=node.line,
                column=node.column
            ),
            error_code="SEMANTIC_UNDEFINED_VARIABLE",
            suggestion=f"请声明变量 '{name}' 或检查名称是否正确"
        )
        self.strategy.errors.add(error)
        
        return placeholder
    
    def handle_type_mismatch(
        self,
        expected: str,
        actual: str,
        node: ASTNode,
        context: str
    ) -> str:
        """处理类型不匹配"""
        
        # 记录错误
        error = SemanticError(
            message=f"类型不匹配: 期望 '{expected}', 实际 '{actual}'",
            location=SourceLocation(
                file_path=self.analyzer.current_file,
                line=node.line,
                column=node.column
            ),
            error_code="SEMANTIC_TYPE_MISMATCH",
            context=context
        )
        self.strategy.errors.add(error)
        
        # 返回一个兼容类型继续分析
        return '未知'
```

#### 2.2.4 ErrorModeManager（新增）

```python
class ErrorMode(Enum):
    """错误处理模式"""
    STRICT = "严格模式"      # 遇到错误立即停止
    LENIENT = "宽松模式"     # 收集所有错误继续
    RECOVER = "恢复模式"     # 尝试恢复并继续


class ErrorModeManager:
    """错误模式管理"""
    
    def __init__(self, mode: ErrorMode = ErrorMode.LENIENT):
        self.mode = mode
        self.errors = ErrorCollection()
        self.recovery_strategy = ErrorRecoveryStrategy(self.errors)
    
    def should_continue(self) -> bool:
        """判断是否应该继续编译"""
        if self.mode == ErrorMode.STRICT:
            return not self.errors.has_errors()
        
        if self.mode == ErrorMode.LENIENT:
            return self.errors.error_count() < self.recovery_strategy.max_errors
        
        return True
    
    def handle_error(
        self,
        error: ZHCError,
        context: RecoveryContext = None
    ) -> Optional[RecoveryAction]:
        """处理错误"""
        self.errors.add(error)
        
        if self.mode == ErrorMode.STRICT:
            raise error
        
        if self.mode == ErrorMode.RECOVER and context:
            return self.recovery_strategy.recover(error, context)
        
        return None
    
    def get_summary(self) -> str:
        """获取错误摘要"""
        return self.errors.detailed_report()
```

### 2.3 编译流程集成

```python
class Compiler:
    """编译器 - 集成错误恢复"""
    
    def compile(
        self,
        source: str,
        error_mode: ErrorMode = ErrorMode.LENIENT
    ) -> CompileResult:
        """编译源代码"""
        
        # 初始化错误模式管理器
        error_manager = ErrorModeManager(error_mode)
        
        # 词法分析
        try:
            tokens = self.lexer.tokenize(source)
        except LexerError as e:
            error_manager.handle_error(e)
            if not error_manager.should_continue():
                return CompileResult(success=False, errors=error_manager.errors)
        
        # 语法分析（带恢复）
        parser_recovery = ParserErrorRecovery(
            self.parser,
            error_manager.recovery_strategy
        )
        
        ast = self._parse_with_recovery(tokens, parser_recovery, error_manager)
        
        if not error_manager.should_continue():
            return CompileResult(success=False, errors=error_manager.errors)
        
        # 语义分析（带恢复）
        semantic_recovery = SemanticErrorRecovery(
            self.semantic_analyzer,
            error_manager.recovery_strategy
        )
        
        self._analyze_with_recovery(ast, semantic_recovery, error_manager)
        
        # 返回结果
        if error_manager.errors.has_errors():
            return CompileResult(
                success=False,
                errors=error_manager.errors,
                warnings=error_manager.errors.get_warnings()
            )
        
        return CompileResult(
            success=True,
            output=self._generate_code(ast),
            warnings=error_manager.errors.get_warnings()
        )
    
    def _parse_with_recovery(
        self,
        tokens: List[Token],
        recovery: ParserErrorRecovery,
        error_manager: ErrorModeManager
    ) -> ASTNode:
        """带恢复的语法分析"""
        
        idx = 0
        statements = []
        
        while idx < len(tokens) and error_manager.should_continue():
            try:
                stmt, idx = self.parser.parse_statement(tokens, idx)
                if stmt:
                    statements.append(stmt)
            except ParserError as e:
                stmt, idx = recovery.handle_error(e, tokens, idx)
                if stmt:
                    statements.append(stmt)
        
        return ProgramNode(statements=statements)
```
