## 1. 需求分析

### 1.1 当前状态

ZhC 编译器目前缺少函数指针支持：
- 无法声明函数指针变量
- 无法将函数作为参数传递
- 无法实现回调机制
- 无法支持高阶函数模式

**现有相关代码**：
- `src/zhc/parser/parser.py`：类型解析
- `src/zhc/backend/type_system.py`：类型映射
- `src/zhc/ir/instructions.py`：IR 指令

### 1.2 目标状态

实现完整的函数指针类型系统：

```zhc
// 函数指针声明
函数指针型<整数型, 整数型> 回调 = 计算平方;

// 函数指针作为参数
整数型 处理数据(整数型 数据, 函数指针型<整数型, 整数型> 处理器) {
    返回 处理器(数据);
}

// 回调使用
整数型 主函数() {
    自动 结果 = 处理数据(5, 计算平方);
    输出("结果: %d\n", 结果);
    返回 0;
}

整数型 计算平方(整数型 x) {
    返回 x * x;
}
```

**功能目标**：
1. 支持函数指针类型声明
2. 支持函数指针赋值
3. 支持函数指针调用
4. 支持函数指针作为参数/返回值
5. 支持函数指针数组

```python
from dataclasses import dataclass
from typing import List, Optional

@dataclass
class FunctionPointerTypeNode(ASTNode):
    """
    函数指针类型节点
    
    语法：函数指针型<返回类型, 参数类型1, 参数类型2, ...>
    
    例如：
        函数指针型<整数型, 整数型>  // 整数型 -> 整数型
        函数指针型<无类型, 整数型, 字符串型>  // (整数型, 字符串型) -> 无类型
    """
    return_type: 'TypeNode'           # 返回类型
    param_types: List['TypeNode']     # 参数类型列表
    
    def __post_init__(self):
        self.node_type = NodeType.FUNCTION_POINTER_TYPE
    
    def accept(self, visitor: 'ASTVisitor'):
        return visitor.visit_function_pointer_type(self)


@dataclass
class FunctionPointerDeclNode(ASTNode):
    """
    函数指针声明节点
    
    语法：函数指针型<...> 名称 = 函数名;
    
    例如：
        函数指针型<整数型, 整数型> 回调 = 计算平方;
    """
    func_ptr_type: FunctionPointerTypeNode  # 函数指针类型
    name: str                                # 变量名
    target_func: Optional[str] = None       # 目标函数名（可选）
    
    def __post_init__(self):
        self.node_type = NodeType.FUNCTION_POINTER_DECL
    
    def accept(self, visitor: 'ASTVisitor'):
        return visitor.visit_function_pointer_decl(self)


@dataclass  
class FunctionPointerCallNode(ASTNode):
    """
    函数指针调用节点
    
    语法：函数指针变量(参数1, 参数2, ...)
    
    例如：
        回调(5);
        处理器(数据);
    """
    func_ptr_name: str                 # 函数指针变量名
    arguments: List['ExpressionNode']  # 参数列表
    
    def __post_init__(self):
        self.node_type = NodeType.FUNCTION_POINTER_CALL
    
    def accept(self, visitor: 'ASTVisitor'):
        return visitor.visit_function_pointer_call(self)
```

#### 2.2.2 语法解析器增强

**修改文件**：`src/zhc/parser/parser.py`

```python
class Parser:
    # ... 现有代码 ...
    
    def parse_function_pointer_type(self) -> FunctionPointerTypeNode:
        """
        解析函数指针类型
        
        语法：函数指针型<返回类型, 参数类型1, ...>
        """
        # 匹配 "函数指针型"
        self.expect(TokenType.FUNCTION_POINTER)
        
        # 匹配 "<"
        self.expect(TokenType.LANGLE)
        
        # 解析返回类型
        return_type = self.parse_type()
        
        # 解析参数类型列表
        param_types = []
        while self.current_token.type == TokenType.COMMA:
            self.advance()  # 跳过逗号
            param_type = self.parse_type()
            param_types.append(param_type)
        
        # 匹配 ">"
        self.expect(TokenType.RANGLE)
        
        return FunctionPointerTypeNode(
            return_type=return_type,
            param_types=param_types,
            line=self.current_token.line,
            column=self.current_token.column
        )
    
    def parse_function_pointer_decl(self) -> FunctionPointerDeclNode:
        """
        解析函数指针声明
        
        语法：函数指针型<...> 名称 = 函数名;
        """
        # 解析函数指针类型
        func_ptr_type = self.parse_function_pointer_type()
        
        # 解析变量名
        name_token = self.expect(TokenType.IDENTIFIER)
        name = name_token.value
        
        # 可选：赋值
        target_func = None
        if self.current_token.type == TokenType.ASSIGN:
            self.advance()  # 跳过 "="
            func_name_token = self.expect(TokenType.IDENTIFIER)
            target_func = func_name_token.value
        
        # 匹配 ";"
        self.expect(TokenType.SEMICOLON)
        
        return FunctionPointerDeclNode(
            func_ptr_type=func_ptr_type,
            name=name,
            target_func=target_func,
            line=self.current_token.line,
            column=self.current_token.column
        )
    
    def parse_function_pointer_call(self) -> FunctionPointerCallNode:
        """
        解析函数指针调用
        
        语法：函数指针变量(参数1, ...)
        """
        # 解析函数指针变量名
        func_ptr_name = self.current_token.value
        self.advance()
        
        # 匹配 "("
        self.expect(TokenType.LPAREN)
        
        # 解析参数列表
        arguments = self.parse_argument_list()
        
        # 匹配 ")"
        self.expect(TokenType.RPAREN)
        
        return FunctionPointerCallNode(
            func_ptr_name=func_ptr_name,
            arguments=arguments,
            line=self.current_token.line,
            column=self.current_token.column
        )
```

```python
from dataclasses import dataclass
from typing import List, Optional, Dict
import llvmlite.ir as ll

@dataclass
class FunctionPointerTypeInfo:
    """函数指针类型信息"""
    return_type: str              # 返回类型名
    param_types: List[str]        # 参数类型名列表
    llvm_type: Optional[ll.Type]  # LLVM 函数类型
    
    def signature_str(self) -> str:
        """生成类型签名字符串"""
        params = ", ".join(self.param_types)
        return f"{self.return_type}({params})"


class TypeMapper:
    # ... 现有代码 ...
    
    def map_function_pointer_type(
        self, 
        return_type: str, 
        param_types: List[str]
    ) -> ll.PointerType:
        """
        映射函数指针类型到 LLVM
        
        返回 LLVM 函数指针类型：ReturnType (*)(ParamType1, ParamType2, ...)
        """
        # 获取返回类型的 LLVM 类型
        llvm_return = self.to_llvm(return_type)
        
        # 获取参数类型的 LLVM 类型列表
        llvm_params = [self.to_llvm(pt) for pt in param_types]
        
        # 创建 LLVM 函数类型
        func_type = ll.FunctionType(llvm_return, llvm_params)
        
        # 返回函数指针类型
        return ll.PointerType(func_type)
    
    def get_function_pointer_info(
        self, 
        func_ptr_type_node: 'FunctionPointerTypeNode'
    ) -> FunctionPointerTypeInfo:
        """获取函数指针类型信息"""
        return_type = func_ptr_type_node.return_type.type_name
        param_types = [pt.type_name for pt in func_ptr_type_node.param_types]
        
        llvm_type = self.map_function_pointer_type(return_type, param_types)
        
        return FunctionPointerTypeInfo(
            return_type=return_type,
            param_types=param_types,
            llvm_type=llvm_type
        )
```

```python
@dataclass
class FunctionPointerDeclInstruction(Instruction):
    """
    函数指针声明指令
    
    IR 格式：
        %func_ptr = func_ptr_decl <return_type>(param_types)
    """
    return_type: str
    param_types: List[str]
    
    def __post_init__(self):
        self.opcode = Opcode.FUNC_PTR_DECL


@dataclass
class FunctionPointerAssignInstruction(Instruction):
    """
    函数指针赋值指令
    
    IR 格式：
        func_ptr_assign %func_ptr, @target_func
    """
    func_ptr: str       # 函数指针变量
    target_func: str    # 目标函数名
    
    def __post_init__(self):
        self.opcode = Opcode.FUNC_PTR_ASSIGN


@dataclass
class FunctionPointerCallInstruction(Instruction):
    """
    函数指针调用指令
    
    IR 格式：
        %result = func_ptr_call %func_ptr(args)
    """
    func_ptr: str               # 函数指针变量
    arguments: List[str]        # 参数列表
    return_type: str            # 返回类型
    
    def __post_init__(self):
        self.opcode = Opcode.FUNC_PTR_CALL
```

#### 2.2.5 LLVM 后端实现

**修改文件**：`src/zhc/backend/llvm_instruction_strategy.py`

```python
class FunctionPointerDeclStrategy(InstructionStrategy):
    """
    函数指针声明策略
    
    LLVM IR：
        %func_ptr = alloca ReturnType (*)(ParamTypes)
    """
    opcode = Opcode.FUNC_PTR_DECL
    
    def compile(self, builder, instr, context):
        import llvmlite.ir as ll
        
        # 获取函数指针类型
        return_type = instr.return_type
        param_types = instr.param_types
        
        # 映射到 LLVM 类型
        mapper = context.type_mapper
        func_ptr_type = mapper.map_function_pointer_type(return_type, param_types)
        
        # 分配函数指针变量
        func_ptr_alloca = builder.alloca(
            func_ptr_type,
            name=context.get_result_name(instr)
        )
        
        context.store_result(instr, func_ptr_alloca)
        return func_ptr_alloca


class FunctionPointerAssignStrategy(InstructionStrategy):
    """
    函数指针赋值策略
    
    LLVM IR：
        store @target_func, %func_ptr
    """
    opcode = Opcode.FUNC_PTR_ASSIGN
    
    def compile(self, builder, instr, context):
        # 获取函数指针变量
        func_ptr = context.get_value(instr.func_ptr)
        
        # 获取目标函数
        target_func = context.module.get_global(instr.target_func)
        
        # 存储函数指针
        builder.store(target_func, func_ptr)


class FunctionPointerCallStrategy(InstructionStrategy):
    """
    函数指针调用策略
    
    LLVM IR：
        %func = load ReturnType (*)(ParamTypes), %func_ptr
        %result = call ReturnType %func(ParamTypes args)
    """
    opcode = Opcode.FUNC_PTR_CALL
    
    def compile(self, builder, instr, context):
        # 加载函数指针
        func_ptr_alloca = context.get_value(instr.func_ptr)
        loaded_func = builder.load(func_ptr_alloca, name="loaded_func")
        
        # 准备参数
        args = [context.get_value(arg) for arg in instr.arguments]
        
        # 调用函数
        result = builder.call(
            loaded_func,
            args,
            name=context.get_result_name(instr)
        )
        
        context.store_result(instr, result)
        return result
```

### 2.3 编译流程集成

```python
class IRGenerator:
    # ... 现有代码 ...
    
    def visit_function_pointer_decl(self, node: FunctionPointerDeclNode):
        """生成函数指针声明 IR"""
        
        # 创建函数指针声明指令
        decl_instr = FunctionPointerDeclInstruction(
            return_type=node.func_ptr_type.return_type.type_name,
            param_types=[pt.type_name for pt in node.func_ptr_type.param_types],
            result=node.name
        )
        self.emit(decl_instr)
        
        # 如果有初始赋值
        if node.target_func:
            assign_instr = FunctionPointerAssignInstruction(
                func_ptr=node.name,
                target_func=node.target_func
            )
            self.emit(assign_instr)
    
    def visit_function_pointer_call(self, node: FunctionPointerCallNode):
        """生成函数指针调用 IR"""
        
        # 生成参数 IR
        arg_results = []
        for arg in node.arguments:
            arg_result = self.visit(arg)
            arg_results.append(arg_result)
        
        # 获取函数指针类型信息
        func_ptr_info = self.current_scope.get_variable_info(node.func_ptr_name)
        return_type = func_ptr_info.return_type
        
        # 创建函数指针调用指令
        call_instr = FunctionPointerCallInstruction(
            func_ptr=node.func_ptr_name,
            arguments=arg_results,
            return_type=return_type,
            result=self.new_temp()
        )
        self.emit(call_instr)
        
        return call_instr.result
```
