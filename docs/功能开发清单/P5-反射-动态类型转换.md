# P5-反射-动态类型转换

> **优先级**: P5
> **功能模块**: 反射
> **功能名称**: 动态类型转换
> **创建日期**: 2026-04-10
> **状态**: 规划中

## 1. 功能概述

实现安全的运行时动态类型转换，支持向下转型、交叉转型、类型验证，并在转换失败时提供详细的错误信息。

### 1.1 目标

- 支持安全的向下转型（downcast）
- 支持交叉转型（crosscast）
- 支持类型转换验证
- 提供转换失败的详细诊断

### 1.2 语法设计

```zhc
// 基本动态转换
类 动物 {
    函数型 无 -> 无 叫声;
}

类 狗 继承 动物 {
    函数 叫() {
        打印("汪汪!");
    }
    
    函数 摇尾巴() {
        打印("摇尾巴");
    }
}

动物 a = 新 狗();

// 安全转换 - 使用 as 语法
狗 d = a as 狗;
如果 (d != 空) {
    d.叫();
    d.摇尾巴();
}

// 转换失败处理
猫 c = a as 猫;  // 返回 空，因为 a 实际是狗
如果 (c == 空) {
    打印("转换失败：a 不是猫");
}

// 强制转换 - 使用强制转换语法
狗 d2 = (狗) a;  // 如果失败会抛出异常

// 类型检查 + 转换
如果 (a is 狗) {
    狗 d3 = a as 狗;
    d3.叫();
}

// 多态转换
函数 处理动物(动物 a) {
    如果 (a is 狗) {
        狗 d = a as 狗;
        d.叫();
    } 否则 如果 (a is 猫) {
        猫 c = a as 猫;
        c.叫();
    }
}

// 接口转换
接口 可序列化 {
    函数型 无 -> 字符串 序列化();
}

类 数据 implements 可序列化 {
    整数型 值;
    
    函数 序列化() -> 字符串 {
        返回 "数据: " + 字符串(本.值);
    }
}

可序列化 s = 新 数据();
数据 d = s as 数据;  // 向下转换
如果 (d != 空) {
    打印(d.值);
}

// 类型转换函数
函数 转换示例(对象 obj) {
    // 尝试转换为不同类型
    如果 (obj is 字符串) {
        字符串 s = obj as 字符串;
        打印("字符串长度: " + s.长度());
    } 否则 如果 (obj is 整数型) {
        整数型 n = obj as 整数型;
        打印("整数: " + n);
    } 否则 如果 (obj is 数组) {
        数组 arr = obj as 数组;
        打印("数组长度: " + arr.长度());
    }
}
```

### 3.2 核心数据结构

```python
# src/zhc/reflection/dynamic_cast.py

from dataclasses import dataclass, field
from typing import Any, Optional, Type, TypeVar, Generic
from enum import Enum

T = TypeVar('T')

class CastErrorType(Enum):
    """转换错误类型"""
    NULL_SOURCE = "null_source"          # 源对象为空
    INVALID_CAST = "invalid_cast"        # 无效转换
    AMBIGUOUS_CAST = "ambiguous_cast"    # 歧义转换
    INTERFACE_NOT_FOUND = "interface_not_found"  # 接口未找到

@dataclass
class CastError:
    """转换错误"""
    error_type: CastErrorType
    message: str
    source_type: Optional[str] = None
    target_type: Optional[str] = None
    source_object: Optional[Any] = None
    
    def __str__(self) -> str:
        return f"CastError({self.error_type.value}): {self.message}"

@dataclass
class CastResult(Generic[T]):
    """转换结果"""
    success: bool
    value: Optional[T] = None
    error: Optional[CastError] = None
    
    @classmethod
    def ok(cls, value: T) -> 'CastResult[T]':
        """成功结果"""
        return cls(success=True, value=value)
    
    @classmethod
    def fail(cls, error: CastError) -> 'CastResult[T]':
        """失败结果"""
        return cls(success=False, error=error)
    
    @classmethod
    def fail_invalid(cls, source_type: str, target_type: str) -> 'CastResult[T]':
        """无效转换结果"""
        error = CastError(
            error_type=CastErrorType.INVALID_CAST,
            message=f"无法将 {source_type} 转换为 {target_type}",
            source_type=source_type,
            target_type=target_type,
        )
        return cls.fail(error)
    
    @classmethod
    def fail_null(cls) -> 'CastResult[T]':
        """空对象结果"""
        error = CastError(
            error_type=CastErrorType.NULL_SOURCE,
            message="源对象为空",
        )
        return cls.fail(error)
    
    def unwrap(self) -> T:
        """获取值，失败则抛出异常"""
        if self.success:
            return self.value
        raise TypeError(str(self.error))
    
    def unwrap_or(self, default: T) -> T:
        """获取值，失败则返回默认值"""
        return self.value if self.success else default

class DynamicCast:
    """动态类型转换器"""
    
    def __init__(self):
        from .type_hierarchy import TypeHierarchy
        self.hierarchy = TypeHierarchy.instance()
    
    def try_cast(self, obj: Any, target_type: str) -> CastResult:
        """尝试转换，返回结果对象"""
        if obj is None:
            return CastResult.fail_null()
        
        source_type = type(obj).__name__
        
        if source_type == target_type:
            return CastResult.ok(obj)
        
        if self.hierarchy.is_a(source_type, target_type):
            return CastResult.ok(obj)
        
        return CastResult.fail_invalid(source_type, target_type)
    
    def cast(self, obj: Any, target_type: str) -> Any:
        """强制转换，失败抛出异常"""
        result = self.try_cast(obj, target_type)
        return result.unwrap()
    
    def cast_or_none(self, obj: Any, target_type: str) -> Optional[Any]:
        """尝试转换，失败返回 None"""
        result = self.try_cast(obj, target_type)
        return result.value if result.success else None
    
    def cast_to_interface(self, obj: Any, interface_name: str) -> CastResult:
        """转换为接口类型"""
        if obj is None:
            return CastResult.fail_null()
        
        source_type = type(obj).__name__
        
        if self.hierarchy.is_a(source_type, interface_name):
            return CastResult.ok(obj)
        
        error = CastError(
            error_type=CastErrorType.INTERFACE_NOT_FOUND,
            message=f"{source_type} 未实现接口 {interface_name}",
            source_type=source_type,
            target_type=interface_name,
        )
        return CastResult.fail(error)
    
    def get_cast_path(self, source_type: str, target_type: str) -> Optional[list]:
        """获取转换路径"""
        if source_type == target_type:
            return [source_type]
        
        # 查找向上路径
        path = [source_type]
        current = source_type
        while current:
            parent = self.hierarchy.get_parent(current)
            if parent:
                path.append(parent)
                if parent == target_type:
                    return path
                current = parent
            else:
                break
        
        return None
    
    def can_cast(self, source_type: str, target_type: str) -> bool:
        """检查是否可以转换"""
        return self.hierarchy.is_a(source_type, target_type)

class CastValidator:
    """转换验证器"""
    
    def __init__(self):
        self.cast = DynamicCast()
    
    def validate(self, obj: Any, target_type: str) -> CastResult:
        """验证转换是否有效"""
        return self.cast.try_cast(obj, target_type)
    
    def validate_all(self, obj: Any, target_types: list) -> list:
        """验证对多个类型的转换"""
        results = []
        for target_type in target_types:
            results.append({
                "target": target_type,
                "result": self.cast.try_cast(obj, target_type),
            })
        return results
    
    def find_best_cast(self, obj: Any, target_types: list) -> Optional[str]:
        """找到最佳转换目标"""
        for target_type in target_types:
            if self.cast.try_cast(obj, target_type).success:
                return target_type
        return None

# 全局实例
_dynamic_cast = DynamicCast()
_cast_validator = CastValidator()

def try_cast(obj: Any, target_type: str) -> CastResult:
    """尝试转换"""
    return _dynamic_cast.try_cast(obj, target_type)

def dynamic_cast(obj: Any, target_type: str) -> Any:
    """动态转换"""
    return _dynamic_cast.cast(obj, target_type)

def cast_or_none(obj: Any, target_type: str) -> Optional[Any]:
    """转换或返回 None"""
    return _dynamic_cast.cast_or_none(obj, target_type)

def is_instance(obj: Any, type_name: str) -> bool:
    """检查实例类型"""
    return _dynamic_cast.can_cast(type(obj).__name__, type_name)
```
