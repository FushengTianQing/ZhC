## 1. 功能概述

代码重构工具提供自动化的代码改进功能，帮助开发者安全地改进代码结构和质量，而不改变代码行为。

### 1.1 核心目标

- 提供安全的代码重构操作
- 自动检测重构机会
- 保持代码语义不变
- 支持批量重构

### 1.2 重构类型

| 重构 | 描述 | 安全性 |
|------|------|--------|
| **重命名** | 变量/函数/类型重命名 | 高 |
| **提取函数** | 将代码片段提取为新函数 | 高 |
| **内联变量** | 内联局部变量 | 高 |
| **提取变量** | 提取表达式为变量 | 高 |
| **移动函数** | 移动函数到其他模块 | 中 |
| **改变签名** | 修改函数签名 | 中 |
| **引入参数** | 引入新参数 | 中 |

### 2.1 命令行接口

```bash
# 重命名
zhc refactor rename <旧名称> <新名称> --scope file src/main.zhc
zhc refactor rename 旧函数名 新函数名 --scope project

# 提取函数
zhc refactor extract-function <函数名> --start-line 10 --end-line 20 src/main.zhc

# 内联变量
zhc refactor inline-variable <变量名> src/main.zhc

# 提取变量
zhc refactor extract-variable <变量名> --expr "x + y" src/main.zhc

# 移动函数
zhc refactor move-function <函数名> --target utils.zhc src/main.zhc

# 改变签名
zhc refactor change-signature <函数名> --add-param "新参数: 整数" src/main.zhc

# 自动重构建议
zhc refactor suggest src/main.zhc

# 预览模式
zhc refactor rename 旧名 新名 --preview src/main.zhc
```

### 2.2 重构引擎架构

```python
# src/zhc/refactor/engine.py

from dataclasses import dataclass, field
from typing import List, Dict, Optional, Tuple
from abc import ABC, abstractmethod

@dataclass
class RefactoringContext:
    """重构上下文"""
    source_file: str
    source_code: str
    ast: ASTNode
    symbol_table: SymbolTable
    cursor_position: Optional[Tuple[int, int]] = None

@dataclass
class TextEdit:
    """文本编辑"""
    start_line: int
    start_column: int
    end_line: int
    end_column: int
    new_text: str
    description: str

@dataclass
class RefactoringResult:
    """重构结果"""
    edits: List[TextEdit]
    affected_files: List[str]
    preview: str
    warnings: List[str] = field(default_factory=list)

class Refactoring(ABC):
    """重构操作基类"""

    @property
    @abstractmethod
    def name(self) -> str:
        """重构名称"""
        pass

    @property
    @abstractmethod
    def description(self) -> str:
        """重构描述"""
        pass

    @abstractmethod
    def is_applicable(self, context: RefactoringContext) -> bool:
        """检查重构是否适用"""
        pass

    @abstractmethod
    def apply(self, context: RefactoringContext) -> RefactoringResult:
        """应用重构"""
        pass

    @abstractmethod
    def preview(self, context: RefactoringContext) -> str:
        """预览重构结果"""
        pass


class RefactoringEngine:
    """重构引擎"""

    def __init__(self):
        self.refactorings: Dict[str, Refactoring] = {}
        self._register_builtin_refactorings()

    def _register_builtin_refactorings(self):
        """注册内置重构操作"""
        self.register(RenameRefactoring())
        self.register(ExtractFunctionRefactoring())
        self.register(InlineVariableRefactoring())
        self.register(ExtractVariableRefactoring())
        self.register(MoveFunctionRefactoring())
        self.register(ChangeSignatureRefactoring())

    def register(self, refactoring: Refactoring):
        """注册重构操作"""
        self.refactorings[refactoring.name] = refactoring

    def get_available_refactorings(self, context: RefactoringContext) -> List[Refactoring]:
        """获取可用的重构操作"""
        return [r for r in self.refactorings.values() if r.is_applicable(context)]

    def apply_refactoring(self, name: str, context: RefactoringContext) -> RefactoringResult:
        """应用重构"""
        if name not in self.refactorings:
            raise UnknownRefactoringError(name)

        refactoring = self.refactorings[name]
        if not refactoring.is_applicable(context):
            raise RefactoringNotApplicableError(name)

        return refactoring.apply(context)
```

### 2.3 重命名重构

```python
# src/zhc/refactor/rename.py

class RenameRefactoring(Refactoring):
    """重命名重构"""

    name = "rename"
    description = "重命名变量、函数或类型"

    def __init__(self):
        self.old_name: str = ""
        self.new_name: str = ""
        self.scope: str = "file"  # "file", "module", "project"

    def is_applicable(self, context: RefactoringContext) -> bool:
        """检查是否可以重命名"""
        # 检查光标位置是否有标识符
        if context.cursor_position:
            line, col = context.cursor_position
            node = self._find_node_at_position(context.ast, line, col)
            return node is not None and isinstance(node, Identifier)
        return False

    def apply(self, context: RefactoringContext) -> RefactoringResult:
        """应用重命名"""
        edits = []
        affected_files = [context.source_file]
        warnings = []

        # 查找所有引用
        references = self._find_all_references(
            context.ast,
            context.symbol_table,
            self.old_name,
            self.scope
        )

        # 检查新名称是否冲突
        if self._has_name_conflict(context.symbol_table, self.new_name):
            warnings.append(f"名称 '{self.new_name}' 已存在，可能导致冲突")

        # 生成编辑
        for ref in references:
            edits.append(TextEdit(
                start_line=ref.line,
                start_column=ref.column,
                end_line=ref.line,
                end_column=ref.column + len(self.old_name),
                new_text=self.new_name,
                description=f"重命名 '{self.old_name}' 为 '{self.new_name}'"
            ))

        # 应用编辑生成预览
        preview = self._apply_edits(context.source_code, edits)

        return RefactoringResult(
            edits=edits,
            affected_files=affected_files,
            preview=preview,
            warnings=warnings
        )

    def _find_all_references(self, ast: ASTNode, symbols: SymbolTable,
                            name: str, scope: str) -> List[Reference]:
        """查找所有引用"""
        references = []

        # 从符号表获取定义位置
        definition = symbols.find_definition(name)
        if definition:
            references.append(Reference(
                file=definition.file,
                line=definition.line,
                column=definition.column,
                kind="definition"
            ))

        # 查找所有使用位置
        for node in ast.walk():
            if isinstance(node, Identifier) and node.name == name:
                references.append(Reference(
                    file=node.file,
                    line=node.line,
                    column=node.column,
                    kind="usage"
                ))

        return references

    def _has_name_conflict(self, symbols: SymbolTable, name: str) -> bool:
        """检查名称冲突"""
        return symbols.find_definition(name) is not None
```

### 2.4 提取函数重构

```python
# src/zhc/refactor/extract_function.py

class ExtractFunctionRefactoring(Refactoring):
    """提取函数重构"""

    name = "extract-function"
    description = "将选中的代码片段提取为新函数"

    def __init__(self):
        self.function_name: str = ""
        self.start_line: int = 0
        self.end_line: int = 0

    def is_applicable(self, context: RefactoringContext) -> bool:
        """检查是否可以提取函数"""
        # 检查是否选中了有效的代码块
        if self.start_line <= 0 or self.end_line <= 0:
            return False

        # 检查选中的代码是否是完整的语句
        statements = self._get_selected_statements(context)
        return len(statements) > 0

    def apply(self, context: RefactoringContext) -> RefactoringResult:
        """应用提取函数"""
        edits = []

        # 1. 分析选中的代码
        statements = self._get_selected_statements(context)
        local_vars = self._find_local_variables(statements)
        used_vars = self._find_used_variables(statements)
        assigned_vars = self._find_assigned_variables(statements)

        # 2. 确定参数和返回值
        parameters = used_vars - local_vars
        return_values = assigned_vars & used_vars

        # 3. 生成新函数
        new_function = self._generate_function(
            self.function_name,
            parameters,
            return_values,
            statements
        )

        # 4. 生成调用代码
        call_code = self._generate_call(
            self.function_name,
            parameters,
            return_values
        )

        # 5. 添加新函数定义
        insert_pos = self._find_insert_position(context)
        edits.append(TextEdit(
            start_line=insert_pos.line,
            start_column=1,
            end_line=insert_pos.line,
            end_column=1,
            new_text="\n" + new_function + "\n",
            description=f"添加新函数 '{self.function_name}'"
        ))

        # 6. 替换原代码为函数调用
        edits.append(TextEdit(
            start_line=self.start_line,
            start_column=1,
            end_line=self.end_line + 1,
            end_column=1,
            new_text=call_code,
            description=f"替换为函数调用 '{self.function_name}'"
        ))

        preview = self._apply_edits(context.source_code, edits)

        return RefactoringResult(
            edits=edits,
            affected_files=[context.source_file],
            preview=preview
        )

    def _generate_function(self, name: str, params: Set[str],
                          returns: Set[str], statements: List[Statement]) -> str:
        """生成新函数代码"""
        lines = []

        # 函数签名
        param_list = ", ".join(sorted(params))
        lines.append(f"函数 {name}({param_list})")

        # 函数体
        for stmt in statements:
            lines.append(f"    {stmt.to_code()}")

        # 返回值
        if returns:
            return_vars = ", ".join(sorted(returns))
            lines.append(f"    返回 ({return_vars})")

        lines.append("结束")

        return "\n".join(lines)

    def _generate_call(self, name: str, params: Set[str],
                       returns: Set[str]) -> str:
        """生成函数调用代码"""
        args = ", ".join(sorted(params))

        if returns:
            result_vars = ", ".join(sorted(returns))
            return f"    ({result_vars}) = {name}({args})"
        else:
            return f"    {name}({args})"
```

### 2.5 内联变量重构

```python
# src/zhc/refactor/inline_variable.py

class InlineVariableRefactoring(Refactoring):
    """内联变量重构"""

    name = "inline-variable"
    description = "内联局部变量，用其初始化表达式替换所有使用"

    def __init__(self):
        self.variable_name: str = ""

    def is_applicable(self, context: RefactoringContext) -> bool:
        """检查是否可以内联"""
        # 检查变量是否存在
        var = context.symbol_table.find_variable(self.variable_name)
        if var is None:
            return False

        # 检查是否是局部变量
        if not var.is_local:
            return False

        # 检查是否有初始化表达式
        return var.initial_value is not None

    def apply(self, context: RefactoringContext) -> RefactoringResult:
        """应用内联"""
        edits = []

        # 获取变量定义
        var = context.symbol_table.find_variable(self.variable_name)
        expr = var.initial_value

        # 查找所有使用位置
        uses = self._find_uses(context.ast, self.variable_name)

        # 替换所有使用
        for use in uses:
            edits.append(TextEdit(
                start_line=use.line,
                start_column=use.column,
                end_line=use.line,
                end_column=use.column + len(self.variable_name),
                new_text=expr.to_code(),
                description=f"内联变量 '{self.variable_name}'"
            ))

        # 删除变量声明
        edits.append(TextEdit(
            start_line=var.line,
            start_column=1,
            end_line=var.line + 1,
            end_column=1,
            new_text="",
            description=f"删除变量声明 '{self.variable_name}'"
        ))

        preview = self._apply_edits(context.source_code, edits)

        return RefactoringResult(
            edits=edits,
            affected_files=[context.source_file],
            preview=preview
        )
```

### 2.6 自动重构建议

```python
# src/zhc/refactor/suggestions.py

class RefactoringSuggester:
    """重构建议器"""

    def __init__(self):
        self.suggesters = [
            LongFunctionSuggester(),
            DuplicateCodeSuggester(),
            ComplexConditionSuggester(),
            UnusedVariableSuggester(),
            MagicNumberSuggester(),
        ]

    def suggest(self, context: RefactoringContext) -> List[RefactoringSuggestion]:
        """生成重构建议"""
        suggestions = []

        for suggester in self.suggesters:
            suggestions.extend(suggester.suggest(context))

        return suggestions


class LongFunctionSuggester:
    """长函数建议器"""

    def suggest(self, context: RefactoringContext) -> List[RefactoringSuggestion]:
        suggestions = []

        for func in context.symbol_table.functions:
            lines = func.end_line - func.start_line + 1
            if lines > 50:
                suggestions.append(RefactoringSuggestion(
                    kind="extract-function",
                    message=f"函数 '{func.name}' 过长 ({lines} 行)，建议拆分",
                    location=(func.start_line, 1),
                    priority=1,
                    auto_fixable=True
                ))

        return suggestions


class DuplicateCodeSuggester:
    """重复代码建议器"""

    def suggest(self, context: RefactoringContext) -> List[RefactoringSuggestion]:
        suggestions = []

        # 使用 AST 哈希检测重复代码
        code_blocks = self._hash_code_blocks(context.ast)

        for hash_val, blocks in code_blocks.items():
            if len(blocks) > 1:
                suggestions.append(RefactoringSuggestion(
                    kind="extract-function",
                    message=f"发现 {len(blocks)} 处重复代码块，建议提取为函数",
                    location=blocks[0].location,
                    priority=2,
                    auto_fixable=False
                ))

        return suggestions
```
