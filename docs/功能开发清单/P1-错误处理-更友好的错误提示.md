## 1. 需求分析

### 1.1 当前状态

ZhC 编译器的错误提示已包含：
- 错误消息
- 错误代码
- 基本位置信息
- 初步的修复建议

**现有问题**：
1. 错误消息不够详细，缺少上下文解释
2. 缺少"常见的错误原因"提示
3. 类型错误缺少期望类型和实际类型对比
4. 未提供类似错误的参考
5. 缺少错误链追踪（如错误的根本原因）

### 1.2 目标状态

实现智能、友好的错误提示：

```
错误[E001]: 类型不匹配
   --> test.zhc:15:8
   |
15 |     整数型 结果 = "hello" + 1;
   |            ^^^^^^
   |
   = 提示: 运算符 '+' 的左操作数类型为 '字符串型'
   = 提示: 但期望的类型是 '整数型' 或 '浮点型'
   |
   = 常见错误: 你可能想把字符串转换为数字？
   |   尝试: 使用 字符串转整数(字符串变量) 将字符串转换为整数
   |   示例: 整数型 结果 = 字符串转整数("42") + 1;
   |
   = 参考: 查阅文档 '类型转换' 章节了解更多

帮助: 使用 'zhc --explain E001' 查看详细说明
```

#### 2.2.2 SmartHintGenerator（新增）

```python
class SmartHintGenerator:
    """智能提示生成器"""
    
    def __init__(self, type_system, symbol_table):
        self.type_system = type_system
        self.symbol_table = symbol_table
    
    def generate_hints(
        self,
        error: ZHCError,
        context: ErrorContext
    ) -> List[Hint]:
        """根据错误上下文生成智能提示"""
        hints = []
        
        # 1. 添加期望/实际类型对比
        if error.error_code == "E001":  # 类型不匹配
            hints.extend(self._generate_type_hints(error, context))
        
        # 2. 添加常见错误提示
        if error.error_code in ERROR_CODE_REGISTRY:
            hints.extend(self._generate_common_cause_hints(error))
        
        # 3. 添加相似符号提示
        if error.error_code == "E002":  # 未定义符号
            hints.extend(self._generate_similar_symbols_hint(error, context))
        
        # 4. 添加文档链接
        hints.extend(self._generate_documentation_hint(error))
        
        return hints
    
    def _generate_type_hints(
        self,
        error: ZHCError,
        context: ErrorContext
    ) -> List[Hint]:
        """生成类型相关提示"""
        hints = []
        
        # 获取期望类型和实际类型
        expected = context.expected_type
        actual = context.actual_type
        
        if expected and actual:
            hints.append(Hint(
                type="info",
                message=f"期望类型: '{expected}'",
                highlight=f"期望: {expected}"
            ))
            hints.append(Hint(
                type="info",
                message=f"实际类型: '{actual}'",
                highlight=f"实际: {actual}"
            ))
            
            # 生成类型转换建议
            conversion = self._find_conversion(actual, expected)
            if conversion:
                hints.append(Hint(
                    type="suggestion",
                    message=f"可以使用 '{conversion.function}' 进行转换",
                    code_example=conversion.example
                ))
        
        return hints
    
    def _generate_similar_symbols_hint(
        self,
        error: ZHCError,
        context: ErrorContext
    ) -> List[Hint]:
        """生成相似符号提示"""
        undefined_name = error.context.get("symbol_name")
        if not undefined_name:
            return []
        
        # 查找相似的已定义符号
        similar = self._find_similar_symbols(undefined_name)
        
        if similar:
            return [Hint(
                type="info",
                message=f"你是否指的是: {', '.join(similar[:3])}?",
                similar_symbols=similar[:3]
            )]
        
        return []
    
    def _find_similar_symbols(self, name: str) -> List[str]:
        """使用编辑距离查找相似符号"""
        from difflib import SequenceMatcher
        
        all_symbols = self.symbol_table.get_all_symbol_names()
        scored = [
            (s, SequenceMatcher(None, name, s).ratio())
            for s in all_symbols
        ]
        
        # 返回相似度 > 0.6 的符号
        return [s for s, score in scored if score > 0.6][:3]
```

#### 2.2.3 ErrorExplainer（新增）

```python
class ErrorExplainer:
    """错误解释器"""
    
    def explain(
        self,
        error: ZHCError,
        verbose: bool = False
    ) -> str:
        """生成错误详细解释"""
        
        lines = []
        
        # 基础解释
        lines.append(f"错误 {error.error_code}: {error.message}")
        
        if verbose:
            # 获取错误定义
            definition = ERROR_CODE_REGISTRY.get(error.error_code)
            
            if definition:
                lines.append("")
                lines.append("=" * 50)
                lines.append("详细说明")
                lines.append("=" * 50)
                lines.append(f"类别: {definition.category}")
                lines.append(f"严重程度: {definition.severity}")
                lines.append("")
                lines.append("可能的原因:")
                for i, cause in enumerate(definition.common_causes, 1):
                    lines.append(f"  {i}. {cause}")
        
        return "\n".join(lines)
```

```python
class ErrorMessageTemplate:
    """错误消息模板"""
    
    TEMPLATES = {
        "E001_TYPE_MISMATCH": {
            "brief": "类型不匹配: '{actual}' 不能用于 '{context}'",
            "detailed": (
                "运算符 '{operator}' 的操作数类型不匹配\n"
                "  期望: {expected}\n"
                "  实际: {actual}\n"
                "  位置: {location}"
            ),
            "hint": "考虑进行类型转换"
        },
        "E002_UNDEFINED_SYMBOL": {
            "brief": "未定义的符号: '{symbol}'",
            "detailed": (
                "符号 '{symbol}' 在当前作用域中未定义\n"
                "  行: {line}\n"
                "  列: {column}"
            ),
            "hint": "检查符号名称是否正确，或添加声明"
        },
        "E003_DUPLICATE_DECLARATION": {
            "brief": "重复声明: '{symbol}' 已在此作用域声明",
            "detailed": (
                "符号 '{symbol}' 在 {first_location} 首次声明\n"
                "  又在 {second_location} 重复声明"
            ),
            "hint": "使用不同的名称，或删除重复声明"
        }
    }
    
    @classmethod
    def format(cls, code: str, **kwargs) -> str:
        """格式化错误消息"""
        template = cls.TEMPLATES.get(code, {})
        brief = template.get("brief", "未知错误")
        
        try:
            return brief.format(**kwargs)
        except KeyError:
            return brief
```
