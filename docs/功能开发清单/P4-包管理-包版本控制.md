## 1. 需求分析

### 1.1 功能目标

实现完整的包版本控制系统：

1. **语义化版本**: 遵循 SemVer 规范
2. **版本发布**: 支持版本发布流程
3. **版本标签**: Git 标签管理
4. **变更日志**: 自动生成 CHANGELOG
5. **版本验证**: 版本号合法性检查

**示例使用**:
```bash
# 初始化版本
zhc version init

# 查看当前版本
zhc version
# 输出: 1.2.3

# 升级版本
zhc version major    # 1.2.3 → 2.0.0
zhc version minor    # 1.2.3 → 1.3.0
zhc version patch    # 1.2.3 → 1.2.4

# 发布版本
zhc version release  # 创建 Git 标签，生成 CHANGELOG

# 预发布版本
zhc version prerelease --type beta  # 1.2.3 → 1.2.4-beta.0
```

### 1.2 语义化版本规范

根据 [SemVer 2.0.0](https://semver.org/)：

| 版本格式 | 说明 | 示例 |
|---------|------|------|
| MAJOR.MINOR.PATCH | 基本版本 | 1.2.3 |
| MAJOR.MINOR.PATCH-PRERELEASE | 预发布版本 | 1.2.3-beta.1 |
| MAJOR.MINOR.PATCH+BUILD | 构建元数据 | 1.2.3+build.123 |

**版本升级规则**:
- **MAJOR**: 不兼容的 API 变更
- **MINOR**: 向后兼容的功能新增
- **PATCH**: 向后兼容的问题修复

**预发布标识**:
- `alpha`: 内部测试版本
- `beta`: 公开测试版本
- `rc`: 候选发布版本

### 1.3 边界情况

| 边界情况 | 处理方式 |
|---------|---------|
| 版本号回退 | 报错，不允许回退 |
| 跳过版本号 | 警告，允许但不推荐 |
| 预发布版本排序 | alpha < beta < rc < 正式版 |
| 构建元数据 | 不参与版本比较 |
| 多个预发布标识 | 只允许一个 |

```python
# src/package/version.py

"""
语义化版本管理（扩展）

支持：
- 基本版本号
- 预发布版本
- 构建元数据
"""

import re
from typing import Optional, Tuple
from dataclasses import dataclass
from enum import Enum

class PrereleaseType(Enum):
    """预发布类型"""
    ALPHA = "alpha"
    BETA = "beta"
    RC = "rc"

@dataclass
class Prerelease:
    """预发布信息"""
    type: PrereleaseType
    number: int = 0
    
    def __str__(self) -> str:
        return f"{self.type.value}.{self.number}"
    
    def __lt__(self, other: 'Prerelease') -> bool:
        # alpha < beta < rc
        type_order = {
            PrereleaseType.ALPHA: 0,
            PrereleaseType.BETA: 1,
            PrereleaseType.RC: 2
        }
        
        if self.type != other.type:
            return type_order[self.type] < type_order[other.type]
        
        return self.number < other.number

@dataclass
class Version:
    """语义化版本（扩展版）"""
    major: int
    minor: int
    patch: int
    prerelease: Optional[Prerelease] = None
    build: Optional[str] = None
    
    @classmethod
    def parse(cls, version_str: str) -> 'Version':
        """解析版本字符串
        
        支持格式：
        - 1.2.3
        - 1.2.3-alpha.1
        - 1.2.3+build.123
        - 1.2.3-beta.2+build.456
        """
        # 正则匹配
        pattern = r'^(\d+)\.(\d+)\.(\d+)(?:-([a-z]+)\.(\d+))?(?:\+(.+))?$'
        match = re.match(pattern, version_str)
        
        if not match:
            raise ValueError(f"无效的版本格式: {version_str}")
        
        major = int(match.group(1))
        minor = int(match.group(2))
        patch = int(match.group(3))
        
        prerelease = None
        if match.group(4):
            prerelease_type = PrereleaseType(match.group(4))
            prerelease_num = int(match.group(5))
            prerelease = Prerelease(prerelease_type, prerelease_num)
        
        build = match.group(6)
        
        return cls(major, minor, patch, prerelease, build)
    
    def __str__(self) -> str:
        base = f"{self.major}.{self.minor}.{self.patch}"
        
        if self.prerelease:
            base += f"-{self.prerelease}"
        
        if self.build:
            base += f"+{self.build}"
        
        return base
    
    def __lt__(self, other: 'Version') -> bool:
        # 比较主版本号
        if self.major != other.major:
            return self.major < other.major
        
        # 比较次版本号
        if self.minor != other.minor:
            return self.minor < other.minor
        
        # 比较补丁版本号
        if self.patch != other.patch:
            return self.patch < other.patch
        
        # 正式版 > 预发布版
        if self.prerelease is None and other.prerelease is None:
            return False
        if self.prerelease is None:
            return False  # 正式版更大
        if other.prerelease is None:
            return True   # 预发布版更小
        
        # 比较预发布版本
        return self.prerelease < other.prerelease
    
    def bump_major(self) -> 'Version':
        """升级主版本号"""
        return Version(self.major + 1, 0, 0)
    
    def bump_minor(self) -> 'Version':
        """升级次版本号"""
        return Version(self.major, self.minor + 1, 0)
    
    def bump_patch(self) -> 'Version':
        """升级补丁版本号"""
        return Version(self.major, self.minor, self.patch + 1)
    
    def bump_prerelease(self, prerelease_type: PrereleaseType) -> 'Version':
        """升级预发布版本"""
        if self.prerelease and self.prerelease.type == prerelease_type:
            # 同类型预发布，增加编号
            new_prerelease = Prerelease(
                prerelease_type,
                self.prerelease.number + 1
            )
        else:
            # 新类型预发布，从 0 开始
            new_prerelease = Prerelease(prerelease_type, 0)
        
        return Version(
            self.major,
            self.minor,
            self.patch,
            new_prerelease
        )
    
    def to_release(self) -> 'Version':
        """转换为正式版本"""
        return Version(self.major, self.minor, self.patch)
```

#### 步骤2：实现版本控制管理器

```python
# src/package/version_control.py

"""
版本控制管理器

提供版本升级、发布、验证等功能
"""

from typing import Optional, List
from pathlib import Path
import subprocess

class VersionControl:
    """版本控制管理器"""
    
    def __init__(self, project_root: Path):
        self.project_root = project_root
        self.config_path = project_root / 'zhc.json'
        self.changelog_path = project_root / 'CHANGELOG.md'
        
        # 加载当前版本
        self.current_version = self._load_version()
    
    def _load_version(self) -> Version:
        """从配置文件加载版本"""
        import json
        
        with open(self.config_path, 'r', encoding='utf-8') as f:
            data = json.load(f)
        
        return Version.parse(data['version'])
    
    def _save_version(self, version: Version):
        """保存版本到配置文件"""
        import json
        
        with open(self.config_path, 'r', encoding='utf-8') as f:
            data = json.load(f)
        
        data['version'] = str(version)
        
        with open(self.config_path, 'w', encoding='utf-8') as f:
            json.dump(data, f, indent=2, ensure_ascii=False)
        
        self.current_version = version
    
    def bump(
        self,
        level: str,
        prerelease_type: Optional[PrereleaseType] = None
    ) -> Version:
        """升级版本
        
        Args:
            level: 升级级别 (major, minor, patch, prerelease)
            prerelease_type: 预发布类型（仅 prerelease 级别需要）
            
        Returns:
            新版本号
        """
        if level == 'major':
            new_version = self.current_version.bump_major()
        elif level == 'minor':
            new_version = self.current_version.bump_minor()
        elif level == 'patch':
            new_version = self.current_version.bump_patch()
        elif level == 'prerelease':
            if not prerelease_type:
                raise ValueError("预发布版本需要指定类型")
            new_version = self.current_version.bump_prerelease(prerelease_type)
        else:
            raise ValueError(f"未知的升级级别: {level}")
        
        # 验证版本升级
        if new_version <= self.current_version:
            raise VersionError(
                f"版本号不能回退: {self.current_version} → {new_version}"
            )
        
        # 保存新版本
        self._save_version(new_version)
        
        return new_version
    
    def release(
        self,
        message: Optional[str] = None,
        push: bool = False
    ) -> str:
        """发布版本
        
        Args:
            message: 发布消息
            push: 是否推送到远程仓库
            
        Returns:
            Git 标签名
        """
        version = self.current_version
        
        # 如果是预发布版本，先转换为正式版本
        if version.prerelease:
            version = version.to_release()
            self._save_version(version)
        
        # 创建 Git 标签
        tag_name = f"v{version}"
        
        # 检查标签是否已存在
        if self._tag_exists(tag_name):
            raise VersionError(f"标签 {tag_name} 已存在")
        
        # 生成 CHANGELOG
        self._generate_changelog(version, message)
        
        # 提交变更
        self._git_commit(f"chore: 发布版本 {version}")
        
        # 创建标签
        self._git_tag(tag_name, message or f"Release {version}")
        
        # 推送
        if push:
            self._git_push(tag_name)
        
        return tag_name
    
    def validate_version(self, version_str: str) -> bool:
        """验证版本号格式"""
        try:
            version = Version.parse(version_str)
            return True
        except ValueError:
            return False
    
    def compare_versions(self, v1: str, v2: str) -> int:
        """比较两个版本
        
        Returns:
            -1: v1 < v2
            0: v1 == v2
            1: v1 > v2
        """
        version1 = Version.parse(v1)
        version2 = Version.parse(v2)
        
        if version1 < version2:
            return -1
        elif version1 > version2:
            return 1
        else:
            return 0
    
    def get_version_history(self) -> List[Version]:
        """获取版本历史"""
        tags = self._get_git_tags()
        
        versions = []
        for tag in tags:
            if tag.startswith('v'):
                try:
                    version = Version.parse(tag[1:])
                    versions.append(version)
                except ValueError:
                    continue
        
        return sorted(versions, reverse=True)
    
    def _generate_changelog(
        self,
        version: Version,
        message: Optional[str] = None
    ):
        """生成 CHANGELOG"""
        from .changelog import ChangelogGenerator
        
        generator = ChangelogGenerator(self.project_root)
        generator.generate(version, message)
    
    # Git 操作
    def _git_commit(self, message: str):
        """Git 提交"""
        subprocess.run(
            ['git', 'add', '.'],
            cwd=self.project_root,
            check=True
        )
        subprocess.run(
            ['git', 'commit', '-m', message],
            cwd=self.project_root,
            check=True
        )
    
    def _git_tag(self, tag_name: str, message: str):
        """创建 Git 标签"""
        subprocess.run(
            ['git', 'tag', '-a', tag_name, '-m', message],
            cwd=self.project_root,
            check=True
        )
    
    def _git_push(self, tag_name: str):
        """推送代码和标签"""
        subprocess.run(
            ['git', 'push'],
            cwd=self.project_root,
            check=True
        )
        subprocess.run(
            ['git', 'push', 'origin', tag_name],
            cwd=self.project_root,
            check=True
        )
    
    def _tag_exists(self, tag_name: str) -> bool:
        """检查标签是否存在"""
        result = subprocess.run(
            ['git', 'tag', '-l', tag_name],
            cwd=self.project_root,
            capture_output=True,
            text=True
        )
        return tag_name in result.stdout
    
    def _get_git_tags(self) -> List[str]:
        """获取所有 Git 标签"""
        result = subprocess.run(
            ['git', 'tag', '-l'],
            cwd=self.project_root,
            capture_output=True,
            text=True
        )
        return result.stdout.strip().split('\n')
```

#### 步骤3：实现 CHANGELOG 生成器

```python
# src/package/changelog.py

"""
CHANGELOG 自动生成器

根据 Git 提交记录生成变更日志
"""

from typing import List, Dict, Optional
from pathlib import Path
from datetime import datetime
import subprocess
from dataclasses import dataclass

@dataclass
class Commit:
    """Git 提交"""
    hash: str
    message: str
    author: str
    date: datetime

class ChangelogGenerator:
    """CHANGELOG 生成器"""
    
    # 提交类型映射
    TYPE_MAP = {
        'feat': '新功能',
        'fix': '问题修复',
        'docs': '文档更新',
        'style': '代码格式',
        'refactor': '代码重构',
        'perf': '性能优化',
        'test': '测试相关',
        'chore': '构建/工具',
        'ci': 'CI/CD',
    }
    
    def __init__(self, project_root: Path):
        self.project_root = project_root
        self.changelog_path = project_root / 'CHANGELOG.md'
    
    def generate(
        self,
        version: Version,
        message: Optional[str] = None
    ):
        """生成 CHANGELOG"""
        
        # 获取上一个版本的标签
        last_tag = self._get_last_tag()
        
        # 获取提交记录
        commits = self._get_commits(last_tag)
        
        # 分类提交
        categorized = self._categorize_commits(commits)
        
        # 生成内容
        content = self._format_changelog(version, categorized, message)
        
        # 写入文件
        self._write_changelog(content)
    
    def _get_last_tag(self) -> Optional[str]:
        """获取上一个版本标签"""
        result = subprocess.run(
            ['git', 'describe', '--tags', '--abbrev=0', 'HEAD^'],
            cwd=self.project_root,
            capture_output=True,
            text=True
        )
        
        if result.returncode == 0:
            return result.stdout.strip()
        
        return None
    
    def _get_commits(self, since_tag: Optional[str]) -> List[Commit]:
        """获取提交记录"""
        if since_tag:
            cmd = ['git', 'log', f'{since_tag}..HEAD', '--pretty=format:%H|%s|%an|%ai']
        else:
            cmd = ['git', 'log', '--pretty=format:%H|%s|%an|%ai']
        
        result = subprocess.run(
            cmd,
            cwd=self.project_root,
            capture_output=True,
            text=True
        )
        
        commits = []
        for line in result.stdout.strip().split('\n'):
            if not line:
                continue
            
            parts = line.split('|')
            if len(parts) >= 4:
                commits.append(Commit(
                    hash=parts[0],
                    message=parts[1],
                    author=parts[2],
                    date=datetime.fromisoformat(parts[3])
                ))
        
        return commits
    
    def _categorize_commits(
        self,
        commits: List[Commit]
    ) -> Dict[str, List[Commit]]:
        """分类提交记录"""
        categorized = {}
        
        for commit in commits:
            # 提取类型（格式: type: message）
            if ':' in commit.message:
                commit_type = commit.message.split(':')[0].strip()
                
                # 提取作用域（格式: type(scope): message）
                if '(' in commit_type:
                    commit_type = commit_type.split('(')[0]
                
                if commit_type in self.TYPE_MAP:
                    category = self.TYPE_MAP[commit_type]
                else:
                    category = '其他'
            else:
                category = '其他'
            
            if category not in categorized:
                categorized[category] = []
            
            categorized[category].append(commit)
        
        return categorized
    
    def _format_changelog(
        self,
        version: Version,
        categorized: Dict[str, List[Commit]],
        message: Optional[str]
    ) -> str:
        """格式化 CHANGELOG"""
        lines = []
        
        # 版本标题
        date_str = datetime.now().strftime('%Y-%m-%d')
        lines.append(f"## [{version}] - {date_str}")
        lines.append("")
        
        # 发布说明
        if message:
            lines.append(message)
            lines.append("")
        
        # 分类变更
        for category, commits in categorized.items():
            if not commits:
                continue
            
            lines.append(f"### {category}")
            lines.append("")
            
            for commit in commits:
                # 提取消息（移除类型前缀）
                msg = commit.message
                if ':' in msg:
                    msg = msg.split(':', 1)[1].strip()
                
                lines.append(f"- {msg} ([{commit.hash[:7]}])")
            
            lines.append("")
        
        return '\n'.join(lines)
    
    def _write_changelog(self, content: str):
        """写入 CHANGELOG 文件"""
        if self.changelog_path.exists():
            # 读取现有内容
            with open(self.changelog_path, 'r', encoding='utf-8') as f:
                existing = f.read()
            
            # 插入新内容（在第一个版本标题之前）
            if '## [' in existing:
                parts = existing.split('## [', 1)
                new_content = parts[0] + content + '\n## [' + parts[1]
            else:
                new_content = existing + '\n\n' + content
        else:
            # 创建新文件
            header = "# CHANGELOG\n\n所有重要的变更都将记录在此文件中。\n\n"
            new_content = header + content
        
        with open(self.changelog_path, 'w', encoding='utf-8') as f:
            f.write(new_content)
```
